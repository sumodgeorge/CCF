16 // This is responsible for managing a collection of strategies, and ensuring
17 // each has been given every transaction up to the commit point, in-order.
18 class Indexer : public IndexingStrategies
19 {
20 public:
21 static constexpr size_t MAX_REQUESTABLE = 500;
22
23 protected:
24 std::shared_ptr<TransactionFetcher> transaction_fetcher;
25
26 using PendingTx = std::pair<ccf::TxID, std::vector<uint8_t>>;
27 std::vector<PendingTx> uncommitted_entries;
28
29 ccf::TxID committed = {};

31 static bool tx_id_less(const ccf::TxID& a, const ccf::TxID& b)
32 {
33 // NB: This will return true for 2.10 < 4.5, which isn't necessarily
34 // what's wanted in all comparisons (that's why this isn't implemented for
35 // ccf::TxID directly). But it's fine for everywhere we use it here, since
36 // we assume we only see a node's transitions in-order.
37 return a.view < b.view || a.seqno < b.seqno;
38 }
39
40 static bool uncommitted_entries_cmp(const ccf::TxID& a, const PendingTx& b)
41 {
42 return tx_id_less(a, b.first);
43 }
44
45 void update_commit(const ccf::TxID& tx_id)
46 {
47 if (tx_id_less(tx_id, committed))
48 {
49 throw std::logic_error(fmt::format(
50 "Committing out-of-order. Committed to {}, trying to commit {}",
51 committed.to_str(),
52 tx_id.to_str()));
53 }
54
55 committed = tx_id;
56 }
57
58 public:
59 Indexer(const std::shared_ptr<TransactionFetcher>& tf) :
60 transaction_fetcher(tf)
61 {}
62
63 // Returns true if it looks like there's still a gap to fill. Useful for
64 // testing
65 bool update_strategies(
66 std::chrono::milliseconds elapsed, const ccf::TxID& newly_committed)
67 {
68 update_commit(newly_committed);
69
70 std::optional<ccf::SeqNo> min_requested = std::nullopt;
71 for (auto& strategy : strategies)
72 {
73 strategy->tick();
74
75 const auto next_requested = strategy->next_requested();
76 if (!next_requested.has_value())
77 {
78 // If this strategy does not want any more Txs, don't consider
79 // advancing it any further If this strategy has an upper-bound on Txs
80 // it cares about, and we've already provided that, don't consider
81 // advancing it any further
82 continue;
83 }
84
85 if (!min_requested.has_value() || *next_requested < *min_requested)
86 {
87 min_requested = next_requested;
88 }
89 }
90
91 if (min_requested.has_value())
92 {
93 if (*min_requested <= committed.seqno)
94 {
95 // Request a prefix of the missing entries. Cap the requested range,
96 // so we don't overload the node with a huge historical request
97 const auto first_requested = *min_requested;
98 auto additional = std::min(
99 MAX_REQUESTABLE - uncommitted_entries.size(),
100 committed.seqno - first_requested);
101
102 SeqNoCollection seqnos;
103 for (auto i = first_requested; i <= first_requested + additional; ++i)
104 {
105 seqnos.insert(i);
106 }
107
108 auto stores = transaction_fetcher->fetch_transactions(seqnos);
109 for (auto& store : stores)
110 {
111 const ccf::TxID tx_id = store->get_txid();
112
113 for (auto& strategy : strategies)
114 {
115 const auto next_requested = strategy->next_requested();
116 if (
117 next_requested.has_value() && (tx_id.seqno == *next_requested))
118 {
119 strategy->handle_committed_transaction(tx_id, store);
120 }
121 }
122 }
123
124 return true;
125 }
126 }
127
128 return false;
129 }
130 };

14 class ACMERpcEndpoints : public CommonEndpointRegistry
15 {
16 public:
17 ACMERpcEndpoints(
18 NetworkState& network, ccfapp::AbstractNodeContext& context) :
19 CommonEndpointRegistry(get_actor_prefix(ActorsType::well_known), context)
20 {
21 auto handler = [this](auto& ctx) {
22 http_status response_status = HTTP_STATUS_INTERNAL_SERVER_ERROR;
23 std::string response_body;
24
25 const auto& path_params = ctx.rpc_ctx->get_request_path_params();
26 const auto url_token_it = path_params.find("token");
27
28 if (url_token_it == path_params.end())
29 {
30 ctx.rpc_ctx->set_response_status(HTTP_STATUS_NOT_FOUND);
31 ctx.rpc_ctx->set_response_body("no token in URL");
32 }
33
34 std::string token = url_token_it->second;
35 LOG_DEBUG_FMT("ACME: challenge request for token '{}'", token);
36
37 auto tit = prepared_responses.find(token);
38 if (tit == prepared_responses.end())
39 {
40 auto prit = prepared_responses.find("");
41 if (prit != prepared_responses.end())
42 {
43 response_status = HTTP_STATUS_OK;
44 response_body = token + "." + prit->second;
45 }
46 else
47 {
48 response_status = HTTP_STATUS_NOT_FOUND;
49 response_body =
50 fmt::format("Challenge response for token '{}' not found", token);
51 }
52 }
53 else
54 {
55 response_status = HTTP_STATUS_OK;
56 response_body = token + "." + tit->second;
57 }
58
59 ctx.rpc_ctx->set_response_status(response_status);
60 ctx.rpc_ctx->set_response_body(std::move(response_body));
61 };
62
63 make_endpoint(
64 "/acme-challenge/{token}", HTTP_GET, handler, no_auth_required)
65 .set_forwarding_required(endpoints::ForwardingRequired::Never)
66 .set_auto_schema<void, std::string>()
67 .install();
68 }
69
70 virtual ~ACMERpcEndpoints() = default;
71
72 void add(const std::string& token, const std::string& response)
73 {
74 LOG_TRACE_FMT(
75 "ACME: challenge server received response for token '{}' ({})",
76 token,
77 response);
78
79 prepared_responses.emplace(token, response);
80 }
81
82 void remove(const std::string& token)
83 {
84 prepared_responses.erase(token);
85 }
86
87 protected:
88 std::map<std::string, std::string> prepared_responses;
89 };

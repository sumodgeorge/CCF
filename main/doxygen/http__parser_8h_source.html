32 inline auto split_url_path(const std::string_view& url)
33 {
34 LOG_TRACE_FMT("Received url to parse: {}", std::string_view(url));
35
36 const auto path_end = url.find('?');
37 const auto query_start =
38 path_end == std::string::npos ? url.size() : path_end + 1;
39
40 const auto query_end = url.find('#', query_start);
41 const auto fragment_start =
42 query_end == std::string::npos ? url.size() : query_end + 1;
43
44 return std::make_tuple(
45 url.substr(0, path_end),
46 url.substr(query_start, query_end - query_start),
47 url.substr(fragment_start));
48 }

50 static std::string url_decode(const std::string_view& s_)
51 {
52 std::string s(s_);
53 char const* src = s.c_str();
54 char const* end = s.c_str() + s.size();
55 char* dst = s.data();
56
57 while (src < end)
58 {
59 char const c = *src++;
60 if (c == '%' && (src + 1) < end && isxdigit(src[0]) && isxdigit(src[1]))
61 {
62 const auto a = ds::hex_char_to_int(*src++);
63 const auto b = ds::hex_char_to_int(*src++);
64 *dst++ = (a << 4) | b;
65 }
66 else if (c == '+')
67 {
68 *dst++ = ' ';
69 }
70 else
71 {
72 *dst++ = c;
73 }
74 }
75
76 s.resize(dst - s.data());
77 return s;
78 }

154 inline URL parse_url_full(const std::string& url)
155 {
156 LOG_TRACE_FMT("Received url to parse: {}", url);
157
158 // From https://tools.ietf.org/html/rfc3986#appendix-B
159 std::regex url_regex(
160 "^(([^:/?#]+):)?(//([^/?#]*))?([^?#]*)(\\?([^#]*))?(#(.*))?");
161
162 std::smatch match;
163 if (!std::regex_match(url, match, url_regex))
164 {
165 throw std::invalid_argument(fmt::format("Unable to parse url: {}", url));
166 }
167
168 const auto host_port = match[4].str();
169
170 // IPv6 hosts may contain colons, so only search for port after the closing
171 // square bracket
172 const auto closing_bracket = host_port.rfind(']');
173 const auto port_delim_start =
174 closing_bracket == std::string::npos ? 0 : closing_bracket;
175 const auto port_delim = host_port.find(':', port_delim_start);
176
177 URL u;
178 u.scheme = match[2].str();
179 u.host = host_port.substr(0, port_delim);
180 if (port_delim != std::string::npos)
181 {
182 u.port = host_port.substr(port_delim + 1);
183 }
184 u.path = match[5].str();
185 u.query = match[7].str();
186 u.fragment = match[9].str();
187 return u;
188 }

229 void execute(const uint8_t* data, size_t size)
230 {
231 const auto data_char = (const char*)data;
232 auto err_no = llhttp_execute(&parser, data_char, size);
233
234 if (err_no == HPE_PAUSED_UPGRADE)
235 {
236 // Assume Upgrade requests will be handled by caller inspecting headers,
237 // so we can instantly resume the parser.
238 llhttp_resume_after_upgrade(&parser);
239 }
240 else if (err_no != HPE_OK)
241 {
242 throw std::runtime_error(fmt::format(
243 "HTTP parsing failed ({}: {}) around byte {}",
244 llhttp_errno_name(err_no),
245 llhttp_get_error_reason(&parser),
246 llhttp_get_error_pos(&parser) - data_char));
247 }
248 }

250 void append_body(const char* at, size_t length)
251 {
252 if (state == IN_MESSAGE)
253 {
254 LOG_TRACE_FMT("Appending chunk [{} bytes]", length);
255 std::copy(at, at + length, std::back_inserter(body_buf));
256
257 auto const& max_body_size =
258 configuration.max_body_size.value_or(default_max_body_size);
259 if (body_buf.size() > max_body_size)
260 {
261 throw RequestPayloadTooLarge(fmt::format(
262 "HTTP request body is too large (max size allowed: {})",
263 max_body_size));
264 }
265 }
266 else
267 {
268 throw std::runtime_error("Receiving content outside of message");
269 }
270 }

304 void header_field(const char* at, size_t length)
305 {
306 if (!partial_parsed_header.second.empty())
307 {
308 complete_header();
309 auto const& max_headers_count =
310 configuration.max_headers_count.value_or(default_max_headers_count);
311 if (headers.size() > max_headers_count)
312 {
313 throw RequestHeaderTooLarge(fmt::format(
314 "Too many headers (max number allowed: {})", max_headers_count));
315 }
316 }
317
318 // HTTP headers are stored lowercase as it is easier to verify HTTP
319 // signatures later on
320 auto f = std::string(at, length);
321 nonstd::to_lower(f);
322 auto& partial_header_key = partial_parsed_header.first;
323 partial_header_key.append(f);
324
325 auto const& max_header_size =
326 configuration.max_header_size.value_or(default_max_header_size);
327 if (partial_header_key.size() > max_header_size)
328 {
329 throw RequestHeaderTooLarge(fmt::format(
330 "Header key for '{}' is too large (max size allowed: {})",
331 partial_parsed_header.first,
332 max_header_size));
333 }
334 }
335
336 void header_value(const char* at, size_t length)
337 {
338 auto& partial_header_value = partial_parsed_header.second;
339 partial_header_value.append(at, length);
340 auto const& max_header_size =
341 configuration.max_header_size.value_or(default_max_header_size);
342 if (partial_header_value.size() > max_header_size)
343 {
344 throw RequestHeaderTooLarge(fmt::format(
345 "Header value for '{}' is too large (max size allowed: {})",
346 partial_parsed_header.first,
347 max_header_size));
348 }
349 }

354 }

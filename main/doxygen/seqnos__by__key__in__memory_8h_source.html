43 std::optional<SeqNoCollection> get_write_txs_in_range(
44 const typename M::Key& key,
45 ccf::SeqNo from,
46 ccf::SeqNo to,
47 std::optional<size_t> max_seqnos = std::nullopt)
48 {
49 if (to < from)
50 {
51 throw std::logic_error(
52 fmt::format("Range goes backwards: {} -> {}", from, to));
53 }
54
55 if (to > current_txid.seqno)
56 {
57 // If the requested range hasn't been populated yet, indicate
58 // that with nullopt
59 return std::nullopt;
60 }
61
62 return get_write_txs_impl(
63 M::KeySerialiser::to_serialised(key), from, to, max_seqnos);
64 }
65
66 std::optional<SeqNoCollection> get_all_write_txs(const typename M::Key& key)
67 {
68 return get_write_txs_in_range(key, 0, current_txid.seqno);
69 }

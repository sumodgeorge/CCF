9 // Unit serialisations are used as a utility type to convert kv::Maps to
10 // kv::Maps and kv::Sets. Specifically, these are implemented as wrappers so
11 // that kv::Value<T> is essentially kv::Map<Unit, T>, and kv::Set<T> is
12 // kv::Map<T, Unit>.
13 // This is used as a template parameter allowing the caller to specify what
14 // value is inserted into the ledger.
15
16 // This is the default UnitCreator, returning 8 null bytes for compatibility
17 // with old ledgers (where Values were previously Maps with a single entry
18 // at key 0, serialised as a uint64_t)
19 struct ZeroBlitUnitCreator
20 {
21 static kv::serialisers::SerialisedEntry get()
22 {
23 kv::serialisers::SerialisedEntry e;
24 e.assign(sizeof(uint64_t), 0u);
25 return e;
26 }
27 };

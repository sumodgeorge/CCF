70 template <class THandle>
71 THandle* get_handle_by_name(const std::string& map_name)
72 {
73 auto possible_handles = get_possible_handles(map_name);
74 for (auto handle : possible_handles)
75 {
76 auto typed_handle = dynamic_cast<THandle*>(handle);
77 if (typed_handle != nullptr)
78 {
79 return typed_handle;
80 }
81 }
82
83 auto it = all_changes.find(map_name);
84 if (it != all_changes.end())
85 {
86 auto& [abstract_map, change_set] = it->second;
87
88 auto typed_handle = new THandle(*change_set, map_name);
89 std::unique_ptr<AbstractHandle> abstract_handle(typed_handle);
90 retain_handle(map_name, std::move(abstract_handle));
91 return typed_handle;
92 }
93 else
94 {
95 auto [abstract_map, change_set] =
96 get_map_and_change_set_by_name(map_name);
97
98 if (change_set == nullptr)
99 {
100 compacted_version_conflict(map_name);
101 }
102
103 auto typed_handle = new THandle(*change_set, map_name);
104 std::unique_ptr<AbstractHandle> abstract_handle(typed_handle);
105 retain_handle(map_name, std::move(abstract_handle));
106 retain_change_set(map_name, std::move(change_set), abstract_map);
107 return typed_handle;
108 }
109 }

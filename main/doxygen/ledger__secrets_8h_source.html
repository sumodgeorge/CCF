27 // Set once when the LedgerSecrets are initialised. This prevents a backup
28 // node to rollback not-yet-applicable ledger secrets when catching up.
29 // All rollback that would result in the removal of some of these secrets
30 // would imply that the transaction that added the node itself was rolled
31 // back.
32 kv::Version initial_latest_ledger_secret_version = 0;
33
34 std::optional<LedgerSecretsMap::iterator> last_used_secret_it =
35 std::nullopt;

37 LedgerSecretPtr get_secret_for_version(
38 kv::Version version, bool historical_hint = false)
39 {
40 if (ledger_secrets.empty())
41 {
42 LOG_FAIL_FMT("Ledger secrets map is empty");
43 return nullptr;
44 }
45
46 if (!historical_hint && last_used_secret_it.has_value())
47 {
48 // Fast path for non-historical queries as both primary and backup nodes
49 // encrypt/decrypt transactions in order, it is sufficient to keep an
50 // iterator on the last used secret to access ledger secrets in constant
51 // time.
52 auto& last_used_secret_it_ = last_used_secret_it.value();
53 if (
54 std::next(last_used_secret_it_) != ledger_secrets.end() &&
55 version >= std::next(last_used_secret_it_)->first)
56 {
57 // Across a rekey, start using the next key
58 ++last_used_secret_it_;
59 }
60
61 return last_used_secret_it_->second;
62 }
63
64 // Slow path, e.g. for historical queries. The ledger secret used to
65 // encrypt/decrypt a transaction at a given version is the one with the
66 // highest version that is lower than the given version (e.g. if
67 // ledger_secrets contains two keys for version 0 and 10 then the key
68 // associated with version 0 is used for version [0..9] and version 10 for
69 // versions 10+)
70 auto search = std::upper_bound(
71 ledger_secrets.begin(),
72 ledger_secrets.end(),
73 version,
74 [](auto a, const auto& b) { return b.first > a; });
75
76 if (search == ledger_secrets.begin())
77 {
78 LOG_FAIL_FMT("Could not find ledger secret for seqno {}", version);
79 return nullptr;
80 }
81
82 if (!historical_hint)
83 {
84 // Only update the last secret iterator on non-historical queries so
85 // that the fast path is always preserved for transactions on the main
86 // store
87 last_used_secret_it = std::prev(search);
88 }
89
90 return std::prev(search)->second;
91 }

93 void take_dependency_on_secrets(kv::ReadOnlyTx& tx)
94 {
95 // Ledger secrets are not stored in the KV. Instead, they are
96 // cached in a unique LedgerSecrets instance that can be accessed
97 // without reading the KV. However, it is possible that the ledger
98 // secrets are updated (e.g. rekey tx) concurrently to their access by
99 // another tx. To prevent conflicts, accessing the ledger secrets
100 // require access to a tx object, which must take a dependency on the
101 // secrets table.
102 auto secrets = tx.ro<Secrets>(Tables::ENCRYPTED_LEDGER_SECRETS);
103 secrets->get();
104 }

128 void adjust_previous_secret_stored_version(kv::Version version)
129 {
130 // To be able to lookup the last active ledger secret before the service
131 // crashed, the ledger secret created after the public recovery is
132 // complete should point to the version at which the past ledger secret
133 // has just been written to the store. This can only be done once the
134 // private recovery is complete.

143 ledger_secrets.rbegin()->second->previous_secret_stored_version = version;

205 LedgerSecretsMap get(
206 kv::ReadOnlyTx& tx, std::optional<kv::Version> up_to = std::nullopt)
207 {
208 std::lock_guard<ccf::Mutex> guard(lock);
209
210 take_dependency_on_secrets(tx);
211
212 if (!up_to.has_value())
213 {
214 return ledger_secrets;
215 }
216
217 auto search = ledger_secrets.find(up_to.value());
218 if (search == ledger_secrets.end())
219 {
220 throw std::logic_error(
221 fmt::format("No ledger secrets at {}", up_to.has_value()));
222 }
223
224 return LedgerSecretsMap(ledger_secrets.begin(), ++search);
225 }

228 {
229 std::lock_guard<ccf::Mutex> guard(lock);
230
231 if (
232 !ledger_secrets.empty() && !restored_ledger_secrets.empty() &&
233 restored_ledger_secrets.rbegin()->first >=

235 {
236 throw std::logic_error(fmt::format(

279 void rollback(kv::Version version)
280 {
281 std::lock_guard<ccf::Mutex> guard(lock);
282 if (ledger_secrets.empty())
283 {
284 return;
285 }
286
287 if (version < ledger_secrets.begin()->first)
288 {
289 LOG_DEBUG_FMT(
290 "Cannot rollback ledger secrets at {}: first secret is at {}",
291 version,
292 ledger_secrets.begin()->first);
293 return;
294 }
295
296 while (ledger_secrets.size() > 1)
297 {
298 auto k = ledger_secrets.rbegin();
299 if (
300 k->first <= version ||
301 k->first <= initial_latest_ledger_secret_version)
302 {
303 break;
304 }
305
306 LOG_TRACE_FMT("Rollback ledger secrets at seqno {}", k->first);
307 ledger_secrets.erase(k->first);
308 }
309
310 // Invalidate last used ledger secret iterator. Next key usage will need
311 // to find the appropriate key on the slow path.
312 last_used_secret_it = std::nullopt;
313 }

35 // Most transactions produce a receipt constructed from a combination of 3
36 // digests. Note that transactions emitted by old code versions may not
37 // include a claims_digest or a commit_evidence_digest, but from 2.0 onwards
38 // every transaction will contain a (potentially default-zero'd) claims digest
39 // and a commit evidence digest.
40 class ProofReceipt : public Receipt
41 {
42 public:
43 struct Components
44 {
45 crypto::Sha256Hash write_set_digest;
46 std::string commit_evidence;
47 ccf::ClaimsDigest claims_digest;
48 };
49 Components leaf_components;
50
51 struct ProofStep
52 {
53 enum
54 {
55 Left,
56 Right
57 } direction;
58
59 crypto::Sha256Hash hash = {};
60
61 bool operator==(const ProofStep& other) const
62 {
63 return direction == other.direction && hash == other.hash;
64 }
65 };
66 using Proof = std::vector<ProofStep>;
67
68 // A merkle-tree path from the leaf digest to the signed root
69 Proof proof = {};
70
71 crypto::Sha256Hash calculate_root() override
72 {
73 auto current = get_leaf_digest();
74
75 for (const auto& element : proof)
76 {
77 if (element.direction == ProofStep::Left)
78 {
79 current = crypto::Sha256Hash(element.hash, current);
80 }
81 else
82 {
83 current = crypto::Sha256Hash(current, element.hash);
84 }
85 }
86
87 return current;
88 }
89
90 crypto::Sha256Hash get_leaf_digest()
91 {
92 crypto::Sha256Hash ce_dgst(leaf_components.commit_evidence);
93 if (!leaf_components.claims_digest.empty())
94 {
95 return crypto::Sha256Hash(
96 leaf_components.write_set_digest,
97 ce_dgst,
98 leaf_components.claims_digest.value());
99 }
100 else
101 {
102 return crypto::Sha256Hash(leaf_components.write_set_digest, ce_dgst);
103 }
104 }
105
106 bool is_signature_transaction() const override
107 {
108 return false;
109 }
110 };

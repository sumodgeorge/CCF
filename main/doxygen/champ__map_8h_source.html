22 static constexpr size_t index_mask_bits = 5;
23 static constexpr size_t index_mask = (1 << index_mask_bits) - 1;
24
25 using Hash = uint32_t;
26 static constexpr size_t hash_bits = sizeof(Hash) * 8;
27
28 using SmallIndex = uint8_t;
29 static constexpr size_t small_index_bits = sizeof(SmallIndex) * 8;
30 static_assert(small_index_bits > index_mask_bits);
31
32 static constexpr size_t collision_node_bits = hash_bits % index_mask_bits;
33 static_assert(collision_node_bits > 0);
34 static constexpr SmallIndex collision_depth = hash_bits / index_mask_bits;
35 static constexpr size_t collision_bins = 1 << collision_node_bits;
36
37 static constexpr SmallIndex mask(Hash hash, SmallIndex depth)
38 {
39 return (hash >> ((Hash)depth * index_mask_bits)) & index_mask;
40 }

45 class Bitmap

47 uint32_t _bits;
48
49 public:
50 constexpr Bitmap() : _bits(0) {}
51
52 constexpr Bitmap(uint32_t bits) : _bits(bits) {}
53
54 constexpr Bitmap operator&(const Bitmap& other) const
55 {
56 return Bitmap(_bits & other._bits);
57 }
58
59 constexpr SmallIndex pop() const
60 {
61 return __builtin_popcount(_bits);
62 }
63
64 constexpr Bitmap set(SmallIndex idx) const
65 {
66 return Bitmap(_bits | ((uint32_t)1 << idx));
67 }
68
69 constexpr Bitmap clear(SmallIndex idx) const
70 {
71 return Bitmap(_bits & ~((uint32_t)1 << idx));
72 }
73
74 constexpr bool check(SmallIndex idx) const
75 {
76 return (_bits & ((uint32_t)1 << idx)) != 0;
77 }

103 template <class K, class V, class H>
104 struct Collisions
105 {
106 std::array<std::vector<std::shared_ptr<Entry<K, V>>>, collision_bins> bins;
107
108 const V* getp(Hash hash, const K& k) const
109 {
110 const auto idx = mask(hash, collision_depth);
111 const auto& bin = bins[idx];
112 for (const auto& node : bin)
113 {
114 if (k == node->key)
115 return &node->value;
116 }
117 return nullptr;
118 }
119
120 size_t put_mut(Hash hash, const K& k, const V& v)
121 {
122 const auto idx = mask(hash, collision_depth);
123 auto& bin = bins[idx];
124 for (size_t i = 0; i < bin.size(); ++i)
125 {
126 const auto& entry = bin[i];
127 if (k == entry->key)
128 {
129 bin[i] = std::make_shared<Entry<K, V>>(k, v);
130 return map::get_size(k) + map::get_size(v);
131 }
132 }
133 bin.push_back(std::make_shared<Entry<K, V>>(k, v));
134 return 0;
135 }
136
137 size_t remove_mut(Hash hash, const K& k)
138 {
139 const auto idx = mask(hash, collision_depth);
140 auto& bin = bins[idx];
141 for (size_t i = 0; i < bin.size(); ++i)
142 {
143 const auto& entry = bin[i];
144 if (k == entry->key)
145 {
146 const auto diff =
147 map::get_size(entry->key) + map::get_size(entry->value);
148 bin.erase(bin.begin() + i);
149 return diff;
150 }
151 }
152 return 0;
153 }
154
155 template <class F>
156 bool foreach(F&& f) const
157 {
158 for (const auto& bin : bins)
159 {
160 for (const auto& entry : bin)
161 if (!f(entry->key, entry->value))
162 return false;
163 }
164 return true;
165 }
166 };

168 template <class K, class V, class H>
169 struct SubNodes
170 {
171 std::vector<Node<K, V, H>> nodes;
172 Bitmap node_map;
173 Bitmap data_map;
174
175 SubNodes() {}
176
177 SubNodes(std::vector<Node<K, V, H>>&& ns) : nodes(std::move(ns)) {}
178
179 SubNodes(std::vector<Node<K, V, H>>&& ns, Bitmap nm, Bitmap dm) :
180 nodes(std::move(ns)),
181 node_map(nm),
182 data_map(dm)
183 {}
184
185 SmallIndex compressed_idx(SmallIndex idx) const
186 {
187 if (!node_map.check(idx) && !data_map.check(idx))
188 return (SmallIndex)-1;
189
190 const auto mask = Bitmap(~((uint32_t)-1 << idx));
191 if (data_map.check(idx))
192 return (data_map & mask).pop();
193
194 return data_map.pop() + (node_map & mask).pop();
195 }
196
197 const V* getp(SmallIndex depth, Hash hash, const K& k) const
198 {
199 const auto idx = mask(hash, depth);
200 const auto c_idx = compressed_idx(idx);
201
202 if (c_idx == (SmallIndex)-1)
203 return nullptr;
204
205 if (data_map.check(idx))
206 return node_as<Entry<K, V>>(c_idx)->getp(k);
207
208 if (depth == (collision_depth - 1))
209 return node_as<Collisions<K, V, H>>(c_idx)->getp(hash, k);
210
211 return node_as<SubNodes<K, V, H>>(c_idx)->getp(depth + 1, hash, k);
212 }
213
214 size_t put_mut(SmallIndex depth, Hash hash, const K& k, const V& v)
215 {
216 const auto idx = mask(hash, depth);
217 auto c_idx = compressed_idx(idx);
218
219 if (c_idx == (SmallIndex)-1)
220 {
221 data_map = data_map.set(idx);
222 c_idx = compressed_idx(idx);
223 nodes.insert(
224 nodes.begin() + c_idx, std::make_shared<Entry<K, V>>(k, v));
225 return 0;
226 }
227
228 if (node_map.check(idx))
229 {
230 size_t insert;
231 if (depth < (collision_depth - 1))
232 {
233 auto sn = *node_as<SubNodes<K, V, H>>(c_idx);
234 insert = sn.put_mut(depth + 1, hash, k, v);
235 nodes[c_idx] = std::make_shared<SubNodes<K, V, H>>(std::move(sn));
236 }
237 else
238 {
239 auto sn = *node_as<Collisions<K, V, H>>(c_idx);
240 insert = sn.put_mut(hash, k, v);
241 nodes[c_idx] = std::make_shared<Collisions<K, V, H>>(std::move(sn));
242 }
243 return insert;
244 }
245
246 const auto& entry0 = node_as<Entry<K, V>>(c_idx);
247 if (k == entry0->key)
248 {
249 auto current_size = map::get_size_with_padding(entry0->key) +
250 map::get_size_with_padding(entry0->value);
251 nodes[c_idx] = std::make_shared<Entry<K, V>>(k, v);
252 return current_size;
253 }
254
255 if (depth < (collision_depth - 1))
256 {
257 const auto hash0 = H()(entry0->key);
258 const auto idx0 = mask(hash0, depth + 1);
259 auto sub_node =
260 SubNodes<K, V, H>({entry0}, Bitmap(0), Bitmap(0).set(idx0));
261 sub_node.put_mut(depth + 1, hash, k, v);
262
263 nodes.erase(nodes.begin() + c_idx);
264 data_map = data_map.clear(idx);
265 node_map = node_map.set(idx);
266 c_idx = compressed_idx(idx);
267 nodes.insert(
268 nodes.begin() + c_idx,
269 std::make_shared<SubNodes<K, V, H>>(std::move(sub_node)));
270 }
271 else
272 {
273 auto sub_node = Collisions<K, V, H>();
274 const auto hash0 = H()(entry0->key);
275 const auto idx0 = mask(hash0, collision_depth);
276 sub_node.bins[idx0].push_back(entry0);
277 const auto idx1 = mask(hash, collision_depth);
278 sub_node.bins[idx1].push_back(std::make_shared<Entry<K, V>>(k, v));
279
280 nodes.erase(nodes.begin() + c_idx);
281 data_map = data_map.clear(idx);
282 node_map = node_map.set(idx);
283 c_idx = compressed_idx(idx);
284 nodes.insert(
285 nodes.begin() + c_idx,
286 std::make_shared<Collisions<K, V, H>>(std::move(sub_node)));
287 }
288 return 0;
289 }

292 SmallIndex depth, Hash hash, const K& k, const V& v) const
293 {
294 auto node = *this;

300 size_t remove_mut(SmallIndex depth, Hash hash, const K& k)
301 {
302 const auto idx = mask(hash, depth);
303 const auto c_idx = compressed_idx(idx);
304
305 if (c_idx == (SmallIndex)-1)
306 return 0;
307
308 if (data_map.check(idx))
309 {
310 const auto& entry = node_as<Entry<K, V>>(c_idx);
311 if (entry->key != k)
312 return 0;
313
314 const auto diff = map::get_size_with_padding(entry->key) +
315 map::get_size_with_padding(entry->value);
316 nodes.erase(nodes.begin() + c_idx);
317 data_map = data_map.clear(idx);
318 return diff;
319 }
320
321 if (depth == (collision_depth - 1))
322 {
323 auto sn = *node_as<Collisions<K, V, H>>(c_idx);
324 const auto diff = sn.remove_mut(hash, k);
325 nodes[c_idx] = std::make_shared<Collisions<K, V, H>>(std::move(sn));
326 return diff;
327 }
328
329 auto sn = *node_as<SubNodes<K, V, H>>(c_idx);
330 const auto diff = sn.remove_mut(depth + 1, hash, k);
331 nodes[c_idx] = std::make_shared<SubNodes<K, V, H>>(std::move(sn));
332 return diff;
333 }

344 template <class F>
345 bool foreach(SmallIndex depth, F&& f) const
346 {
347 const auto entries = data_map.pop();
348 for (SmallIndex i = 0; i < entries; ++i)
349 {
350 const auto& entry = node_as<Entry<K, V>>(i);
351 if (!f(entry->key, entry->value))
352 return false;
353 }
354 for (size_t i = entries; i < nodes.size(); ++i)
355 {
356 if (depth == (collision_depth - 1))
357 {
358 if (!node_as<Collisions<K, V, H>>(i)->foreach(std::forward<F>(f)))
359 return false;
360 }
361 else
362 {
363 if (!node_as<SubNodes<K, V, H>>(i)->foreach(
364 depth + 1, std::forward<F>(f)))
365 return false;
366 }
367 }
368 return true;
369 }

468 template <class K, class V, class H>
469 class Snapshot
470 {
471 private:
472 const Map<K, V, H> map;
473
474 struct KVTuple
475 {
476 K* k;
477 Hash h_k;
478 V* v;
479
480 KVTuple(K* k_, Hash h_k_, V* v_) : k(k_), h_k(h_k_), v(v_) {}
481 };
482
483 public:
484 Snapshot(const Map<K, V, H>& map_) : map(map_) {}
485
486 size_t get_serialized_size()
487 {
488 return map.get_serialized_size();
489 }
490
491 void serialize(uint8_t* data)
492 {
493 std::vector<KVTuple> ordered_state;
494 ordered_state.reserve(map.size());
495 size_t size = 0;
496
497 map.foreach([&](auto& key, auto& value) {
498 K* k = &key;
499 V* v = &value;
500 uint32_t ks = map::get_size(key);
501 uint32_t vs = map::get_size(value);
502 uint32_t key_size = ks + map::get_padding(ks);
503 uint32_t value_size = vs + map::get_padding(vs);
504
505 size += (key_size + value_size);
506
507 ordered_state.emplace_back(k, static_cast<Hash>(H()(key)), v);
508
509 return true;
510 });
511
512 // Sort keys to be able to generate byte-for-byte serialised snapshot from
513 // the same state
514 std::sort(
515 ordered_state.begin(), ordered_state.end(), [](KVTuple& i, KVTuple& j) {
516 return i.h_k < j.h_k;
517 });
518
519 CCF_ASSERT_FMT(
520 size == map.get_serialized_size(),
521 "size:{}, map->size:{} ==> count:{}, vect:{}",
522 size,
523 map.get_serialized_size(),
524 map.size(),
525 ordered_state.size());
526
527 for (const auto& p : ordered_state)
528 {
529 // Serialize the key
530 uint32_t key_size = map::serialize(*p.k, data, size);
531 map::add_padding(key_size, data, size);
532
533 // Serialize the value
534 uint32_t value_size = map::serialize(*p.v, data, size);
535 map::add_padding(value_size, data, size);
536 }
537
538 CCF_ASSERT_FMT(size == 0, "buffer not filled, remaining:{}", size);
539 }
540 };

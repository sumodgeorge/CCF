201 Map insert(const K& x, const V& v) const

231 static Map balance(const Map& lft, const K& x, const V& v, const Map& rgt)
232 {

236 lft.left().paint(B),

303 void serialize(uint8_t* data)
304 {
305 size_t size = map.get_serialized_size();
306
307 map.foreach([&data, &size](const K& k, const V& v) {
308 // Serialize the key
309 uint32_t key_size = map::serialize(k, data, size);
310 map::add_padding(key_size, data, size);
311
312 // Serialize the value
313 uint32_t value_size = map::serialize(v, data, size);
314 map::add_padding(value_size, data, size);
315 return true;
316 });
317
318 CCF_ASSERT_FMT(size == 0, "buffer not filled, remaining:{}", size);
319 }

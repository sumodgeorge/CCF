125 public:
126 void write(
127 const LogLine& ll,
128 const std::optional<double>& enclave_offset = std::nullopt) override
129 {
130 // Fetch time
131 ::timespec host_ts;
132 ::timespec_get(&host_ts, TIME_UTC);
133 std::tm host_tm;
134 ::gmtime_r(&host_ts.tv_sec, &host_tm);
135
136 const auto escaped_msg = nlohmann::json(ll.msg).dump();
137
138 std::string s;
139 if (enclave_offset.has_value())
140 {
141 ::timespec enc_ts = host_ts;
142 enc_ts.tv_sec += (size_t)enclave_offset.value();
143 enc_ts.tv_nsec +=
144 (long long)(enclave_offset.value() * ns_per_s) % ns_per_s;
145
146 if (enc_ts.tv_nsec > ns_per_s)
147 {
148 enc_ts.tv_sec += 1;
149 enc_ts.tv_nsec -= ns_per_s;
150 }
151
152 std::tm enclave_tm;
153 gmtime_r(&enc_ts.tv_sec, &enclave_tm);
154
155 s = fmt::format(
156 "{{\"h_ts\":\"{}\",\"e_ts\":\"{}\",\"thread_id\":\"{}\",\"level\":\"{"
157 "}\",\"tag\":\"{}\",\"file\":\"{}\",\"number\":\"{}\",\"msg\":{}}}\n",
158 get_timestamp(host_tm, host_ts),
159 get_timestamp(enclave_tm, enc_ts),
160 ll.thread_id,
161 to_string(ll.log_level),
162 ll.tag,
163 ll.file_name,
164 ll.line_number,
165 escaped_msg);
166 }
167 else
168 {
169 s = fmt::format(
170 "{{\"h_ts\":\"{}\",\"thread_id\":\"{}\",\"level\":\"{}\",\"tag\":\"{}"
171 "\",\"file\":\"{}\",\"number\":\"{}\",\"msg\":{}}}\n",
172 get_timestamp(host_tm, host_ts),
173 ll.thread_id,
174 to_string(ll.log_level),
175 ll.tag,
176 ll.file_name,
177 ll.line_number,
178 escaped_msg);
179 }
180
181 emit(s);
182 }

185 static std::string format_to_text(
186 const LogLine& ll,
187 const std::optional<double>& enclave_offset = std::nullopt)
188 {
189 // Fetch time
190 ::timespec host_ts;
191 ::timespec_get(&host_ts, TIME_UTC);
192 std::tm host_tm;
193 ::gmtime_r(&host_ts.tv_sec, &host_tm);
194
195 auto file_line = fmt::format("{}:{} ", ll.file_name, ll.line_number);
196 auto file_line_data = file_line.data();
197
198 // The preamble is the level, then tag, then file line. If the file line is
199 // too long, the final characters are retained.
200 auto preamble = fmt::format(
201 "[{:<5}]{} ",
202 to_string(ll.log_level),
203 (ll.tag.empty() ? "" : fmt::format("[{}]", ll.tag)))
204 .substr(0, preamble_length);
205 const auto max_file_line_len = preamble_length - preamble.size();
206
207 const auto len = file_line.size();
208 if (len > max_file_line_len)
209 {
210 file_line_data += len - max_file_line_len;
211 }
212
213 preamble += file_line_data;
214
215 if (enclave_offset.has_value())
216 {
217 // Sample: "2019-07-19 18:53:25.690183 -0.130 " where -0.130 indicates
218 // that the time inside the enclave was 130 milliseconds earlier than
219 // the host timestamp printed on the line
220 return fmt::format(
221 "{} {:+01.3f} {:<3} {:<45}| {}\n",
222 get_timestamp(host_tm, host_ts),
223 enclave_offset.value(),
224 ll.thread_id,
225 preamble,
226 ll.msg);
227 }
228 else
229 {
230 // Padding on the right to align the rest of the message
231 // with lines that contain enclave time offsets
232 return fmt::format(
233 "{} {:<3} {:<45}| {}\n",
234 get_timestamp(host_tm, host_ts),
235 ll.thread_id,
236 preamble,
237 ll.msg);
238 }
239 }

333// The == operator is being used to:
334// 1. Be a lower precedence than <<, such that using << on the LogLine will
335// happen before the LogLine is "equalitied" with the Out.
336// 2. Be a higher precedence than &&, such that the log statement is bound
337// more tightly than the short-circuiting.

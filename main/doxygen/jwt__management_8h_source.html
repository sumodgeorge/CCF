19 static void remove_jwt_public_signing_keys(kv::Tx& tx, std::string issuer)
20 {
21 auto keys = tx.rw<JwtPublicSigningKeys>(Tables::JWT_PUBLIC_SIGNING_KEYS);
22 auto key_issuer =
23 tx.rw<JwtPublicSigningKeyIssuer>(Tables::JWT_PUBLIC_SIGNING_KEY_ISSUER);
24
25 key_issuer->foreach(
26 [&issuer, &keys, &key_issuer](const auto& k, const auto& v) {
27 if (v == issuer)
28 {
29 keys->remove(k);
30 key_issuer->remove(k);
31 }
32 return true;
33 });
34 }

36 static oe_result_t oe_verify_attestation_certificate_with_evidence_cb(
37 oe_claim_t* claims, size_t claims_length, void* arg)
38 {
39 auto claims_map = (std::map<std::string, std::vector<uint8_t>>*)arg;
40 for (size_t i = 0; i < claims_length; i++)
41 {
42 std::string claim_name(claims[i].name);
43 std::vector<uint8_t> claim_value(
44 claims[i].value, claims[i].value + claims[i].value_size);
45 claims_map->emplace(std::move(claim_name), std::move(claim_value));
46 }
47 return OE_OK;
48 }

50 static bool set_jwt_public_signing_keys(
51 kv::Tx& tx,
52 const ProposalId& proposal_id,
53 std::string issuer,
54 const JwtIssuerMetadata& issuer_metadata,
55 const JsonWebKeySet& jwks)
56 {
57 auto keys = tx.rw<JwtPublicSigningKeys>(Tables::JWT_PUBLIC_SIGNING_KEYS);
58 auto key_issuer =
59 tx.rw<JwtPublicSigningKeyIssuer>(Tables::JWT_PUBLIC_SIGNING_KEY_ISSUER);
60
61 auto log_prefix = proposal_id.empty() ?
62 "JWT key auto-refresh" :
63 fmt::format("Proposal {}", proposal_id);
64
65 // add keys
66 if (jwks.keys.empty())
67 {
68 LOG_FAIL_FMT("{}: JWKS has no keys", log_prefix, proposal_id);
69 return false;
70 }
71 std::map<std::string, std::vector<uint8_t>> new_keys;
72 for (auto& jwk : jwks.keys)
73 {
74 if (keys->has(jwk.kid) && key_issuer->get(jwk.kid).value() != issuer)
75 {
76 LOG_FAIL_FMT(
77 "{}: key id {} already added for different issuer",
78 log_prefix,
79 jwk.kid);
80 return false;
81 }
82 if (jwk.x5c.empty())
83 {
84 LOG_FAIL_FMT("{}: JWKS is invalid (empty x5c)", log_prefix);
85 return false;
86 }
87
88 auto& der_base64 = jwk.x5c[0];
89 ccf::Cert der;
90 try
91 {
92 der = crypto::raw_from_b64(der_base64);
93 }
94 catch (const std::invalid_argument& e)
95 {
96 LOG_FAIL_FMT(
97 "{}: Could not parse x5c of key id {}: {}",
98 log_prefix,
99 jwk.kid,
100 e.what());
101 return false;
102 }
103
104 std::map<std::string, std::vector<uint8_t>> claims;
105 bool has_key_policy_sgx_claims = issuer_metadata.key_policy.has_value() &&
106 issuer_metadata.key_policy.value().sgx_claims.has_value() &&
107 !issuer_metadata.key_policy.value().sgx_claims.value().empty();
108 if (
109 issuer_metadata.key_filter == JwtIssuerKeyFilter::SGX ||
110 has_key_policy_sgx_claims)
111 {
112 oe_verify_attestation_certificate_with_evidence(
113 der.data(),
114 der.size(),
115 oe_verify_attestation_certificate_with_evidence_cb,
116 &claims);
117 }
118
119 if (
120 issuer_metadata.key_filter == JwtIssuerKeyFilter::SGX && claims.empty())
121 {
122 LOG_INFO_FMT(
123 "{}: Skipping JWT signing key with kid {} (not OE "
124 "attested)",
125 log_prefix,
126 jwk.kid);
127 continue;
128 }
129
130 if (has_key_policy_sgx_claims)
131 {
132 for (auto& [claim_name, expected_claim_val_hex] :
133 issuer_metadata.key_policy.value().sgx_claims.value())
134 {
135 if (claims.find(claim_name) == claims.end())
136 {
137 LOG_FAIL_FMT(
138 "{}: JWKS kid {} is missing the {} SGX claim",
139 log_prefix,
140 jwk.kid,
141 claim_name);
142 return false;
143 }
144 auto& actual_claim_val = claims[claim_name];
145 auto actual_claim_val_hex = ds::to_hex(actual_claim_val);
146 if (expected_claim_val_hex != actual_claim_val_hex)
147 {
148 LOG_FAIL_FMT(
149 "{}: JWKS kid {} has a mismatching {} SGX claim: {} != {}",
150 log_prefix,
151 jwk.kid,
152 claim_name,
153 expected_claim_val_hex,
154 actual_claim_val_hex);
155 return false;
156 }
157 }
158 }
159 else
160 {
161 try
162 {
163 crypto::make_unique_verifier(
164 (std::vector<uint8_t>)der); // throws on error
165 }
166 catch (std::invalid_argument& exc)
167 {
168 LOG_FAIL_FMT(
169 "{}: JWKS kid {} has an invalid X.509 certificate: {}",
170 log_prefix,
171 jwk.kid,
172 exc.what());
173 return false;
174 }
175 }
176 LOG_INFO_FMT(
177 "{}: Storing JWT signing key with kid {}", log_prefix, jwk.kid);
178 new_keys.emplace(jwk.kid, der);
179 }
180 if (new_keys.empty())
181 {
182 LOG_FAIL_FMT("{}: no keys left after applying filter", log_prefix);
183 return false;
184 }
185
186 remove_jwt_public_signing_keys(tx, issuer);
187 for (auto& [kid, der] : new_keys)
188 {
189 keys->put(kid, der);
190 key_issuer->put(kid, issuer);
191 }
192
193 return true;
194 }

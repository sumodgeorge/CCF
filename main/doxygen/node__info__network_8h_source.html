182 // The JSON representation of a NodeInfoNetwork is the union of a
183 // NodeInfoNetwork_v1 and a NodeInfoNetwork_v2. It contains the fields of
184 // both, so can be read as (or from!) either
185 inline void to_json(nlohmann::json& j, const NodeInfoNetwork& nin)
186 {
187 {
188 NodeInfoNetwork_v1 v1;
189 std::tie(v1.nodehost, v1.nodeport) =
190 split_net_address(nin.node_to_node_interface.bind_address);
191
192 if (nin.rpc_interfaces.size() > 0)
193 {
194 const auto& primary_interface = nin.rpc_interfaces.begin()->second;
195 std::tie(v1.rpchost, v1.rpcport) =
196 split_net_address(primary_interface.bind_address);
197 std::tie(v1.pubhost, v1.pubport) =
198 split_net_address(primary_interface.published_address);
199 }
200 to_json(j, v1);
201 }
202
203 to_json(j, (const NodeInfoNetwork_v2&)nin);
204 }
205
206 inline void from_json(const nlohmann::json& j, NodeInfoNetwork& nin)
207 {
208 try
209 {
210 NodeInfoNetwork_v2 v2;
211 from_json(j, v2);
212 nin = NodeInfoNetwork(v2);
213 }
214 catch (const JsonParseError& jpe)
215 {
216 NodeInfoNetwork_v1 v1;
217 from_json(j, v1);
218
219 nin.node_to_node_interface.bind_address =
220 make_net_address(v1.nodehost, v1.nodeport);
221
222 NodeInfoNetwork::NetInterface primary_interface;
223 primary_interface.bind_address = make_net_address(v1.rpchost, v1.rpcport);
224 primary_interface.published_address =
225 make_net_address(v1.pubhost, v1.pubport);
226
227 nin.rpc_interfaces.emplace(
228 PRIMARY_RPC_INTERFACE, std::move(primary_interface));
229 }
230 }

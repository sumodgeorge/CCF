15 // A Serializer is generally used as a template argument. It should provide:
16 // - serialize(...): returning a tuple of PartialSerializations representing
17 // all passed args. It may be templated and variadic or accept only precise
18 // argument types. Ideally it should give a useful error message at compile
19 // time when called with argument types it doesn't support.
20 // - deserialize(const uint8_t* data, size_t size): returning a tuple of
21 // elements parsed from byte buffer, potentially throwing logic errors if
22 // anything is missing/malformed. May need to be templated on expected
23 // types.

34 template <size_t I = 0, typename F, typename... Ts>
35 static void tuple_for_each(const std::tuple<Ts...>& t, const F& f)
36 {
37 if constexpr (I < sizeof...(Ts))
38 {
39 f(std::get<I>(t));
40 tuple_for_each<I + 1>(t, f);
41 }
42 }
43
46 template <size_t I = 0, typename F, typename... Ts>
47 static auto tuple_apply(const std::tuple<Ts...>& t, const F& f)
48 {
49 if constexpr (sizeof...(Ts) == 1)
50 {
51 return std::make_tuple();
52 }
53 else if constexpr (I == sizeof...(Ts) - 1)
54 {
55 return f(std::get<I>(t));
56 }
57 else
58 {
59 return std::tuple_cat(f(std::get<I>(t)), tuple_apply<I + 1>(t, f));
60 }
61 }

77 template <size_t I, typename T>
78 struct close_enough_at
79 {
80 using CanonTarget =
81 nonstd::remove_cvref_t<typename std::tuple_element_t<I, Tup>>;
82 using CanonArgument = nonstd::remove_cvref_t<T>;
83
84 // This determines what types a Serializer will accept as arguments to
85 // serialize(...), relative to the declared param types.
86 // The main feature is the removal of const, volatile, and references
87 // from the type so that a Serializer<int> will accept an argument of
88 // type const int&.
89 // Additionally, this will accept ByteRange arguments for parameters
90 // declared as std::vector<uint8_t> and vice versa - when serializing we
91 // can copy bytes from either in the same way, but there is a
92 // distinction in the deserialized type of whether we are copying (to a
93 // byte vector) or referring to an existing byte range. It may be
94 // possible to generalise this further and replace with
95 // std::is_constructible, but these restrictions are sufficient for the
96 // current uses.
97 // Additionally again, this will accept an array-like of ByteRanges for
98 // a single ByteRange parameter. These will be serialised in-order, and
99 // produce a single ByteRange in deserialisation.
100 static constexpr bool value =
101 std::is_same_v<CanonTarget, CanonArgument> ||
102 (std::is_same_v<CanonTarget, std::vector<uint8_t>> &&
103 std::is_same_v<CanonArgument, ByteRange>) ||
104 (std::is_same_v<CanonTarget, ByteRange> &&
105 std::is_same_v<CanonArgument, std::vector<uint8_t>>) ||
106 (std::is_same_v<CanonTarget, ByteRange> &&
107 (std::is_array_v<CanonArgument> &&
108 std::is_same_v<std::remove_extent_t<CanonArgument>, ByteRange>));
109 };

392 template <typename T, typename... Ts>
393 static auto deserialize_impl(const uint8_t* data, size_t size)
394 {
395 using StrippedT = nonstd::remove_cvref_t<T>;
396
397 if constexpr (sizeof...(Ts) == 0)
398 {
399 return std::make_tuple(
400 deserialize_value_final(data, size, Tag<StrippedT>{}));
401 }
402 else
403 {
404 const auto next =
405 std::make_tuple(deserialize_value(data, size, Tag<StrippedT>{}));
406 return std::tuple_cat(next, deserialize_impl<Ts...>(data, size));
407 }
408 }
409
410 public:
411 // Can also serialize empty messages
412 using EmptySerializer::serialize;
413
416 template <typename T, typename... Ts>
417 static auto serialize(T&& t, Ts&&... ts)
418 {
419 if constexpr (sizeof...(Ts) == 0)
420 {
421 return serialize_value_final(std::forward<T>(t));
422 }
423 else
424 {
425 const auto next = serialize_value(std::forward<T>(t));
426 return std::tuple_cat(next, serialize(std::forward<Ts>(ts)...));
427 }
428 }
429
430 template <typename... Ts>
431 static auto deserialize(const uint8_t* data, size_t size)
432 {
433 if constexpr (sizeof...(Ts) == 0)
434 {
435 return EmptySerializer::deserialize(data, size);
436 }
437 else
438 {
439 return deserialize_impl<Ts...>(data, size);
440 }

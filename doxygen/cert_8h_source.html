18 // This class represents the authentication/authorization context for a TLS
19 // session. At least, it contains the peer's CA. At most, it also contains our
20 // own private key/certificate which will be presented in the TLS handshake.
21 // The peer's certificate verification can be overridden with the auth
22 // parameter.
23 class Cert
24 {
25 private:
26 std::shared_ptr<CA> peer_ca;
27 std::optional<std::string> peer_hostname;
28 bool auth_required;
29
30 Unique_X509 own_cert;
31 std::shared_ptr<crypto::KeyPair_OpenSSL> own_pkey;
32 bool has_own_cert = false;
33
34 public:
35 Cert(
36 std::shared_ptr<CA> peer_ca_,
37 const std::optional<crypto::Pem>& own_cert_ = std::nullopt,
38 const std::optional<crypto::Pem>& own_pkey_ = std::nullopt,
39 const std::optional<std::string>& peer_hostname_ = std::nullopt,
40 bool auth_required_ = true) :
41 peer_ca(peer_ca_),
42 peer_hostname(peer_hostname_),
43 auth_required(auth_required_)
44 {
45 if (own_cert_.has_value() && own_pkey_.has_value())
46 {
47 Unique_BIO certbio(*own_cert_);
48 own_cert = Unique_X509(certbio, true);
49 own_pkey = std::make_shared<crypto::KeyPair_OpenSSL>(*own_pkey_);
50 has_own_cert = true;
51 }
52 }
53
54 ~Cert() = default;

56 void use(SSL* ssl, SSL_CTX* ssl_ctx)
57 {
58 if (peer_hostname.has_value())
59 {
60 // Peer hostname for SNI
61 SSL_set_tlsext_host_name(ssl, peer_hostname->c_str());
62 }
63
64 if (peer_ca)
65 {
66 peer_ca->use(ssl_ctx);
67 }
68
69 if (auth_required)
70 {
71 int opts = SSL_VERIFY_PEER | SSL_VERIFY_FAIL_IF_NO_PEER_CERT;
72 auto cb = [](int ok, x509_store_ctx_st*) {
73 LOG_DEBUG_FMT("peer certificate verified: {}", ok);
74 return ok;
75 };
76 SSL_CTX_set_verify(ssl_ctx, opts, cb);
77 SSL_set_verify(ssl, opts, cb);
78 }
79 else
80 {
81 // Calling set_verify with SSL_VERIFY_PEER forces the handshake to
82 // request a peer certificate. The server always sends it to the client
83 // but not the other way around. Some code relies on the server doing
84 // that, so we set this here. We return 1 from the validation callback
85 // (a common pattern in OpenSSL implementations) because we don't want
86 // to verify it here, just request it.
87 auto cb = [](int, x509_store_ctx_st*) { return 1; };
88 SSL_CTX_set_verify(ssl_ctx, SSL_VERIFY_PEER, cb);
89 SSL_set_verify(ssl, SSL_VERIFY_PEER, cb);
90 }
91
92 if (has_own_cert)
93 {
94 CHECK1(SSL_CTX_use_cert_and_key(ssl_ctx, own_cert, *own_pkey, NULL, 1));
95 CHECK1(SSL_use_cert_and_key(ssl, own_cert, *own_pkey, NULL, 1));
96 }
97 }
98 };

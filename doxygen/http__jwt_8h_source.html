43 static bool parse_auth_scheme(
44 std::string_view& auth_header_value, std::string& error_reason)
45 {
46 auto next_space = auth_header_value.find(" ");
47 if (next_space == std::string::npos)
48 {
49 error_reason = "Authorization header only contains one field";
50 return false;
51 }
52 auto auth_scheme = auth_header_value.substr(0, next_space);
53 if (auth_scheme != auth::BEARER_AUTH_SCHEME)
54 {
55 error_reason = fmt::format(
56 "Authorization header does not have {} scheme",
57 auth::BEARER_AUTH_SCHEME);
58 return false;
59 }
60 auth_header_value = auth_header_value.substr(next_space + 1);
61 return true;
62 }

64 static std::optional<Token> parse_token(
65 std::string_view& token, std::string& error_reason)
66 {
67 constexpr char separator = '.';
68 size_t first_dot = token.find(separator);
69 size_t second_dot = std::string::npos;
70 if (first_dot != std::string::npos)
71 {
72 second_dot = token.find(separator, first_dot + 1);
73 }
74 size_t extra_dot = std::string::npos;
75 if (second_dot != std::string::npos)
76 {
77 extra_dot = token.find(separator, second_dot + 1);
78 }
79 if (
80 first_dot == std::string::npos || second_dot == std::string::npos ||
81 extra_dot != std::string::npos)
82 {
83 error_reason = "Malformed JWT: must contain exactly 3 parts";
84 return std::nullopt;
85 }
86 size_t header_size = first_dot;
87 size_t payload_size = second_dot - first_dot - 1;
88 std::string_view header_b64url = token.substr(0, header_size);
89 std::string_view payload_b64url =
90 token.substr(first_dot + 1, payload_size);
91 std::string_view signature_b64url = token.substr(second_dot + 1);
92 auto header_raw = crypto::raw_from_b64url(header_b64url);
93 auto payload_raw = crypto::raw_from_b64url(payload_b64url);
94 auto signature_raw = crypto::raw_from_b64url(signature_b64url);
95 auto signed_content = token.substr(0, second_dot);
96 nlohmann::json header;
97 nlohmann::json payload;
98 try
99 {
100 header = nlohmann::json::parse(header_raw);
101 payload = nlohmann::json::parse(payload_raw);
102 }
103 catch (const nlohmann::json::parse_error& e)
104 {
105 error_reason =
106 fmt::format("JWT header or payload is not valid JSON: {}", e.what());
107 return std::nullopt;
108 }
109 if (!header.is_object() || !payload.is_object())
110 {
111 error_reason = "JWT header or payload is not an object";
112 return std::nullopt;
113 }
114 JwtHeader header_typed;
115 try
116 {
117 header_typed = header.get<JwtHeader>();
118 }
119 catch (const nlohmann::json::exception& e)
120 {
121 error_reason =
122 fmt::format("JWT header does not follow schema: {}", e.what());
123 return std::nullopt;
124 }
125 Token parsed = {
126 header, header_typed, payload, signature_raw, signed_content};
127 return parsed;
128 }

130 static std::optional<Token> extract_token(
131 const http::HeaderMap& headers, std::string& error_reason)
132 {
133 const auto auth_it = headers.find(headers::AUTHORIZATION);
134 if (auth_it == headers.end())
135 {
136 error_reason = fmt::format("Missing {} header", headers::AUTHORIZATION);
137 return std::nullopt;
138 }
139 std::string_view token = auth_it->second;
140 if (!parse_auth_scheme(token, error_reason))
141 {
142 return std::nullopt;
143 }
144 auto parsed = parse_token(token, error_reason);
145 return parsed;
146 }

148 static bool validate_token_signature(
149 const Token& token, std::vector<uint8_t> cert_der)
150 {
151 auto verifier = crypto::make_unique_verifier(cert_der);
152 bool valid = verifier->verify(
153 (uint8_t*)token.signed_content.data(),
154 token.signed_content.size(),
155 token.signature.data(),
156 token.signature.size(),
157 crypto::MDType::SHA256);
158 return valid;

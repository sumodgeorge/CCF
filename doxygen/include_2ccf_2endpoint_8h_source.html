218 template <typename In, typename Out>
219 Endpoint& set_auto_schema(std::optional<http_status> status = std::nullopt)
220 {
221 if constexpr (!std::is_same_v<In, void>)
222 {
223 params_schema = ds::json::build_schema<In>();
224
225 schema_builders.push_back(
226 [](nlohmann::json& document, const Endpoint& endpoint) {
227 const auto http_verb = endpoint.dispatch.verb.get_http_method();
228 if (!http_verb.has_value())
229 {
230 // Non-HTTP endpoints are not documented
231 return;
232 }
233
234 ds::openapi::add_request_body_schema<In>(
235 document, endpoint.full_uri_path, http_verb.value());
236 });
237 }
238 else
239 {
240 params_schema = nullptr;
241 }
242
243 if constexpr (!std::is_same_v<Out, void>)
244 {
245 success_status = status.value_or(HTTP_STATUS_OK);
246
247 result_schema = ds::json::build_schema<Out>();
248
249 schema_builders.push_back(
250 [](nlohmann::json& document, const Endpoint& endpoint) {
251 const auto http_verb = endpoint.dispatch.verb.get_http_method();
252 if (!http_verb.has_value())
253 {
254 return;
255 }
256
257 ds::openapi::add_response_schema<Out>(
258 document,
259 endpoint.full_uri_path,
260 http_verb.value(),
261 endpoint.success_status);
262 });
263 }
264 else
265 {
266 success_status = status.value_or(HTTP_STATUS_NO_CONTENT);
267 result_schema = nullptr;
268 }
269
270 return *this;
271 }

304 template <typename T>
305 Endpoint& add_query_parameter(
306 const std::string& param_name,
307 QueryParamPresence presence = RequiredParameter)
308 {
309 schema_builders.push_back(
310 [param_name,
311 presence](nlohmann::json& document, const Endpoint& endpoint) {
312 const auto http_verb = endpoint.dispatch.verb.get_http_method();
313 if (!http_verb.has_value())
314 {
315 // Non-HTTP endpoints are not documented
316 return;
317 }
318
319 const auto schema_name = ds::json::schema_name<T>();
320 const auto query_schema = ds::json::build_schema<T>();
321
322 auto parameter = nlohmann::json::object();
323 parameter["name"] = param_name;
324 parameter["in"] = "query";
325 parameter["required"] = presence == RequiredParameter;
326 parameter["schema"] = ds::openapi::add_schema_to_components(
327 document, schema_name, query_schema);
328 ds::openapi::add_request_parameter_schema(
329 document, endpoint.full_uri_path, http_verb.value(), parameter);
330 });
331
332 return *this;
333 }

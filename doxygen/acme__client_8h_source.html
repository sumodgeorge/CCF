71 void get_certificate(
72 std::shared_ptr<crypto::KeyPair> service_key, bool override_time = false)
73 {
74 using namespace std::chrono_literals;
75 using namespace std::chrono;
76
77 bool ok = true;
78 system_clock::duration delta(0);
79
80 if (last_request && !override_time)
81 {
82 // Let's encrypt recommends this retry strategy in their integration
83 // guide, see https://letsencrypt.org/docs/integration-guide/
84
85 delta = system_clock::now() - *last_request;
86 ok = false;
87 switch (num_failed_attempts)
88 {
89 case 0:
90 ok = true;
91 break;
92 case 1:
93 ok = delta >= 1min;
94 break;
95 case 2:
96 ok = delta >= 10min;
97 break;
98 case 3:
99 ok = delta >= 100min;
100 break;
101 default:
102 ok = delta >= 24h;
103 break;
104 }
105 }
106
107 if (ok)
108 {
109 this->service_key = service_key;
110 last_request = system_clock::now();
111 num_failed_attempts++;
112 request_directory();
113 }
114 else
115 {
116 LOG_INFO_FMT(
117 "ACME: Ignoring certificate request due to {} recent failed "
118 "attempt(s) within {} seconds",
119 num_failed_attempts,
120 duration_cast<seconds>(delta).count());
121 }
122 }

173 void make_request(
174 llhttp_method method,
175 const http::URL& url,
176 const std::vector<uint8_t>& body,
177 http_status expected_status,
178 std::function<bool(const http::HeaderMap&, const std::vector<uint8_t>&)>
179 ok_callback)
180 {
181 std::unique_lock<ccf::Mutex> guard(req_lock);
182
183 try
184 {
185 auto port = url.port.empty() ? "443" : url.port;
186 LOG_INFO_FMT(
187 "ACME: Requesting https://{}:{}{}", url.host, port, url.path);
188
189 http::Request r(url.path, method);
190 r.set_header(http::headers::ACCEPT, "*/*");
191 r.set_header(
192 http::headers::HOST, fmt::format("{}:{}", url.host, url.port));
193 if (!body.empty())
194 {
195 r.set_header(http::headers::CONTENT_TYPE, "application/jose+json");
196 r.set_body(&body);
197 }
198 auto req = r.build_request();
199 std::string reqs(req.begin(), req.end());
200 LOG_TRACE_FMT("ACME: Request:\n{}", reqs);
201
202 on_http_request(
203 url,
204 std::move(r),
205 [this, expected_status, ok_callback](
206 http_status status,
207 http::HeaderMap&& headers,
208 std::vector<uint8_t>&& data) {
209 for (auto& [k, v] : headers)
210 {
211 LOG_TRACE_FMT("ACME: H: {}: {}", k, v);
212 }
213
214 if (status != expected_status && status != HTTP_STATUS_OK)
215 {
216 LOG_INFO_FMT(
217 "ACME: request failed with status={} and body={}",
218 (int)status,
219 std::string(data.begin(), data.end()));
220 return false;
221 }
222 else
223 {
224 LOG_TRACE_FMT(
225 "ACME: data: {}", std::string(data.begin(), data.end()));
226 }
227
228 auto nonce_opt = get_header_value(headers, "replay-nonce");
229 if (nonce_opt)
230 {
231 nonces.push_back(*nonce_opt);
232 }
233
234 try
235 {
236 ok_callback(headers, data);
237 }
238 catch (const std::exception& ex)
239 {
240 LOG_FAIL_FMT("ACME: response callback failed: {}", ex.what());
241 return false;
242 }
243 return true;
244 });
245 }
246 catch (const std::exception& ex)
247 {
248 LOG_FAIL_FMT("Failed to connect to ACME server: {}", ex.what());
249 }
250 }

290 const std::string& resource_url,

293 {

424 static void convert_signature_to_ieee_p1363(
425 std::vector<uint8_t>& sig, const crypto::KeyPair& signer)
426 {
427 // Convert signature from ASN.1 format to IEEE P1363
428 const unsigned char* pp = sig.data();
429 ECDSA_SIG* sig_r_s = d2i_ECDSA_SIG(NULL, &pp, sig.size());
430 const BIGNUM* r = ECDSA_SIG_get0_r(sig_r_s);
431 const BIGNUM* s = ECDSA_SIG_get0_s(sig_r_s);
432 int r_n = BN_num_bytes(r);
433 int s_n = BN_num_bytes(s);
434 size_t sz = signer.coordinates().x.size();
435 sig = std::vector<uint8_t>(2 * sz, 0);
436 BN_bn2binpad(r, sig.data(), sz);
437 BN_bn2binpad(s, sig.data() + sz, sz);
438 ECDSA_SIG_free(sig_r_s);
439 }

441 class JWS : public nlohmann::json::object_t
442 {
443 public:
444 JWS(
445 const nlohmann::json& header_,
446 const nlohmann::json& payload_,
447 crypto::KeyPair& signer_,
448 bool empty_payload = false)
449 {
450 LOG_TRACE_FMT("JWS header: {}", header_.dump());
451 LOG_TRACE_FMT("JWS payload: {}", payload_.dump());
452 auto header_b64 = json_to_b64url(header_, false);
453 auto payload_b64 = empty_payload ? "" : json_to_b64url(payload_, false);
454 set(header_b64, payload_b64, signer_);
455 }
456
457 JWS(const nlohmann::json& header_, crypto::KeyPair& signer_) :
458 JWS(header_, nlohmann::json::object_t(), signer_, true)
459 {}
460
461 virtual ~JWS() {}
462
463 protected:
464 void set(
465 const std::string& header_b64,
466 const std::string& payload_b64,
467 crypto::KeyPair& signer)
468 {
469 auto msg = header_b64 + "." + payload_b64;
470 auto sig = signer.sign(s2v(msg));
471 convert_signature_to_ieee_p1363(sig, signer);
472 auto sig_b64 = crypto::b64url_from_raw(sig);
473
474 (*this)["protected"] = header_b64;
475 (*this)["payload"] = payload_b64;
476 (*this)["signature"] = sig_b64;
477 }
478 };

543 static std::pair<std::string, std::string> get_crv_alg(
544 const std::shared_ptr<crypto::KeyPair>& key_pair)
545 {
546 std::string crv, alg;
547 if (key_pair->get_curve_id() == crypto::CurveID::SECP256R1)
548 {
549 crv = "P-256";
550 alg = "ES256";
551 }
552 else if (key_pair->get_curve_id() == crypto::CurveID::SECP384R1)
553 {
554 crv = "P-384";
555 alg = "ES384";
556 }
557 else
558 throw std::runtime_error("Unsupported curve");
559
560 return std::make_pair(crv, alg);
561 }

563 Order* get_order(const std::string& order_url)
564 {
565 auto oit = std::find_if(
566 active_orders.begin(),
567 active_orders.end(),
568 [&order_url](const Order& other) {
569 return order_url == other.order_url;
570 });
571
572 if (oit != active_orders.end())
573 {
574 return &(*oit);
575 }
576
577 LOG_DEBUG_FMT("ACME: no such order {}", order_url);
578
579 return nullptr;

582 void remove_order(const std::string& order_url)
583 {
584 LOG_TRACE_FMT("ACME: removing order {}", order_url);
585
586 std::unique_lock<ccf::Mutex> guard(orders_lock);
587 for (auto it = active_orders.begin(); it != active_orders.end();)
588 {
589 if (it->order_url == order_url)
590 {
591 for (const auto& [_, challenge] : it->challenges)
592 {
593 on_challenge_finished(challenge.token);
594 }
595 it = active_orders.erase(it);
596 break;
597 }
598 else
599 {
600 it++;
601 }
602 }

605 nlohmann::json mk_kid_header(
606 const std::string& account_url,
607 const std::string& nonce,
608 const std::string& resource_url)
609 {
610 // For all other requests, the request is signed using an existing
611 // account, and there MUST be a "kid" field. This field MUST contain the
612 // account URL received by POSTing to the newAccount resource.
613
614 auto crv_alg = get_crv_alg(account_key_pair);
615
616 nlohmann::json r = {
617 {"alg", crv_alg.second},
618 {"kid", account_url},
619 {"nonce", nonce},
620 {"url", resource_url}};
621
622 return r;
623 }

654 void request_new_account()
655 {
656 std::string new_account_url =
657 directory.at("newAccount").get<std::string>();
658
659 if (nonces.empty())
660 {
661 request_new_nonce([this]() { request_new_account(); });
662 }
663 else
664 {
665 auto nonce = nonces.front();
666 nonces.pop_front();
667
668 auto crv_alg = get_crv_alg(account_key_pair);
669 auto key_coords = account_key_pair->coordinates();
670
671 JWK jwk(
672 "EC",
673 crv_alg.first,
674 crypto::b64url_from_raw(key_coords.x, false),
675 crypto::b64url_from_raw(key_coords.y, false));
676
677 nlohmann::json header = {
678 {"alg", crv_alg.second},
679 {"jwk", jwk},
680 {"nonce", nonce},
681 {"url", new_account_url}};
682
683 nlohmann::json payload = {
684 {"termsOfServiceAgreed", config.terms_of_service_agreed},
685 {"contact", config.contact}};
686
687 JWS jws(header, payload, *account_key_pair);
688
689 http::URL url = with_default_port(new_account_url);
690 make_json_request(
691 HTTP_POST,
692 url,
693 json_to_bytes(jws),
694 HTTP_STATUS_CREATED,
695 [this](const http::HeaderMap& headers, const nlohmann::json& j) {
696 expect_string(j, "status", "valid");
697 account = j;
698 auto loc_opt = get_header_value(headers, "location");
699 request_new_order(loc_opt.value_or(""));
700 });
701 }

704 void authorize_next_challenge(const std::string& order_url)
705 {
706 std::unique_lock<ccf::Mutex> guard(orders_lock);

719
720 void request_new_order(const std::string& account_url)
721 {
722 if (nonces.empty())
723 {
724 request_new_nonce(
725 [this, account_url]() { request_new_order(account_url); });
726 }
727 else
728 {
729 auto nonce = nonces.front();
730 nonces.pop_front();
731
732 auto header =
733 mk_kid_header(account_url, nonce, directory.at("newOrder"));
734
735 nlohmann::json payload = {
736 {"identifiers",
737 nlohmann::json::array({
738 {{"type", "dns"}, {"value", config.service_dns_name}},
739 })}};
740
741 if (config.not_before)
742 {
743 payload["notBefore"] = *config.not_before;
744 }
745 if (config.not_after)
746 {
747 payload["notAfter"] = *config.not_after;
748 }
749
750 JWS jws(header, payload, *account_key_pair);
751
752 http::URL url = with_default_port(directory.at("newOrder"));
753 make_json_request(
754 HTTP_POST,
755 url,
756 json_to_bytes(jws),
757 HTTP_STATUS_CREATED,
758 [this, account_url](
759 const http::HeaderMap& headers, const nlohmann::json& j) {
760 expect(j, "status");
761 expect(j, "finalize");
762
763 auto order_url_opt = get_header_value(headers, "location");
764 if (!order_url_opt)
765 {
766 throw std::runtime_error("missing order location");
767 }
768
769 std::unique_lock<ccf::Mutex> guard(orders_lock);
770 active_orders.emplace_back(Order{
771 ACTIVE, account_url, *order_url_opt, j["finalize"], "", {}, {}});
772
773 Order& order = active_orders.back();
774
775 if (j["status"] == "pending" && j.contains("authorizations"))
776 {
777 expect(j, "authorizations");
778 order.authorizations =
779 j["authorizations"].get<std::unordered_set<std::string>>();
780 guard.unlock();
781 authorize_next_challenge(*order_url_opt);
782 }
783 else if (j["status"] == "ready")
784 {
785 expect(j, "finalize");
786 guard.unlock();
787 request_finalization(*order_url_opt);
788 }
789 else if (j["status"] == "valid")
790 {
791 expect(j, "certificate");
792 order.certificate_url = j["certificate"];
793 guard.unlock();
794 request_certificate(*order_url_opt);
795 }
796 else
797 {
798 LOG_FATAL_FMT(
799 "ACME: unknown order status '{}', aborting", j["status"]);
800 guard.unlock();
801 remove_order(*order_url_opt);
802 }
803 });
804 }
805 }

807 void request_authorization(Order& order, const std::string& authz_url)
808 {
809 post_as_get_json(
810 order.account_url,
811 authz_url,
812 [this, order_url = order.order_url, authz_url](

102 else if constexpr (nonstd::is_specialization<T, std::vector>::value)
103 {
104 if constexpr (std::is_same<T, std::vector<uint8_t>>::value)

107 return "base64string";
108 }
109 else
110 {
111 return fmt::format("{}_array", schema_name<typename T::value_type>());

120 schema_name<typename T::key_type>(),

132 return "string";

192 template <typename T>
193 inline void fill_schema(nlohmann::json& schema)
194 {
195 if constexpr (nonstd::is_specialization<T, std::optional>::value)
196 {
197 fill_schema<typename T::value_type>(schema);
198 }
199 else if constexpr (nonstd::is_specialization<T, std::vector>::value)
200 {
201 if constexpr (std::is_same<T, std::vector<uint8_t>>::value)
202 {
203 // Byte vectors are always base64 encoded
204 schema["type"] = "string";
205 schema["format"] = "base64";
206 }
207 else
208 {
209 schema["type"] = "array";
210 schema["items"] = schema_element<typename T::value_type>();
211 }
212 }
213 else if constexpr (
214 nonstd::is_specialization<T, std::map>::value ||
215 nonstd::is_specialization<T, std::unordered_map>::value)
216 {
217 // Nlohmann JSON serialises some maps as objects, if the keys can be
218 // converted to strings. This should detect those cases. The others are
219 // serialised as list-of-pairs
220 if constexpr (nlohmann::detail::
221 is_compatible_object_type<nlohmann::json, T>::value)
222 {
223 schema["type"] = "object";
224 schema["additionalProperties"] =
225 schema_element<typename T::mapped_type>();
226 }
227 else
228 {
229 schema["type"] = "array";
230 auto items = nlohmann::json::object();
231 {
232 items["type"] = "array";
233
234 auto sub_items = nlohmann::json::array();
235 sub_items.push_back(schema_element<typename T::key_type>());
236 sub_items.push_back(schema_element<typename T::mapped_type>());
237 items["items"] = sub_items;
238 }
239 schema["items"] = items;
240 }
241 }
242 else if constexpr (nonstd::is_specialization<T, std::pair>::value)
243 {
244 schema["type"] = "array";
245 auto items = nlohmann::json::array();
246 items.push_back(schema_element<typename T::first_type>());
247 items.push_back(schema_element<typename T::second_type>());
248 schema["items"] = items;
249 }
250 else if constexpr (std::is_same<T, std::string>::value)
251 {
252 schema["type"] = "string";
253 }
254 else if constexpr (std::is_same<T, bool>::value)
255 {
256 schema["type"] = "boolean";
257 }
258 else if constexpr (std::is_same<T, nlohmann::json>::value)
259 {
260 // Any field that contains more json is completely unconstrained, so we
261 // do not add a type or any other fields
262 schema = nlohmann::json::object();
263 }
264 else if constexpr (std::is_integral<T>::value)
265 {
266 fill_number_schema<T>(schema);
267 }
268 else if constexpr (std::is_floating_point<T>::value)
269 {
270 schema["type"] = "number";
271 }
272 else if constexpr (std::is_same<T, JsonSchema>::value)
273 {
274 schema["type"] = "object";
275 }
276 else
277 {
278 adl::fill_schema<T>(schema);
279 }
280 }

282 template <typename T>
283 inline nlohmann::json build_schema(const std::string& title = "")
284 {
285 auto schema = nlohmann::json::object();
286
287 if (!title.empty())
288 {
289 schema["title"] = title;
290 }
291
292 fill_schema<T>(schema);
293
294 return schema;
295 }

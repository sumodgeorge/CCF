77 {
78 if (ks.size() == 0)

81 }
82
83 std::vector<uint8_t> tmp(ks.begin(), ks.end());

86 int nid = crypto::PublicKey_OpenSSL::get_openssl_group_id(curve);
87 auto pk = crypto::key_from_raw_ec_point(tmp, nid);
88
89 if (!pk)

92 }
93
94 peer_key = std::make_shared<crypto::PublicKey_OpenSSL>(pk);
95 }

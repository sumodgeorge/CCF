147 std::string get_formatted_query() const
148 {
149 std::string formatted_query;
150 bool first = true;
151 for (const auto& it : query_params)

153 formatted_query +=
154 fmt::format("{}{}={}", (first ? '?' : '&'), it.first, it.second);

160 std::vector<uint8_t> build_request(bool header_only = false) const
161 {
162 const auto uri = fmt::format("{}{}", path, get_formatted_query());
163
164 const auto body_view = (header_only || body == nullptr) ?
165 std::string_view() :
166 std::string_view((char const*)body, body_size);
167
168 const auto request_string = fmt::format(
169 "{} {} HTTP/1.1\r\n"
170 "{}"
171 "\r\n"
172 "{}",
173 llhttp_method_name(method),
174 uri,
175 get_header_string(headers),
176 body_view);
177
178 return std::vector<uint8_t>(request_string.begin(), request_string.end());
179 }

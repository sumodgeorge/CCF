34 static size_t read_idx(const std::string& str)
35 {
36 size_t idx = 0;
37 auto end_ptr = str.data() + str.size();
38
39 auto res = std::from_chars(str.data(), end_ptr, idx);
40 if (res.ec != std::errc())
41 {
42 throw std::logic_error(
43 fmt::format("Could not read idx from string \"{}\": {}", str, res.ec));
44 }
45 else if (res.ptr != end_ptr)
46 {
47 throw std::logic_error(fmt::format(
48 "Trailing characters in \"{}\" cannot be converted to idx: \"{}\"",
49 str,
50 std::string(res.ptr, end_ptr)));
51 }
52 return idx;
53 }

55 static std::optional<size_t> get_evidence_commit_idx_from_file_name(
56 const std::string& file_name)
57 {
58 // Only returns an evidence commit index for 1.x committed snapshots.
59 // 1.x committed snapshots file names are of the form:
60 // "snapshot_X_Y.committed_Z" while 2.x+ ones are of the form:
61 // "snapshot_X_Y.committed"
62 auto pos = file_name.find(snapshot_committed_suffix);
63 if (pos == std::string::npos)
64 {
65 throw std::logic_error(
66 fmt::format("Snapshot file \"{}\" is not committed", file_name));
67 }
68
69 pos = file_name.find(snapshot_idx_delimiter, pos);
70 if (pos == std::string::npos)
71 {
72 // 2.x+ snapshot
73 return std::nullopt;
74 }
75
76 return read_idx(file_name.substr(pos + 1));
77 }

84 static size_t get_snapshot_idx_from_file_name(const std::string& file_name)
85 {
86 if (!is_snapshot_file(file_name))
87 {
88 throw std::logic_error(
89 fmt::format("File \"{}\" is not a valid snapshot file", file_name));
90 }
91
92 auto idx_pos = file_name.find_first_of(snapshot_idx_delimiter);
93 if (idx_pos == std::string::npos)
94 {
95 throw std::logic_error(fmt::format(
96 "Snapshot file name {} does not contain snapshot seqno", file_name));
97 }
98
99 auto evidence_idx_pos =
100 file_name.find_first_of(snapshot_idx_delimiter, idx_pos + 1);
101 if (evidence_idx_pos == std::string::npos)
102 {
103 throw std::logic_error(fmt::format(
104 "Snapshot file \"{}\" does not contain evidence index", file_name));
105 }
106
107 return read_idx(
108 file_name.substr(idx_pos + 1, evidence_idx_pos - idx_pos - 1));
109 }

111 static size_t get_snapshot_evidence_idx_from_file_name(
112 const std::string& file_name)
113 {
114 if (!is_snapshot_file(file_name))
115 {
116 throw std::logic_error(
117 fmt::format("File \"{}\" is not a valid snapshot file", file_name));
118 }
119
120 auto idx_pos = file_name.find_first_of(snapshot_idx_delimiter);
121 if (idx_pos == std::string::npos)
122 {
123 throw std::logic_error(
124 fmt::format("Snapshot file \"{}\" does not contain index", file_name));
125 }
126
127 auto evidence_idx_pos =
128 file_name.find_first_of(snapshot_idx_delimiter, idx_pos + 1);
129 if (evidence_idx_pos == std::string::npos)
130 {
131 throw std::logic_error(fmt::format(
132 "Snapshot file \"{}\" does not contain evidence index", file_name));
133 }
134
135 // Note: Snapshot file may not be committed
136 size_t end_str = std::string::npos;
137 auto commit_suffix_pos =
138 file_name.find_first_of(snapshot_committed_suffix, evidence_idx_pos + 1);
139 if (commit_suffix_pos != std::string::npos)
140 {
141 end_str = commit_suffix_pos - evidence_idx_pos - 1;
142 }
143
144 return read_idx(file_name.substr(evidence_idx_pos + 1, end_str));
145 }

147 std::optional<fs::path> find_latest_committed_snapshot_in_directory(
148 const fs::path& directory,
149 size_t ledger_last_idx,
150 size_t& latest_committed_snapshot_idx)
151 {
152 std::optional<fs::path> latest_committed_snapshot_file_name = std::nullopt;
153
154 for (auto& f : fs::directory_iterator(directory))
155 {
156 auto file_name = f.path().filename();
157 if (!is_snapshot_file(file_name))
158 {
159 LOG_INFO_FMT("Ignoring non-snapshot file {}", file_name);
160 continue;
161 }
162
163 if (!is_snapshot_file_committed(file_name))
164 {
165 LOG_INFO_FMT("Ignoring non-committed snapshot file {}", file_name);
166 continue;
167 }
168
169 // 1.x committed snapshot file names contain the evidence commit index
170 // which must be contained in the ledger for the snapshot to be valid
171 auto snapshot_evidence_commit_idx_1_x =
172 get_evidence_commit_idx_from_file_name(file_name);
173 if (
174 snapshot_evidence_commit_idx_1_x.has_value() &&
175 snapshot_evidence_commit_idx_1_x.value() > ledger_last_idx)
176 {
177 LOG_INFO_FMT(
178 "Ignoring {}: ledger does not contain evidence commit "
179 "seqno (evidence commit seqno {} > last ledger seqno {})",
180 file_name,
181 snapshot_evidence_commit_idx_1_x.value(),
182 ledger_last_idx);
183 continue;
184 }
185
186 auto snapshot_idx = get_snapshot_idx_from_file_name(file_name);
187 if (snapshot_idx > latest_committed_snapshot_idx)
188 {
189 latest_committed_snapshot_file_name = file_name;
190 latest_committed_snapshot_idx = snapshot_idx;
191 }
192 }
193
194 return latest_committed_snapshot_file_name;
195 }

239 void write_snapshot(
240 consensus::Index idx,
241 consensus::Index evidence_idx,
242 const uint8_t* snapshot_data,
243 size_t snapshot_size)
244 {
245 TimeBoundLogger log_if_slow(fmt::format(
246 "Writing snapshot - idx={}, evidence_idx={}, size={}",
247 idx,
248 evidence_idx,
249 snapshot_size));
250
251 auto snapshot_file_name = fmt::format(
252 "{}{}{}{}{}",
253 snapshot_file_prefix,
254 snapshot_idx_delimiter,
255 idx,
256 snapshot_idx_delimiter,
257 evidence_idx);
258 auto full_snapshot_path = snapshot_dir / snapshot_file_name;
259
260 if (fs::exists(full_snapshot_path))
261 {
262 LOG_FAIL_FMT(
263 "Cannot write snapshot at {} since file already exists: {}",
264 idx,
265 full_snapshot_path);
266 return;
267 }
268
269 LOG_INFO_FMT(
270 "Writing new snapshot to {} [{}]", snapshot_file_name, snapshot_size);
271
272 std::ofstream snapshot_file(
273 full_snapshot_path, std::ios::out | std::ios::binary);
274 snapshot_file.write(
275 reinterpret_cast<const char*>(snapshot_data), snapshot_size);
276 }

278 void commit_snapshot(
279 consensus::Index snapshot_idx,
280 const uint8_t* receipt_data,
281 size_t receipt_size)
282 {
283 TimeBoundLogger log_if_slow(
284 fmt::format("Committing snapshot - snapshot_idx={}", snapshot_idx));
285
286 try
287 {
288 // Find previously-generated snapshot for snapshot_idx and rename file,
289 // also appending receipt to it
290 for (auto const& f : fs::directory_iterator(snapshot_dir))
291 {
292 auto file_name = f.path().filename().string();
293 if (
294 !is_snapshot_file_committed(file_name) &&
295 get_snapshot_idx_from_file_name(file_name) == snapshot_idx)
296 {
297 auto full_snapshot_path = snapshot_dir / file_name;
298 const auto committed_file_name =
299 fmt::format("{}{}", file_name, snapshot_committed_suffix);
300
301 LOG_INFO_FMT(
302 "Committing snapshot file \"{}\" [{}]",
303 committed_file_name,
304 receipt_size);
305
306 // Append receipt to snapshot file
307 std::ofstream snapshot_file(
308 full_snapshot_path, std::ios::app | std::ios::binary);
309 snapshot_file.write(
310 reinterpret_cast<const char*>(receipt_data), receipt_size);
311
312 fs::rename(
313 snapshot_dir / file_name, snapshot_dir / committed_file_name);
314
315 return;
316 }
317 }
318
319 LOG_FAIL_FMT("Could not find snapshot to commit at {}", snapshot_idx);
320 }
321 catch (std::exception& e)
322 {
323 LOG_FAIL_FMT(
324 "Exception while attempting to commit snapshot at {}: {}",
325 snapshot_idx,
326 e.what());
327 }
328 }

330 std::optional<std::pair<fs::path, fs::path>>
331 find_latest_committed_snapshot()
332 {
333 // Keep track of latest snapshot file in both directories
334 size_t latest_idx = 0;
335 auto ledger_last_idx = ledger.get_last_idx();
336
337 std::optional<fs::path> read_only_latest_committed_snapshot =
338 std::nullopt;
339 if (read_snapshot_dir.has_value())
340 {
341 read_only_latest_committed_snapshot =
342 find_latest_committed_snapshot_in_directory(
343 read_snapshot_dir.value(), ledger_last_idx, latest_idx);
344 }
345
346 auto main_latest_committed_snapshot =
347 find_latest_committed_snapshot_in_directory(
348 snapshot_dir, ledger_last_idx, latest_idx);
349
350 if (main_latest_committed_snapshot.has_value())
351 {
352 return std::make_pair(
353 snapshot_dir, main_latest_committed_snapshot.value());
354 }
355 else if (read_only_latest_committed_snapshot.has_value())
356 {
357 return std::make_pair(
358 read_snapshot_dir.value(),
359 read_only_latest_committed_snapshot.value());
360 }
361
362 return std::nullopt;
363 }

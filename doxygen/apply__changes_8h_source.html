24 // Atomically checks for conflicts then applies the writes in the given change
25 // sets to their underlying Maps. Calls f() at most once, iff the writes are
26 // applied, to retrieve a unique Version for the write set and return the max
27 // version which can have a conflict with the transaction.
28 //
29 // The track_read_versions parameter tells the store if it needs to track the
30 // last read version for every key. This is required for backup execution as
31 // described at the top of tx.h
32
33 using VersionLastNewMap = Version;
34 using VersionResolver = std::function<std::tuple<Version, VersionLastNewMap>(
35 bool tx_contains_new_map)>;
36
37 static inline std::optional<Version> apply_changes(
38 OrderedChanges& changes,
39 VersionResolver version_resolver_fn,
40 kv::ConsensusHookPtrs& hooks,
41 const MapCollection& new_maps = {},
42 const std::optional<Version>& new_maps_conflict_version = std::nullopt,
43 bool track_read_versions = false)
44 {
45 // All maps with pending writes are locked, transactions are prepared
46 // and possibly committed, and then all maps with pending writes are
47 // unlocked. This is to prevent transactions from being committed in an
48 // interleaved fashion.
49 Version version = NoVersion;
50 bool has_writes = false;
51
52 std::map<std::string, std::unique_ptr<AbstractCommitter>> views;
53 for (const auto& [map_name, mc] : changes)
54 {
55 views[map_name] = mc.map->create_committer(mc.changeset.get());
56 }
57
58 for (auto it = changes.begin(); it != changes.end(); ++it)
59 {
60 bool changeset_has_writes = it->second.changeset->has_writes();
61 if (changeset_has_writes)
62 {
63 has_writes = true;
64 }
65 if (changeset_has_writes || track_read_versions)
66 {
67 it->second.map->lock();
68 }
69 }
70
71 bool ok = true;
72 for (auto it = views.begin(); it != views.end(); ++it)
73 {
74 if (!it->second->prepare(track_read_versions))
75 {
76 ok = false;
77 break;
78 }
79 }
80
81 for (const auto& [map_name, map_ptr] : new_maps)
82 {
83 // Check that none of these pending maps have already been created.
84 // It is possible for non-conflicting other transactions to commit here
85 // and increment the version, so we may ask this question at different
86 // versions. This is fine - none can create maps (ie - change their
87 // conflict set with this operation) while we hold the store lock. Assume
88 // that the caller is currently holding store->lock()
89 auto store = map_ptr->get_store();
90
91 // This is to avoid recursively locking version_lock by calling
92 // current_version() in the commit_reserved case.
93 kv::Version current_v;
94 if (new_maps_conflict_version.has_value())
95 {
96 current_v = *new_maps_conflict_version;
97 }
98 else
99 {
100 current_v = store->current_version();
101 }
102
103 if (store->get_map(current_v, map_name) != nullptr)
104 {
105 ok = false;
106 break;
107 }
108 }
109
110 if (ok && has_writes)
111 {
112 // Get the version number to be used for this commit.
113 kv::Version version_last_new_map;
114 std::tie(version, version_last_new_map) =
115 version_resolver_fn(!new_maps.empty());
116
117 if (!track_read_versions)
118 {
119 // Transfer ownership of these new maps to their target stores, iff we
120 // have writes to them
121 for (const auto& [map_name, map_ptr] : new_maps)
122 {
123 const auto it = views.find(map_name);
124 if (it != views.end() && it->second->has_writes())
125 {
126 map_ptr->get_store()->add_dynamic_map(version, map_ptr);
127 }
128 }
129
130 for (auto it = views.begin(); it != views.end(); ++it)
131 {
132 it->second->commit(version, track_read_versions);
133 }
134
135 // Collect ConsensusHooks
136 for (auto it = views.begin(); it != views.end(); ++it)
137 {
138 auto hook_ptr = it->second->post_commit();
139 if (hook_ptr != nullptr)
140 {
141 hooks.push_back(std::move(hook_ptr));
142 }
143 }
144 }
145 else
146 {
147 // A linearizability violation was detected
148 ok = false;
149 }
150 }
151
152 for (auto it = changes.begin(); it != changes.end(); ++it)
153 {
154 if (it->second.changeset->has_writes() || track_read_versions)
155 {
156 it->second.map->unlock();
157 }
158 }
159
160 if (!ok)
161 {
162 return std::nullopt;
163 }
164
165 return version;
166 }
167}

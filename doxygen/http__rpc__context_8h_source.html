13 inline std::vector<uint8_t> error(ccf::ErrorDetails&& error)
14 {
15 nlohmann::json body = ccf::ODataErrorResponse{
16 ccf::ODataError{std::move(error.code), std::move(error.msg)}};
17 const auto s = body.dump();
18
19 std::vector<uint8_t> data(s.begin(), s.end());
20 auto response = http::Response(error.status);
21
22 response.set_header(
23 http::headers::CONTENT_TYPE, http::headervalues::contenttype::JSON);
24 response.set_body(&data);
25
26 return response.build_response();
27 }

63 {

73 ::memcpy(
74 serialised_request.data(),
75 request_prefix.data(),
76 request_prefix.size());
77 if (!request_body.empty())
78 {
79 ::memcpy(
80 serialised_request.data() + request_prefix.size(),
81 request_body.data(),
82 request_body.size());
83 }
84 }

268 inline static std::optional<std::string> extract_actor(HttpRpcContext& ctx)
269 {
270 const auto path = ctx.get_method();
271
272 const auto first_slash = path.find_first_of('/');
273 const auto second_slash = path.find_first_of('/', first_slash + 1);
274
275 if (first_slash != 0 || second_slash == std::string::npos)
276 {
277 return std::nullopt;
278 }
279
280 const auto actor = path.substr(first_slash + 1, second_slash - 1);
281 const auto remaining_path = path.substr(second_slash);
282
283 if (actor.empty() || remaining_path.empty())
284 {
285 return std::nullopt;
286 }
287
288 ctx.set_method(remaining_path);
289 return actor;
290 }

295 inline std::shared_ptr<http::HttpRpcContext> make_rpc_context(
296 std::shared_ptr<ccf::SessionContext> s, const std::vector<uint8_t>& packed)
297 {
298 http::SimpleRequestProcessor processor;
299 http::RequestParser parser(processor);
300
301 parser.execute(packed.data(), packed.size());
302
303 if (processor.received.size() != 1)
304 {
305 throw std::logic_error(fmt::format(
306 "Expected packed to contain a single complete HTTP message. Actually "
307 "parsed {} messages",
308 processor.received.size()));
309 }
310
311 const auto& msg = processor.received.front();
312
313 return std::make_shared<http::HttpRpcContext>(
314 s, msg.method, msg.url, msg.headers, msg.body, packed);
315 }

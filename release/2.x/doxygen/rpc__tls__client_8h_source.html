47 std::vector<uint8_t> gen_http_request_internal(
48 const std::string& method,
49 const std::span<const uint8_t> params,
50 const std::string& content_type,
51 llhttp_method verb,
52 const char* auth_token = nullptr)
53 {
54 auto path = method;
55 if (prefix.has_value())
56 {
57 path = fmt::format("/{}/{}", prefix.value(), path);
58 }
59
60 auto r = http::Request(path, verb);
61 r.set_body(params.data(), params.size());
62 r.set_header(http::headers::CONTENT_TYPE, content_type);
63 if (auth_token != nullptr)
64 {
65 r.set_header(
66 http::headers::AUTHORIZATION, fmt::format("Bearer {}", auth_token));
67 }
68
69 if (key_pair != nullptr)
70 {
71 http::sign_request(r, key_pair, key_id);
72 }
73
74 return r.build_request();
75 }

179 Response get(
180 const std::string& method, const nlohmann::json& params = nullptr)
181 {
182 // GET body is ignored, so params must be placed in query
183 auto full_path = method;
184 if (!params.is_null())
185 {
186 for (auto it = params.begin(); it != params.end(); ++it)
187 {
188 full_path += fmt::format(
189 "{}{}={}",
190 it == params.begin() ? "?" : "&",
191 it.key(),
192 it.value().is_string() ? it.value().get<std::string>() :
193 it.value().dump());
194 }
195 }
196 return call(full_path, nullptr, HTTP_GET);
197 }

200 {

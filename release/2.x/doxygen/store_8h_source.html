267 void add_dynamic_map(
268 kv::Version v, const std::shared_ptr<AbstractMap>& map_) override
269 {
270 auto map = std::dynamic_pointer_cast<kv::untyped::Map>(map_);

273 throw std::logic_error(fmt::format(
274 "Can't add dynamic map - {} is not of expected type",
275 map_->get_name()));
276 }
277
278 const auto map_name = map->get_name();
279 if (get_map(v, map_name) != nullptr)

282 "Can't add dynamic map - already have a map named {}", map_name));
283 }
284
285 LOG_DEBUG_FMT("Adding newly created map '{}' at version {}", map_name, v);
286 maps[map_name] = std::make_pair(v, map);

288 {
289 // If we have any hooks for the given map name, set them on this new map
290 const auto global_it = global_hooks.find(map_name);
291 if (global_it != global_hooks.end())
292 {
293 map->set_global_hook(global_it->second);
294 }
295
296 const auto map_it = map_hooks.find(map_name);
297 if (map_it != map_hooks.end())
298 {
299 map->set_map_hook(map_it->second);
300 }
301 }

335 std::unique_ptr<AbstractSnapshot> snapshot(Version v) override
336 {
337 auto cv = compacted_version();
338 if (v < cv)
339 {
340 throw std::logic_error(fmt::format(
341 "Cannot snapshot at version {} which is earlier than last "
342 "compacted version {} ",
343 v,
344 cv));
345 }
346
347 if (v > current_version())
348 {
349 throw std::logic_error(fmt::format(
350 "Cannot snapshot at version {} which is later than current "
351 "version {} ",
352 v,
353 current_version()));
354 }
355
356 auto snapshot = std::make_unique<StoreSnapshot>(v);
357
358 {
359 std::lock_guard<ccf::Mutex> mguard(maps_lock);
360
361 for (auto& it : maps)
362 {
363 auto& [_, map] = it.second;
364 map->lock();
365 }
366
367 for (auto& it : maps)
368 {
369 auto& [_, map] = it.second;
370 snapshot->add_map_snapshot(map->snapshot(v));
371 }
372
373 auto h = get_history();
374 if (h)
375 {
376 snapshot->add_hash_at_snapshot(h->get_raw_leaf(v));
377 }
378
379 auto c = get_consensus();
380 if (c)
381 {
382 snapshot->add_view_history(c->get_view_history(v));
383 }
384
385 for (auto& it : maps)
386 {
387 auto& [_, map] = it.second;
388 map->unlock();
389 }
390 }
391
392 return snapshot;
393 }

402 ApplyResult deserialise_snapshot(
403 const uint8_t* data,
404 size_t size,
405 kv::ConsensusHookPtrs& hooks,
406 std::vector<Version>* view_history = nullptr,
407 bool public_only = false) override
408 {
409 auto e = get_encryptor();
410 auto d = KvStoreDeserialiser(
411 e,
412 public_only ? kv::SecurityDomain::PUBLIC :
413 std::optional<kv::SecurityDomain>());
414
415 kv::Term term;
416 auto v_ = d.init(data, size, term, is_historical);
417 if (!v_.has_value())
418 {
419 LOG_FAIL_FMT("Initialisation of deserialise object failed");
420 return ApplyResult::FAIL;
421 }
422 auto v = v_.value();
423
424 std::lock_guard<ccf::Mutex> mguard(maps_lock);
425
426 for (auto& it : maps)
427 {
428 auto& [_, map] = it.second;
429 map->lock();
430 }
431
432 std::vector<uint8_t> hash_at_snapshot;
433 auto h = get_history();
434 if (h)
435 {
436 hash_at_snapshot = d.deserialise_raw();
437 }
438
439 std::vector<Version> view_history_;
440 if (view_history)
441 {
442 view_history_ = d.deserialise_view_history();
443 }
444
445 OrderedChanges changes;
446 MapCollection new_maps;
447
448 for (auto r = d.start_map(); r.has_value(); r = d.start_map())
449 {
450 const auto map_name = r.value();
451
452 std::shared_ptr<kv::untyped::Map> map = nullptr;
453
454 auto search = maps.find(map_name);
455 if (search == maps.end())
456 {
457 map = std::make_shared<kv::untyped::Map>(
458 this,
459 map_name,
460 get_security_domain(map_name),
461 is_map_replicated(map_name),
462 should_track_dependencies(map_name));
463 new_maps[map_name] = map;
464 LOG_DEBUG_FMT(
465 "Creating map {} while deserialising snapshot at version {}",
466 map_name,
467 v);
468 }
469 else
470 {
471 map = search->second.second;
472 }
473
474 auto changes_search = changes.find(map_name);
475 if (changes_search != changes.end())
476 {
477 LOG_FAIL_FMT("Failed to deserialise snapshot at version {}", v);
478 LOG_DEBUG_FMT("Multiple writes on map {}", map_name);
479 return ApplyResult::FAIL;
480 }
481
482 auto deserialised_snapshot_changes =
483 map->deserialise_snapshot_changes(d);
484
485 // Take ownership of the produced change set, store it to be committed
486 // later
487 changes.emplace_hint(
488 changes_search,
489 std::piecewise_construct,
490 std::forward_as_tuple(map_name),
491 std::forward_as_tuple(map, std::move(deserialised_snapshot_changes)));
492 }
493
494 for (auto& it : maps)
495 {
496 auto& [_, map] = it.second;
497 map->unlock();
498 }
499
500 if (!d.end())
501 {
502 LOG_FAIL_FMT("Unexpected content in snapshot at version {}", v);
503 return ApplyResult::FAIL;
504 }
505
506 // Each map is committed at a different version, independently of the
507 // overall snapshot version. The commit versions for each map are
508 // contained in the snapshot and applied when the snapshot is committed.
509 auto r = apply_changes(
510 changes,
511 [](bool) { return std::make_tuple(NoVersion, NoVersion); },
512 hooks,
513 new_maps);
514 if (!r.has_value())
515 {
516 LOG_FAIL_FMT("Failed to commit deserialised snapshot at version {}", v);
517 return ApplyResult::FAIL;
518 }
519
520 {
521 std::lock_guard<ccf::Mutex> vguard(version_lock);
522 version = v;
523 last_replicated = v;
524 }
525
526 if (h)
527 {
528 if (!h->init_from_snapshot(hash_at_snapshot))
529 {
530 return ApplyResult::FAIL;
531 }
532 }
533
534 if (view_history)
535 {
536 *view_history = std::move(view_history_);
537 }
538
539 return ApplyResult::PASS;
540 }

542 void compact(Version v) override
543 {
544 // This is called when the store will never be rolled back to any
545 // state before the specified version.
546 // No transactions can be prepared or committed during compaction.
547
548 if (snapshotter)
549 {
550 auto c = get_consensus();
551 bool generate_snapshot = c && c->is_primary();
552 snapshotter->commit(v, generate_snapshot);
553 }
554
555 std::lock_guard<ccf::Mutex> mguard(maps_lock);
556
557 if (v > current_version())
558 {
559 return;
560 }
561
562 for (auto& it : maps)
563 {
564 auto& [_, map] = it.second;
565 map->lock();
566 }
567
568 for (auto& it : maps)
569 {
570 auto& [_, map] = it.second;
571 map->compact(v);
572 }
573
574 for (auto& it : maps)
575 {
576 auto& [_, map] = it.second;
577 map->unlock();
578 }
579
580 {
581 std::lock_guard<ccf::Mutex> vguard(version_lock);
582 compacted = v;
583
584 auto h = get_history();
585 if (h)
586 {
587 h->compact(v);
588 }
589 }
590
591 for (auto& it : maps)
592 {
593 auto& [_, map] = it.second;
594 map->post_compact();
595 }
596 }

598 void rollback(const TxID& tx_id, Term term_of_next_version_) override
599 {
600 // This is called to roll the store back to the state it was in
601 // at the specified version.
602 // No transactions can be prepared or committed during rollback.
603
604 if (snapshotter)
605 {
606 snapshotter->rollback(tx_id.version);
607 }
608
609 std::lock_guard<ccf::Mutex> mguard(maps_lock);
610
611 {
612 std::lock_guard<ccf::Mutex> vguard(version_lock);
613 if (tx_id.version < compacted)
614 {
615 throw std::logic_error(fmt::format(
616 "Attempting rollback to {}, earlier than commit version {}",
617 tx_id.version,
618 compacted));
619 }
620
621 // The term should always be updated on rollback() when passed
622 // regardless of whether version needs to be updated or not
623 term_of_next_version = term_of_next_version_;
624 term_of_last_version = tx_id.term;
625
626 // History must be informed of the term_of_last_version change, even if
627 // no actual rollback is required
628 auto h = get_history();
629 if (h)
630 {
631 h->rollback(tx_id, term_of_next_version);
632 }
633
634 if (tx_id.version >= version)
635 {
636 return;
637 }
638
639 version = tx_id.version;
640 last_replicated = tx_id.version;
641 unset_flag_unsafe(Flag::LEDGER_CHUNK_AT_NEXT_SIGNATURE);
642 unset_flag_unsafe(Flag::SNAPSHOT_AT_NEXT_SIGNATURE);
643 rollback_count++;
644 pending_txs.clear();
645 auto e = get_encryptor();
646 if (e)
647 {
648 e->rollback(tx_id.version);
649 }
650 }
651
652 for (auto& it : maps)
653 {
654 auto& [_, map] = it.second;
655 map->lock();
656 }
657
658 auto it = maps.begin();
659 while (it != maps.end())
660 {
661 auto& [map_creation_version, map] = it->second;
662 // Rollback this map whether we're forgetting about it or not. Anyone
663 // else still holding it should see it has rolled back
664 map->rollback(tx_id.version);
665 if (map_creation_version > tx_id.version)
666 {
667 // Map was created more recently; its creation is being forgotten.
668 // Erase our knowledge of it
669 map->unlock();
670 it = maps.erase(it);
671 }
672 else
673 {
674 ++it;
675 }
676 }
677
678 for (auto& it : maps)
679 {
680 auto& [_, map] = it.second;
681 map->unlock();
682 }
683 }

685 void initialise_term(Term t) override
686 {
687 // Note: This should only be called once, when the store is first
688 // initialised. term_of_next_version is later updated via rollback.
689 std::lock_guard<ccf::Mutex> vguard(version_lock);
690 if (term_of_next_version != 0)
691 {
692 throw std::logic_error("term_of_next_version is already initialised");
693 }
694
695 term_of_next_version = t;
696 auto h = get_history();
697 if (h)
698 {
699 h->set_term(term_of_next_version);
700 }

703 bool fill_maps(
704 const std::vector<uint8_t>& data,
705 bool public_only,
706 kv::Version& v,
707 kv::Term& view,
708 OrderedChanges& changes,
709 MapCollection& new_maps,
710 ccf::ClaimsDigest& claims_digest,
711 std::optional<crypto::Sha256Hash>& commit_evidence_digest,
712 bool ignore_strict_versions = false) override
713 {
714 // This will return FAILED if the serialised transaction is being
715 // applied out of order.
716 // Processing transactions locally and also deserialising to the
717 // same store will result in a store version mismatch and
718 // deserialisation will then fail.
719 auto e = get_encryptor();
720
721 auto d = KvStoreDeserialiser(
722 e,
723 public_only ? kv::SecurityDomain::PUBLIC :
724 std::optional<kv::SecurityDomain>());
725
726 auto v_ = d.init(data.data(), data.size(), view, is_historical);
727 if (!v_.has_value())
728 {
729 LOG_FAIL_FMT("Initialisation of deserialise object failed");
730 return false;
731 }
732 v = v_.value();
733
734 claims_digest = std::move(d.consume_claims_digest());
735 LOG_TRACE_FMT(
736 "Deserialised claim digest {} {}",
737 claims_digest.value(),
738 claims_digest.empty());
739
740 commit_evidence_digest = std::move(d.consume_commit_evidence_digest());
741 if (commit_evidence_digest.has_value())
742 LOG_TRACE_FMT(
743 "Deserialised commit evidence digest {}",
744 commit_evidence_digest.value());
745
746 // Throw away any local commits that have not propagated via the
747 // consensus.
748 rollback({term_of_last_version, v - 1}, term_of_next_version);
749
750 if (strict_versions && !ignore_strict_versions)
751 {
752 // Make sure this is the next transaction.
753 auto cv = current_version();
754 if (cv != (v - 1))
755 {
756 LOG_FAIL_FMT(
757 "Tried to deserialise {} but current_version is {}", v, cv);
758 return false;
759 }
760 }
761
762 // Deserialised transactions express read dependencies as versions,
763 // rather than with the actual value read. As a result, they don't
764 // need snapshot isolation on the map state, and so do not need to
765 // lock each of the maps before creating the transaction.
766 std::lock_guard<ccf::Mutex> mguard(maps_lock);
767
768 for (auto r = d.start_map(); r.has_value(); r = d.start_map())
769 {
770 const auto map_name = r.value();
771
772 auto map = get_map_internal(v, map_name);
773 if (map == nullptr)
774 {
775 auto new_map = std::make_shared<kv::untyped::Map>(
776 this,
777 map_name,
778 get_security_domain(map_name),
779 is_map_replicated(map_name),
780 should_track_dependencies(map_name));
781 map = new_map;
782 new_maps[map_name] = new_map;
783 LOG_DEBUG_FMT(
784 "Creating map '{}' while deserialising transaction at version {}",
785 map_name,
786 v);
787 }
788
789 auto change_search = changes.find(map_name);
790 if (change_search != changes.end())
791 {
792 LOG_FAIL_FMT("Failed to deserialise transaction at version {}", v);
793 LOG_DEBUG_FMT("Multiple writes on map {}", map_name);
794 return false;
795 }
796
797 auto deserialised_changes = map->deserialise_changes(d, v);
798
799 // Take ownership of the produced change set, store it to be applied
800 // later
801 changes.emplace_hint(
802 change_search,
803 std::piecewise_construct,
804 std::forward_as_tuple(map_name),
805 std::forward_as_tuple(map, std::move(deserialised_changes)));
806 }
807
808 if (!d.end())
809 {
810 LOG_FAIL_FMT("Unexpected content in transaction at version {}", v);
811 return false;
812 }
813
814 return true;
815 }

836 bool operator==(const Store& that) const
837 {
838 // Only used for debugging, not thread safe.
839 if (version != that.version)
840 return false;
841
842 if (maps.size() != that.maps.size())
843 return false;
844
845 for (auto it = maps.begin(); it != maps.end(); ++it)
846 {
847 auto search = that.maps.find(it->first);
848
849 if (search == that.maps.end())
850 return false;
851
852 auto& [this_v, this_map] = it->second;
853 auto& [that_v, that_map] = search->second;
854
855 if (this_v != that_v)
856 return false;
857
858 if (*this_map != *that_map)
859 return false;
860 }
861
862 return true;
863 }

906 CommitResult commit(
907 const TxID& txid,
908 std::unique_ptr<PendingTx> pending_tx,
909 bool globally_committable) override
910 {
911 auto c = get_consensus();
912 if (!c)
913 {
914 return CommitResult::SUCCESS;
915 }
916
917 std::lock_guard<ccf::Mutex> cguard(commit_lock);
918
919 LOG_DEBUG_FMT(
920 "Store::commit {}{}",
921 txid.version,
922 (globally_committable ? " globally_committable" : ""));
923
924 BatchVector batch;
925 Version previous_last_replicated = 0;
926 Version next_last_replicated = 0;
927 Version previous_rollback_count = 0;
928 ccf::View replication_view = 0;
929
930 {
931 std::lock_guard<ccf::Mutex> vguard(version_lock);
932 if (txid.term != term_of_next_version && consensus->is_primary())
933 {
934 // This can happen when a transaction started before a view change,
935 // but tries to commit after the view change is complete.
936 LOG_DEBUG_FMT(
937 "Want to commit for term {} but term is {}",
938 txid.term,
939 term_of_next_version);
940
941 return CommitResult::FAIL_NO_REPLICATE;
942 }
943
944 if (globally_committable && txid.version > last_committable)
945 {
946 last_committable = txid.version;
947 }
948
949 pending_txs.insert(
950 {txid.version,
951 std::make_tuple(std::move(pending_tx), globally_committable)});
952
953 LOG_TRACE_FMT("Inserting pending tx at {}", txid.version);
954
955 auto h = get_history();
956 auto c = get_consensus();
957
958 for (Version offset = 1; true; ++offset)
959 {
960 auto search = pending_txs.find(last_replicated + offset);
961 if (search == pending_txs.end())
962 {
963 LOG_TRACE_FMT(
964 "Couldn't find {} = {} + {}, giving up on batch while committing "
965 "{}.{}",
966 last_replicated + offset,
967 last_replicated,
968 offset,
969 txid.term,
970 txid.version);
971 break;
972 }
973
974 auto& [pending_tx_, committable_] = search->second;
975 auto
976 [success_, data_, claims_digest_, commit_evidence_digest_, hooks_] =
977 pending_tx_->call();
978 auto data_shared =
979 std::make_shared<std::vector<uint8_t>>(std::move(data_));
980 auto hooks_shared =
981 std::make_shared<kv::ConsensusHookPtrs>(std::move(hooks_));
982
983 // NB: this cannot happen currently. Regular Tx only make it here if
984 // they did succeed, and signatures cannot conflict because they
985 // execute in order with a read_version that's version - 1, so even
986 // two contiguous signatures are fine
987 if (success_ != CommitResult::SUCCESS)
988 {
989 LOG_DEBUG_FMT("Failed Tx commit {}", last_replicated + offset);
990 }
991
992 if (h)
993 {
994 h->append_entry(ccf::entry_leaf(
995 *data_shared, commit_evidence_digest_, claims_digest_));
996 }
997
998 LOG_DEBUG_FMT(
999 "Batching {} ({}) during commit of {}.{}",
1000 last_replicated + offset,
1001 data_shared->size(),
1002 txid.term,
1003 txid.version);
1004
1005 batch.emplace_back(
1006 last_replicated + offset, data_shared, committable_, hooks_shared);
1007 pending_txs.erase(search);
1008 }
1009
1010 if (batch.size() == 0)
1011 {
1012 return CommitResult::SUCCESS;
1013 }
1014
1015 previous_rollback_count = rollback_count;
1016 previous_last_replicated = last_replicated;
1017 next_last_replicated = last_replicated + batch.size();
1018
1019 replication_view = term_of_next_version;
1020
1021 if (consensus->type() == ConsensusType::BFT && consensus->is_backup())
1022 {
1023 last_replicated = next_last_replicated;
1024 }
1025 }
1026
1027 if (c->replicate(batch, replication_view))
1028 {
1029 std::lock_guard<ccf::Mutex> vguard(version_lock);
1030 if (
1031 last_replicated == previous_last_replicated &&
1032 previous_rollback_count == rollback_count &&
1033 !(consensus->type() == ConsensusType::BFT && consensus->is_backup()))
1034 {
1035 last_replicated = next_last_replicated;
1036 }
1037 return CommitResult::SUCCESS;
1038 }
1039 else
1040 {
1041 LOG_DEBUG_FMT("Failed to replicate");
1042 return CommitResult::FAIL_NO_REPLICATE;
1043 }
1044 }

1125 void swap_private_maps(Store& store)
1126 {
1127 {
1128 const auto source_version = store.current_version();
1129 const auto target_version = current_version();
1130 if (source_version > target_version)
1131 {
1132 throw std::runtime_error(fmt::format(
1133 "Invalid call to swap_private_maps. Source is at version {} while "
1134 "target is at {}",
1135 source_version,
1136 target_version));
1137 }
1138 }
1139
1140 std::lock_guard<ccf::Mutex> this_maps_guard(maps_lock);
1141 std::lock_guard<ccf::Mutex> other_maps_guard(store.maps_lock);
1142
1143 // Each entry is (Name, MyMap, TheirMap)
1144 using MapEntry = std::tuple<std::string, AbstractMap*, AbstractMap*>;
1145 std::vector<MapEntry> entries;
1146
1147 // Get the list of private maps from the source store
1148 for (auto& [name, pair] : store.maps)
1149 {
1150 auto& [_, map] = pair;
1151 if (map->get_security_domain() == SecurityDomain::PRIVATE)
1152 {
1153 map->lock();
1154 entries.emplace_back(name, nullptr, map.get());
1155 }
1156 }
1157
1158 // For each source map, either create it or, where it already exists,
1159 // confirm it is PRIVATE. Lock it and store it in entries
1160 auto entry = entries.begin();
1161 while (entry != entries.end())
1162 {
1163 const auto& [name, _, their_map] = *entry;
1164 std::shared_ptr<AbstractMap> map = nullptr;
1165 const auto it = maps.find(name);
1166 if (it == maps.end())
1167 {
1168 // NB: We lose the creation version from the original map, but assume
1169 // it is irrelevant - its creation should no longer be at risk of
1170 // rollback
1171 auto new_map = std::make_pair(
1172 NoVersion,
1173 std::make_shared<kv::untyped::Map>(
1174 this,
1175 name,
1176 SecurityDomain::PRIVATE,
1177 is_map_replicated(name),
1178 should_track_dependencies(name)));
1179 maps[name] = new_map;
1180 map = new_map.second;
1181 }
1182 else
1183 {
1184 map = it->second.second;
1185 if (map->get_security_domain() != SecurityDomain::PRIVATE)
1186 {
1187 throw std::logic_error(fmt::format(
1188 "Swap mismatch - map {} is private in source but not in target",
1189 name));
1190 }
1191 }
1192
1193 std::get<1>(*entry) = map.get();
1194 map->lock();
1195 ++entry;
1196 }
1197
1198 for (auto& [name, lhs, rhs] : entries)
1199 {
1200 lhs->swap(rhs);
1201 }
1202
1203 for (auto& [name, lhs, rhs] : entries)
1204 {
1205 lhs->unlock();
1206 rhs->unlock();
1207 }
1208 }

88 int handshake()
89 {
90 if (SSL_is_init_finished(ssl))
91 return 0;
92
93 int rc = SSL_do_handshake(ssl);
94 // Success in OpenSSL is 1, MBed is 0
95 if (rc > 0)
96 {
97 LOG_TRACE_FMT("Context::handshake() : Success");
98 return 0;
99 }
100
101 // Want read/write needs special return
102 if (SSL_want_read(ssl))
103 {
104 return TLS_ERR_WANT_READ;
105 }
106 else if (SSL_want_write(ssl))
107 {
108 return TLS_ERR_WANT_WRITE;
109 }
110
111 // So does x509 validation
112 if (!peer_cert_ok())
113 {
114 return TLS_ERR_X509_VERIFY;
115 }
116
117 // Everything else falls here.
118 LOG_TRACE_FMT("Context::handshake() : Error code {}", rc);
119
120 // As an MBedTLS emulation, we return negative for errors.
121 return -SSL_get_error(ssl, rc);
122 }

189 std::vector<uint8_t> peer_cert()
190 {
191 // CodeQL complains that we don't verify the peer certificate. We don't
192 // need to do that because it's been verified before and we use
193 // SSL_get_peer_certificate just to extract it from the context.
194
195 crypto::OpenSSL::Unique_X509 cert(
196 SSL_get_peer_certificate(ssl), /*check_null=*/false);
197 if (!cert)
198 {
199 LOG_TRACE_FMT("Empty peer cert");
200 return {};
201 }
202 crypto::OpenSSL::Unique_BIO bio;
203 if (!i2d_X509_bio(bio, cert))
204 {
205 LOG_TRACE_FMT("Can't convert X509 to DER");
206 return {};
207 }
208
209 // Get the total length of the DER representation
210 auto len = BIO_get_mem_data(bio, nullptr);
211 if (!len)
212 {
213 LOG_TRACE_FMT("Null X509 peer cert");
214 return {};
215 }
216
217 // Get the BIO memory pointer
218 BUF_MEM* ptr = nullptr;
219 if (!BIO_get_mem_ptr(bio, &ptr))
220 {
221 LOG_TRACE_FMT("Invalid X509 peer cert");
222 return {};
223 }
224
225 // Return its contents as a vector
226 auto ret = std::vector<uint8_t>(ptr->data, ptr->data + len);
227 return ret;
228 }

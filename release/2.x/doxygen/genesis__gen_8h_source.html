71 std::map<MemberId, crypto::Pem> get_active_recovery_members()
72 {
73 auto member_info = tx.ro(tables.member_info);
74 auto member_encryption_public_keys =
75 tx.ro(tables.member_encryption_public_keys);
76
77 std::map<MemberId, crypto::Pem> active_recovery_members;
78
79 member_encryption_public_keys->foreach(
80 [&active_recovery_members,
81 &member_info](const auto& mid, const auto& pem) {
82 auto info = member_info->get(mid);
83 if (!info.has_value())
84 {
85 throw std::logic_error(
86 fmt::format("Recovery member {} has no member info", mid));
87 }
88
89 if (info->status == MemberStatus::ACTIVE)
90 {
91 active_recovery_members[mid] = pem;
92 }
93 return true;
94 });
95 return active_recovery_members;
96 }

98 MemberId add_member(const NewMember& member_pub_info)
99 {
100 auto member_certs = tx.rw(tables.member_certs);
101 auto member_info = tx.rw(tables.member_info);
102 auto member_acks = tx.rw(tables.member_acks);
103 auto signatures = tx.ro(tables.signatures);
104
105 auto member_cert_der =
106 crypto::make_verifier(member_pub_info.cert)->cert_der();
107 auto id = crypto::Sha256Hash(member_cert_der).hex_str();
108
109 auto member = member_certs->get(id);
110 if (member.has_value())
111 {
112 // No effect if member already exists
113 return id;
114 }
115
116 member_certs->put(id, member_pub_info.cert);
117 member_info->put(
118 id, {MemberStatus::ACCEPTED, member_pub_info.member_data});
119
120 if (member_pub_info.encryption_pub_key.has_value())
121 {
122 auto member_encryption_public_keys =
123 tx.rw(tables.member_encryption_public_keys);
124 member_encryption_public_keys->put(
125 id, member_pub_info.encryption_pub_key.value());
126 }
127
128 auto s = signatures->get();
129 if (!s)
130 {
131 member_acks->put(id, MemberAck());
132 }
133 else
134 {
135 member_acks->put(id, MemberAck(s->root));
136 }
137 return id;
138 }

165 bool remove_member(const MemberId& member_id)
166 {
167 auto member_certs = tx.rw(tables.member_certs);
168 auto member_encryption_public_keys =
169 tx.rw(tables.member_encryption_public_keys);
170 auto member_info = tx.rw(tables.member_info);
171 auto member_acks = tx.rw(tables.member_acks);
172 auto member_gov_history = tx.rw(tables.governance_history);
173
174 auto member_to_remove = member_info->get(member_id);
175 if (!member_to_remove.has_value())
176 {
177 // The remove member proposal is idempotent so if the member does not
178 // exist, the proposal should succeed with no effect
179 LOG_FAIL_FMT(
180 "Could not remove member {}: member does not exist", member_id);
181 return true;
182 }
183
184 // If the member was active and had a recovery share, check that
185 // the new number of active members is still sufficient for
186 // recovery
187 if (
188 member_to_remove->status == MemberStatus::ACTIVE &&
189 is_recovery_member(member_id))
190 {
191 // Because the member to remove is active, there is at least one active
192 // member (i.e. get_active_recovery_members_count_after >= 0)
193 size_t get_active_recovery_members_count_after =
194 get_active_recovery_members().size() - 1;
195 auto recovery_threshold = get_recovery_threshold();
196 if (get_active_recovery_members_count_after < recovery_threshold)
197 {
198 LOG_FAIL_FMT(
199 "Failed to remove recovery member {}: number of active recovery "
200 "members ({}) would be less than recovery threshold ({})",
201 member_id,
202 get_active_recovery_members_count_after,
203 recovery_threshold);
204 return false;
205 }
206 }
207
208 member_info->remove(member_id);
209 member_encryption_public_keys->remove(member_id);
210 member_certs->remove(member_id);
211 member_acks->remove(member_id);
212 member_gov_history->remove(member_id);
213
214 return true;
215 }

217 UserId add_user(const NewUser& new_user)
218 {
219 auto user_certs = tx.rw(tables.user_certs);
220
221 auto user_cert_der = crypto::make_verifier(new_user.cert)->cert_der();
222 auto id = crypto::Sha256Hash(user_cert_der).hex_str();
223
224 auto user_cert = user_certs->get(id);
225 if (user_cert.has_value())
226 {
227 throw std::logic_error(
228 fmt::format("Certificate already exists for user {}", id));
229 }
230
231 user_certs->put(id, new_user.cert);
232
233 if (new_user.user_data != nullptr)
234 {
235 auto user_info = tx.rw(tables.user_info);
236 auto ui = user_info->get(id);
237 if (ui.has_value())
238 {
239 throw std::logic_error(
240 fmt::format("User data already exists for user {}", id));
241 }
242
243 user_info->put(id, {new_user.user_data});
244 }
245
246 return id;
247 }

265 auto get_trusted_and_learner_nodes(
266 std::optional<NodeId> self_to_exclude = std::nullopt)
267 {
268 // Returns the list of trusted nodes. If self_to_exclude is set,
269 // self_to_exclude is not included in the list of returned nodes.
270 std::map<NodeId, NodeInfo> active_nodes;
271
272 auto nodes = tx.ro(tables.nodes);
273
274 nodes->foreach([&active_nodes,
275 self_to_exclude](const NodeId& nid, const NodeInfo& ni) {
276 if (
277 (ni.status == ccf::NodeStatus::TRUSTED ||
278 ni.status == ccf::NodeStatus::LEARNER) &&
279 (!self_to_exclude.has_value() || self_to_exclude.value() != nid))
280 {
281 active_nodes[nid] = ni;
282 }
283 return true;
284 });
285
286 return active_nodes;
287 }

289 // Service status should use a state machine, very much like NodeState.
290 void create_service(
291 const crypto::Pem& service_cert,
292 ccf::TxID create_txid,
293 nlohmann::json service_data = nullptr,
294 bool recovering = false)
295 {
296 auto service = tx.rw(tables.service);
297
298 size_t recovery_count = 0;
299
300 if (service->has())
301 {
302 const auto prev_service_info = service->get();
303 auto previous_service_identity = tx.wo<ccf::PreviousServiceIdentity>(
304 ccf::Tables::PREVIOUS_SERVICE_IDENTITY);
305 previous_service_identity->put(prev_service_info->cert);
306
307 // Record number of recoveries for service. If the value does
308 // not exist in the table (i.e. pre 2.x ledger), assume it is the first
309 // recovery.
310 recovery_count = prev_service_info->recovery_count.value_or(0) + 1;
311 }
312
313 service->put(
314 {service_cert,
315 recovering ? ServiceStatus::RECOVERING : ServiceStatus::OPENING,
316 recovering ? service->get_version_of_previous_write() : std::nullopt,
317 recovery_count,
318 service_data,
319 create_txid});
320 }

328 bool open_service()
329 {
330 auto service = tx.rw(tables.service);
331
332 auto active_recovery_members_count = get_active_recovery_members().size();
333 if (active_recovery_members_count < get_recovery_threshold())
334 {
335 LOG_FAIL_FMT(
336 "Cannot open network as number of active recovery members ({}) is "
337 "less than recovery threshold ({})",
338 active_recovery_members_count,
339 get_recovery_threshold());
340 return false;
341 }
342
343 auto active_service = service->get();
344 if (!active_service.has_value())
345 {
346 LOG_FAIL_FMT("Failed to get active service");
347 return false;
348 }
349
350 if (active_service->status == ServiceStatus::OPEN)
351 {
352 // If the service is already open, return with no effect
353 return true;
354 }
355
356 if (
357 active_service->status != ServiceStatus::OPENING &&
358 active_service->status != ServiceStatus::WAITING_FOR_RECOVERY_SHARES)
359 {
360 LOG_FAIL_FMT(
361 "Could not open current service: status is not OPENING or "
362 "WAITING_FOR_RECOVERY_SHARES");
363 return false;
364 }
365
366 active_service->status = ServiceStatus::OPEN;
367 active_service->previous_service_identity_version =
368 service->get_version_of_previous_write();
369 service->put(active_service.value());
370
371 return true;
372 }

387 void trust_node(
388 const NodeId& node_id, kv::Version latest_ledger_secret_seqno)
389 {
390 auto nodes = tx.rw(tables.nodes);
391 auto node_info = nodes->get(node_id);
392
393 if (!node_info.has_value())
394 {
395 throw std::logic_error(fmt::format("Node {} does not exist", node_id));
396 }
397
398 if (node_info->status == NodeStatus::RETIRED)
399 {
400 throw std::logic_error(fmt::format("Node {} is retired", node_id));
401 }
402
403 node_info->status = NodeStatus::TRUSTED;
404 node_info->ledger_secret_seqno = latest_ledger_secret_seqno;
405 nodes->put(node_id, node_info.value());
406
407 LOG_INFO_FMT("Node {} is now {}", node_id, node_info->status);
408 }

434 bool set_recovery_threshold(size_t threshold)
435 {
436 auto config = tx.rw(tables.config);
437
438 if (threshold == 0)
439 {
440 LOG_FAIL_FMT("Cannot set recovery threshold to 0");
441 return false;
442 }
443
444 auto service_status = get_service_status();
445 if (!service_status.has_value())
446 {
447 LOG_FAIL_FMT("Failed to get active service");
448 return false;
449 }
450
451 if (service_status.value() == ServiceStatus::WAITING_FOR_RECOVERY_SHARES)
452 {
453 // While waiting for recovery shares, the recovery threshold cannot be
454 // modified. Otherwise, the threshold could be passed without triggering
455 // the end of recovery procedure
456 LOG_FAIL_FMT(
457 "Cannot set recovery threshold: service is currently waiting for "
458 "recovery shares");
459 return false;
460 }
461 else if (service_status.value() == ServiceStatus::OPEN)
462 {
463 auto get_active_recovery_members_count =
464 get_active_recovery_members().size();
465 if (threshold > get_active_recovery_members_count)
466 {
467 LOG_FAIL_FMT(
468 "Cannot set recovery threshold to {} as it is greater than the "
469 "number of active recovery members ({})",
470 threshold,
471 get_active_recovery_members_count);
472 return false;
473 }
474 }
475
476 auto current_config = config->get();
477 if (!current_config.has_value())
478 {
479 throw std::logic_error("Configuration should already be set");
480 }
481
482 current_config->recovery_threshold = threshold;
483 config->put(current_config.value());
484 return true;
485 }
486
487 size_t get_recovery_threshold()
488 {
489 auto config = tx.ro(tables.config);
490 auto current_config = config->get();
491 if (!current_config.has_value())
492 {
493 throw std::logic_error(
494 "Failed to get recovery threshold: No active configuration found");
495 }
496 return current_config->recovery_threshold;
497 }

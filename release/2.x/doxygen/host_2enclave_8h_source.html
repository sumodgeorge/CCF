42 void expect_enclave_file_suffix(
43 const std::string& file,
44 char const* expected_suffix,
45 host::EnclaveType type)
46 {
47 if (!nonstd::ends_with(file, expected_suffix))
48 {
49 // Remove possible suffixes to try and get root of filename, to build
50 // suggested filename
51 auto basename = file;
52 for (const char* suffix :
53 {".signed", ".debuggable", ".so", ".enclave", ".virtual"})
54 {
55 if (nonstd::ends_with(basename, suffix))
56 {
57 basename = basename.substr(0, basename.size() - strlen(suffix));
58 }
59 }
60 const auto suggested = fmt::format("{}{}", basename, expected_suffix);
61 throw std::logic_error(fmt::format(
62 "Given enclave file '{}' does not have suffix expected for enclave "
63 "type "
64 "{}. Did you mean '{}'?",
65 file,
66 nlohmann::json(type).dump(),
67 suggested));
68 }
69 }

94 Enclave(const std::string& path, EnclaveType type)
95 {
96 if (!std::filesystem::exists(path))
97 {
98 throw std::logic_error(
99 fmt::format("No enclave file found at {}", path));
100 }
101
102 switch (type)
103 {
104 case host::EnclaveType::SGX_RELEASE:
105 case host::EnclaveType::SGX_DEBUG:
106 {
107#ifdef CCHOST_SUPPORTS_SGX
108 uint32_t oe_flags = 0;
109 if (type == host::EnclaveType::SGX_DEBUG)
110 {
111 expect_enclave_file_suffix(path, ".enclave.so.debuggable", type);
112 oe_flags |= OE_ENCLAVE_FLAG_DEBUG;
113 }
114 else
115 {
116 expect_enclave_file_suffix(path, ".enclave.so.signed", type);
117 }
118
119# ifndef VERBOSE_LOGGING
120 oe_log_set_callback(nullptr, nop_oe_logger);
121# endif
122
123 auto err = oe_create_ccf_enclave(
124 path.c_str(),
125 OE_ENCLAVE_TYPE_SGX,
126 oe_flags,
127 nullptr,
128 0,
129 &sgx_handle);
130
131 if (err != OE_OK)
132 {
133 throw std::logic_error(
134 fmt::format("Could not create enclave: {}", oe_result_str(err)));
135 }
136#else
137 throw std::logic_error(
138 "SGX enclaves are not supported in current build");
139#endif // CCHOST_SUPPORTS_SGX
140 break;
141 }
142
143 case host::EnclaveType::VIRTUAL:
144 {
145#ifdef CCHOST_SUPPORTS_VIRTUAL
146 expect_enclave_file_suffix(path, ".virtual.so", type);
147 virtual_handle = load_virtual_enclave(path.c_str());
148#else
149 throw std::logic_error(
150 "Virtual enclaves not supported in current build");
151#endif // CCHOST_SUPPORTS_VIRTUAL
152 break;
153 }
154
155 default:
156 {
157 throw std::logic_error(fmt::format(
158 "Unsupported enclave type: {}", nlohmann::json(type).dump()));
159 }
160 }
161 }

163 CreateNodeStatus create_node(
164 const EnclaveConfig& enclave_config,
165 const StartupConfig& ccf_config,
166 std::vector<uint8_t>& node_cert,
167 std::vector<uint8_t>& service_cert,
168 StartType start_type,
169 size_t num_worker_thread,
170 void* time_location)
171 {
172 CreateNodeStatus status = CreateNodeStatus::InternalError;
173 constexpr size_t enclave_version_size = 256;
174 std::vector<uint8_t> enclave_version_buf(enclave_version_size);
175
176 size_t node_cert_len = 0;
177 size_t service_cert_len = 0;
178 size_t enclave_version_len = 0;
179
180 auto config = nlohmann::json(ccf_config).dump();
181
182#define CREATE_NODE_ARGS \
183 &status, (void*)&enclave_config, config.data(), config.size(), \
184 node_cert.data(), node_cert.size(), &node_cert_len, service_cert.data(), \
185 service_cert.size(), &service_cert_len, enclave_version_buf.data(), \
186 enclave_version_buf.size(), &enclave_version_len, start_type, \
187 num_worker_thread, time_location
188
189 oe_result_t err = OE_FAILURE;
190
191// Assume that constructor correctly set the appropriate field, and call
192// appropriate function
193#ifdef CCHOST_SUPPORTS_VIRTUAL
194 if (virtual_handle != nullptr)
195 {
196 err = virtual_create_node(virtual_handle, CREATE_NODE_ARGS);
197 }
198#endif
199#ifdef CCHOST_SUPPORTS_SGX
200 if (sgx_handle != nullptr)
201 {
202 err = enclave_create_node(sgx_handle, CREATE_NODE_ARGS);
203 }
204#endif
205
206 if (err != OE_OK || status != CreateNodeStatus::OK)
207 {
208 // Logs have described the errors already, we just need to allow the
209 // host to read them (via read_all()).
210 return status;
211 }
212
213 // Host and enclave versions must match. Otherwise the node may crash much
214 // later (e.g. unhandled ring buffer message on either end)
215 auto enclave_version = std::string(
216 enclave_version_buf.begin(),
217 enclave_version_buf.begin() + enclave_version_len);
218 if (ccf::ccf_version != enclave_version)
219 {
220 LOG_FAIL_FMT(
221 "Host/Enclave versions mismatch: {} != {}",
222 ccf::ccf_version,
223 enclave_version);
224 return CreateNodeStatus::VersionMismatch;
225 }
226
227 node_cert.resize(node_cert_len);
228 service_cert.resize(service_cert_len);
229
230 return CreateNodeStatus::OK;
231 }
232
233 // Run a processor over this circuit inside the enclave - should be called
234 // from a thread
235 bool run()
236 {
237 bool ret = true;
238 oe_result_t err = OE_FAILURE;
239
240#ifdef CCHOST_SUPPORTS_VIRTUAL
241 if (virtual_handle != nullptr)
242 {
243 err = virtual_run(virtual_handle, &ret);
244 }
245#endif
246#ifdef CCHOST_SUPPORTS_SGX
247 if (sgx_handle != nullptr)
248 {
249 err = enclave_run(sgx_handle, &ret);
250 }
251#endif
252
253 if (err != OE_OK)
254 {
255 throw std::logic_error(
256 fmt::format("Failed to call in enclave_run: {}", oe_result_str(err)));
257 }
258
259 return ret;
260 }
261 };

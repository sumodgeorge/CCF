19 class ConfigurationChangeHook : public kv::ConsensusHook
20 {
21 kv::Version version;
22 std::map<NodeId, std::optional<NodeAddr>> cfg_delta;
23 std::unordered_set<NodeId> learners;
24 std::unordered_set<NodeId> retired_nodes;
25
26 public:
27 ConfigurationChangeHook(kv::Version version_, const Nodes::Write& w) :
28 version(version_)
29 {
30 for (const auto& [node_id, opt_ni] : w)
31 {
32 if (!opt_ni.has_value())
33 {
34 // Deleted node will have already been retired
35 continue;
36 }
37
38 const auto& ni = opt_ni.value();
39 const auto [host, port] =
40 split_net_address(ni.node_to_node_interface.bind_address);
41 switch (ni.status)
42 {
43 case NodeStatus::PENDING:
44 {
45 // Pending nodes are not added to consensus until they are
46 // trusted
47 break;
48 }
49 case NodeStatus::TRUSTED:
50 {
51 cfg_delta.try_emplace(node_id, NodeAddr{host, port});
52 break;
53 }
54 case NodeStatus::RETIRED:
55 {
56 cfg_delta.try_emplace(node_id, std::nullopt);
57 retired_nodes.insert(node_id);
58 break;
59 }
60 case NodeStatus::LEARNER:
61 {
62 cfg_delta.try_emplace(node_id, NodeAddr{host, port});
63 learners.insert(node_id);
64 break;
65 }
66 case NodeStatus::RETIRING:
67 {
68 cfg_delta.try_emplace(node_id, std::nullopt);
69 break;
70 }
71 default:
72 {
73 }
74 }
75 }
76 }
77
78 void call(kv::ConfigurableConsensus* consensus) override
79 {
80 auto configuration = consensus->get_latest_configuration_unsafe();
81 for (const auto& [node_id, opt_ni] : cfg_delta)
82 {
83 if (opt_ni.has_value())
84 {
85 configuration.try_emplace(node_id, opt_ni->hostname, opt_ni->port);
86 }
87 else
88 {
89 configuration.erase(node_id);
90 }
91 }
92 if (!cfg_delta.empty())
93 {
94 consensus->add_configuration(
95 version, configuration, learners, retired_nodes);
96 }
97 }
98 };

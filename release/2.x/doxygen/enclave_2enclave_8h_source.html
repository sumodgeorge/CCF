180 CreateNodeStatus create_new_node(
181 StartType start_type_,
182 StartupConfig&& ccf_config_,
183 uint8_t* node_cert,
184 size_t node_cert_size,
185 size_t* node_cert_len,
186 uint8_t* service_cert,
187 size_t service_cert_size,
188 size_t* service_cert_len)
189 {
190 // node_cert_size and service_cert_size are ignored here, but we pass them
191 // in because it allows us to set EDL an annotation so that node_cert_len
192 // <= node_cert_size is checked by the EDL-generated wrapper
193
194 start_type = start_type_;
195
196 rpcsessions->update_listening_interface_options(ccf_config_.network);
197
198 ccf::NodeCreateInfo r;
199 try
200 {
201 LOG_TRACE_FMT(
202 "Creating node with start_type {}", start_type_to_str(start_type));
203 r = node->create(start_type, std::move(ccf_config_));
204 }
205 catch (const std::exception& e)
206 {
207 LOG_FAIL_FMT("Error starting node: {}", e.what());
208 return CreateNodeStatus::InternalError;
209 }
210
211 // Copy node and service certs out
212 if (r.self_signed_node_cert.size() > node_cert_size)
213 {
214 LOG_FAIL_FMT(
215 "Insufficient space ({}) to copy node_cert out ({})",
216 node_cert_size,
217 r.self_signed_node_cert.size());
218 return CreateNodeStatus::InternalError;
219 }
220 ccf_memcpy(
221 node_cert,
222 r.self_signed_node_cert.data(),
223 r.self_signed_node_cert.size());
224 *node_cert_len = r.self_signed_node_cert.size();
225
226 if (start_type == StartType::Start || start_type == StartType::Recover)
227 {
228 // When starting a node in start or recover modes, fresh network secrets
229 // are created and the associated certificate can be passed to the host
230 if (r.service_cert.size() > service_cert_size)
231 {
232 LOG_FAIL_FMT(
233 "Insufficient space ({}) to copy service_cert out ({})",
234 service_cert_size,
235 r.service_cert.size());
236 return CreateNodeStatus::InternalError;
237 }
238 ccf_memcpy(service_cert, r.service_cert.data(), r.service_cert.size());
239 *service_cert_len = r.service_cert.size();
240 }
241
242 return CreateNodeStatus::OK;
243 }

245 bool run_main()
246 {
247 LOG_DEBUG_FMT("Running main thread");
248#ifndef VIRTUAL_ENCLAVE
249 try
250#endif
251 {
252 messaging::BufferProcessor bp("Enclave");
253
254 // reconstruct oversized messages sent to the enclave
255 oversized::FragmentReconstructor fr(bp.get_dispatcher());
256
257 lfs_access->register_message_handlers(bp.get_dispatcher());
258
259 DISPATCHER_SET_MESSAGE_HANDLER(
260 bp, AdminMessage::stop, [&bp](const uint8_t*, size_t) {
261 bp.set_finished();
262 threading::ThreadMessaging::thread_messaging.set_finished();
263 });
264
265 last_tick_time = ccf::get_enclave_time();
266
267 DISPATCHER_SET_MESSAGE_HANDLER(
268 bp,
269 AdminMessage::tick,
270 [this, &disp = bp.get_dispatcher()](const uint8_t*, size_t) {
271 const auto message_counts = disp.retrieve_message_counts();
272 const auto j = disp.convert_message_counts(message_counts);
273 RINGBUFFER_WRITE_MESSAGE(
274 AdminMessage::work_stats, to_host, j.dump());
275
276 const auto time_now = ccf::get_enclave_time();
277 ringbuffer_logger->set_time(time_now);
278
279 const auto elapsed_ms =
280 std::chrono::duration_cast<std::chrono::milliseconds>(
281 time_now - last_tick_time);
282 if (elapsed_ms.count() > 0)
283 {
284 last_tick_time = time_now;
285
286 node->tick(elapsed_ms);
287 historical_state_cache->tick(elapsed_ms);
288 threading::ThreadMessaging::thread_messaging.tick(elapsed_ms);
289 // When recovering, no signature should be emitted while the
290 // public ledger is being read
291 if (!node->is_reading_public_ledger())
292 {
293 for (auto& [actor, frontend] : rpc_map->frontends())
294 {
295 frontend->tick(elapsed_ms);
296 }
297 }
298 node->tick_end();
299 }
300 });
301
302 DISPATCHER_SET_MESSAGE_HANDLER(
303 bp, ccf::node_inbound, [this](const uint8_t* data, size_t size) {
304 node->recv_node_inbound(data, size);
305 });
306
307 DISPATCHER_SET_MESSAGE_HANDLER(
308 bp,
309 consensus::ledger_entry_range,
310 [this](const uint8_t* data, size_t size) {
311 const auto [from_seqno, to_seqno, purpose, body] =
312 ringbuffer::read_message<consensus::ledger_entry_range>(
313 data, size);
314 switch (purpose)
315 {
316 case consensus::LedgerRequestPurpose::Recovery:
317 {
318 if (
319 node->is_reading_public_ledger() ||
320 node->is_verifying_snapshot())
321 {
322 node->recover_public_ledger_entries(body);
323 }
324 else if (node->is_reading_private_ledger())
325 {
326 node->recover_private_ledger_entries(body);
327 }
328 else
329 {
330 auto [s, _, __] = node->state();
331 LOG_FAIL_FMT(
332 "Cannot recover ledger entry: Unexpected node state {}", s);
333 }
334 break;
335 }
336 case consensus::LedgerRequestPurpose::HistoricalQuery:
337 {
338 historical_state_cache->handle_ledger_entries(
339 from_seqno, to_seqno, body);
340 break;
341 }
342 default:
343 {
344 LOG_FAIL_FMT("Unhandled purpose: {}", purpose);
345 }
346 }
347 });
348
349 DISPATCHER_SET_MESSAGE_HANDLER(
350 bp,
351 consensus::ledger_no_entry_range,
352 [this](const uint8_t* data, size_t size) {
353 const auto [from_seqno, to_seqno, purpose] =
354 ringbuffer::read_message<consensus::ledger_no_entry_range>(
355 data, size);
356 switch (purpose)
357 {
358 case consensus::LedgerRequestPurpose::Recovery:
359 {
360 if (from_seqno != to_seqno)
361 {
362 LOG_FAIL_FMT(
363 "Unexpected range for Recovery response "
364 "ledger_no_entry_range: {}->{} "
365 "(expected single ledger entry)",
366 from_seqno,
367 to_seqno);
368 }
369 if (node->is_verifying_snapshot())
370 {
371 node->verify_snapshot_end();
372 }
373 else
374 {
375 node->recover_ledger_end();
376 }
377 break;
378 }
379 case consensus::LedgerRequestPurpose::HistoricalQuery:
380 {
381 historical_state_cache->handle_no_entry_range(
382 from_seqno, to_seqno);
383 break;
384 }
385 default:
386 {
387 LOG_FAIL_FMT("Unhandled purpose: {}", purpose);
388 }
389 }
390 });
391
392 rpcsessions->register_message_handlers(bp.get_dispatcher());
393
394 if (start_type == StartType::Join)
395 {
396 // When joining from a snapshot, deserialise ledger suffix to verify
397 // snapshot evidence. Otherwise, attempt to join straight away
398 if (node->is_verifying_snapshot())
399 {
400 node->start_ledger_recovery();
401 }
402 else
403 {
404 node->join();
405 }
406 }
407 else if (start_type == StartType::Recover)
408 {
409 node->start_ledger_recovery();
410 }
411
412 // Maximum number of inbound ringbuffer messages which will be
413 // processed in a single iteration
414 static constexpr size_t max_messages = 256;
415
416 size_t consecutive_idles = 0u;
417 while (!bp.get_finished())
418 {
419 // First, read some messages from the ringbuffer
420 auto read = bp.read_n(max_messages, circuit->read_from_outside());
421
422 // Then, execute some thread messages
423 size_t thread_msg = 0;
424 while (thread_msg < max_messages &&
425 threading::ThreadMessaging::thread_messaging.run_one())
426 {
427 thread_msg++;
428 }
429
430 // If no messages were read from the ringbuffer and no thread
431 // messages were executed, idle
432 if (read == 0 && thread_msg == 0)
433 {
434 const auto time_now = ccf::get_enclave_time();
435 static std::chrono::microseconds idling_start_time;
436
437 if (consecutive_idles == 0)
438 {
439 idling_start_time = time_now;
440 }
441
442 // Handle initial idles by pausing, eventually sleep (in host)
443 constexpr std::chrono::milliseconds timeout(5);
444 if ((time_now - idling_start_time) > timeout)
445 {
446 std::this_thread::sleep_for(timeout * 10);
447 }
448 else
449 {
450 CCF_PAUSE();
451 }
452
453 consecutive_idles++;
454 }
455 else
456 {
457 // If some messages were read, reset consecutive idles count
458 consecutive_idles = 0;
459 }
460 }
461
462 LOG_INFO_FMT("Enclave stopped successfully. Stopping host...");
463 RINGBUFFER_WRITE_MESSAGE(AdminMessage::stopped, to_host);
464
465 return true;
466 }
467#ifndef VIRTUAL_ENCLAVE
468 catch (const std::exception& e)
469 {
470 // It is expected that all enclave modules consuming ring buffer
471 // messages catch any thrown exception they can recover from. Uncaught
472 // exceptions bubble up to here and cause the node to shutdown.
473 RINGBUFFER_WRITE_MESSAGE(
474 AdminMessage::fatal_error_msg, to_host, std::string(e.what()));
475 return false;
476 }
477#endif
478 }

31 inline std::string error_string(int ec)
32 {
33 // ERR_error_string doesn't really expect the code could actually be zero
34 // and uses the `static char buf[256]` which is NOT cleaned nor checked
35 // if it has changed. So we use ERR_error_string_n directly.
36 if (ec)
37 {
38 std::string err(256, '\0');
39 ERR_error_string_n((unsigned long)ec, err.data(), err.size());
40 // Remove any trailing NULs before returning
41 err.resize(std::strlen(err.c_str()));
42 return err;
43 }
44 else
45 {
46 return "unknown error";
47 }
48 }

51 inline void CHECK1(int rc)
52 {
53 unsigned long ec = ERR_get_error();
54 if (rc != 1 && ec != 0)
55 {
56 throw std::runtime_error(
57 fmt::format("OpenSSL error: {}", error_string(ec)));
58 }
59 }
60
62 inline void CHECK0(int rc)
63 {
64 unsigned long ec = ERR_get_error();
65 if (rc == 0 && ec != 0)
66 {
67 throw std::runtime_error(
68 fmt::format("OpenSSL error: {}", error_string(ec)));
69 }
70 }

81 /*
82 * Unique pointer wrappers for SSL objects, with SSL' specific constructors
83 * and destructors. Some objects need special functionality, others are just
84 * wrappers around the same template interface Unique_SSL_OBJECT.
85 */
86
91 template <class T, T* (*CTOR)(), void (*DTOR)(T*)>
92 class Unique_SSL_OBJECT
93 {
94 protected:
96 std::unique_ptr<T, void (*)(T*)> p;
97
98 public:
100 Unique_SSL_OBJECT() : p(CTOR(), DTOR)
101 {
102 CHECKNULL(p.get());
103 }
105 Unique_SSL_OBJECT(T* ptr, void (*dtor)(T*), bool check_null = true) :
106 p(ptr, dtor)
107 {
108 if (check_null)
109 CHECKNULL(p.get());
110 }
112 operator T*()
113 {
114 return p.get();
115 }
117 operator T*() const
118 {
119 return p.get();
120 }
122 void reset(T* other)
123 {
124 p.reset(other);
125 }
127 T* release()
128 {
129 return p.release();
130 }
131 };

133 struct Unique_BIO : public Unique_SSL_OBJECT<BIO, nullptr, nullptr>
134 {
135 Unique_BIO() :
136 Unique_SSL_OBJECT(BIO_new(BIO_s_mem()), [](auto x) { BIO_free(x); })
137 {}
138 Unique_BIO(const void* buf, int len) :
139 Unique_SSL_OBJECT(
140 BIO_new_mem_buf(buf, len), [](auto x) { BIO_free(x); })
141 {}
142 Unique_BIO(const std::vector<uint8_t>& d) :
143 Unique_SSL_OBJECT(
144 BIO_new_mem_buf(d.data(), d.size()), [](auto x) { BIO_free(x); })
145 {}
146 Unique_BIO(const Pem& pem) :
147 Unique_SSL_OBJECT(
148 BIO_new_mem_buf(pem.data(), -1), [](auto x) { BIO_free(x); })
149 {}
150 Unique_BIO(SSL_CTX* ctx) :
151 Unique_SSL_OBJECT(
152 BIO_new_ssl_connect(ctx), [](auto x) { BIO_free_all(x); })
153 {}
154 };

28 static inline bool verify_and_decrypt(
29 crypto::KeyAesGcm& encryption_key,
30 const LFSKey& key,
31 LFSEncryptedContents&& encrypted,
32 std::vector<uint8_t>& plaintext)
33 {
34 crypto::GcmCipher gcm;
35 gcm.deserialise(encrypted);
36
37#ifdef PLAINTEXT_CACHE
38 plaintext = gcm.cipher;
39 auto success = true;
40#else
41 auto success = encryption_key.decrypt(
42 gcm.hdr.get_iv(), gcm.hdr.tag, gcm.cipher, {}, plaintext);
43#endif
44
45 // Check key prefix in plaintext
46 {
47 const auto encoded_prefix_size = sizeof(key.size()) + key.size();
48 if (plaintext.size() < encoded_prefix_size)
49 {
50 return false;
51 }
52
53 auto data = (const uint8_t*)plaintext.data();
54 auto size = plaintext.size();
55 const auto prefix_size = serialized::read<size_t>(data, size);
56 if (prefix_size != key.size())
57 {
58 success = false;
59 }
60 else
61 {
62 if (memcmp(data, key.data(), key.size()) != 0)
63 {
64 success = false;
65 }
66
67 plaintext.erase(
68 plaintext.begin(), plaintext.begin() + encoded_prefix_size);
69 }
70 }
71
72 if (success)
73 {
74 return true;
75 }
76 else
77 {
78 LOG_TRACE_FMT("Decryption failed for {}", key);
79 return false;
80 }
81 }

95 LFSEncryptedContents encrypt(const LFSKey& key, LFSContents&& contents)
96 {
97 // Prefix the contents with the key, to be checked during decryption
98 {
99 std::vector<uint8_t> key_prefix(sizeof(key.size()) + key.size());
100 auto data = key_prefix.data();
101 auto size = key_prefix.size();
102 serialized::write(data, size, key);
103 contents.insert(contents.begin(), key_prefix.begin(), key_prefix.end());
104 }
105
106 crypto::GcmCipher gcm(contents.size());
107
108 // Use a random IV for each call
109 gcm.hdr.set_random_iv();
110
111 encryption_key->encrypt(
112 gcm.hdr.get_iv(), contents, {}, gcm.cipher, gcm.hdr.tag);
113
114#ifdef PLAINTEXT_CACHE
115 gcm.cipher = contents;
116#endif
117
118 return gcm.serialise();
119 }

132 void register_message_handlers(
133 messaging::Dispatcher<ringbuffer::Message>& dispatcher)
134 {
135 DISPATCHER_SET_MESSAGE_HANDLER(
136 dispatcher, LFSMsg::response, [this](const uint8_t* data, size_t size) {
137 auto [obfuscated, encrypted] =
138 ringbuffer::read_message<LFSMsg::response>(data, size);
139 auto it = pending.find(obfuscated);
140 if (it != pending.end())
141 {
142 auto result = it->second.lock();
143 if (result != nullptr)
144 {
145 if (result->fetch_result == FetchResult::Fetching)
146 {
147 const auto success = verify_and_decrypt(
148 *encryption_key,
149 obfuscated,
150 std::move(encrypted),
151 result->contents);
152 if (success)
153 {
154 result->fetch_result = FetchResult::Loaded;
155 }
156 else
157 {
158 result->fetch_result = FetchResult::Corrupt;
159 LOG_TRACE_FMT(
160 "Cache was given invalid contents for {} (aka {})",
161 obfuscated,
162 result->key);
163 }
164 }
165 else
166 {
167 LOG_FAIL_FMT(
168 "Retained result for {} (aka {}) in state {}",
169 obfuscated,
170 result->key,
171 result->fetch_result);
172 }
173 }
174 else
175 {
176 LOG_TRACE_FMT(
177 "Received response for {}, but caller has already dropped "
178 "result",
179 obfuscated);
180 }
181 pending.erase(it);
182 }
183 else
184 {
185 LOG_TRACE_FMT(
186 "Ignoring response message for unrequested key {}", obfuscated);
187 }
188 });

236 static inline LFSKey obfuscate_key(const LFSKey& key)
237 {
238#ifdef PLAINTEXT_CACHE
239 return key;
240#else
241 const auto h = crypto::sha256((const uint8_t*)key.data(), key.size());
242 return ds::to_hex(h);
243#endif
244 }

246 void store(const LFSKey& key, LFSContents&& contents) override
247 {
248 const auto obfuscated = obfuscate_key(key);
249 RINGBUFFER_WRITE_MESSAGE(
250 LFSMsg::store,
251 to_host,
252 // To avoid leaking potentially confidential information to the host,
253 // all cached data is encrypted and stored at an obfuscated key
254 obfuscated,
255 encrypt(obfuscated, std::move(contents)));
256 }
257
258 FetchResultPtr fetch(const LFSKey& key) override
259 {
260 const auto obfuscated = obfuscate_key(key);
261 auto it = pending.find(obfuscated);
262
263 FetchResultPtr result;
264
265 if (it != pending.end())
266 {
267 result = it->second.lock();
268 if (result != nullptr)
269 {
270 if (key != result->key)
271 {
272 throw std::runtime_error(fmt::format(
273 "Obfuscation collision for unique keys '{}' and '{}', both "
274 "obfuscated to '{}'",
275 key,
276 result->key,
277 obfuscated));
278 }
279
280 return result;
281 }
282 else
283 {
284 result = std::make_shared<FetchResult>();
285 it->second = result;
286 }
287 }
288 else
289 {
290 result = std::make_shared<FetchResult>();
291 pending.emplace(obfuscated, result);
292 }
293
294 result->fetch_result = FetchResult::Fetching;
295 result->key = key;
296 RINGBUFFER_WRITE_MESSAGE(LFSMsg::get, to_host, obfuscated);
297 return result;
298 }
299 };

109 // Commit-related methods
110 bool has_writes() override
111 {
112 return committed_writes || change_set.has_writes();
113 }
114
115 bool prepare(bool track_read_versions) override
116 {
117 auto& roll = map.get_roll();
118 if (change_set.writes.empty())
119 {
120 return true;
121 }
122
123 // If the parent map has rolled back since this transaction began, this
124 // transaction must fail.
125 if (change_set.rollback_counter != roll.rollback_counter)
126 return false;
127
128 // If we have iterated over the map, check for a global version match.
129 auto current = roll.commits->get_tail();
130 if (
131 (change_set.read_version != NoVersion) &&
132 (change_set.read_version != current->version))
133 {
134 LOG_DEBUG_FMT("Read version {} is invalid", change_set.read_version);
135 return false;
136 }
137
138 // Check each key in our read set.
139 for (auto it = change_set.reads.begin(); it != change_set.reads.end();
140 ++it)
141 {
142 // Get the value from the current state.
143 auto search = current->state.get(it->first);
144
145 if (std::get<0>(it->second) == NoVersion)
146 {
147 // If we depend on the key not existing, it must be absent.
148 if (search.has_value())
149 {
150 LOG_DEBUG_FMT("Read depends on non-existing entry");
151 return false;
152 }
153 }
154 else
155 {
156 // If the transaction depends on the key existing, it must be
157 // present and have the the expected version. If also tracking
158 // conflicts then ensure that the read versions also match.
159 if (
160 !search.has_value() ||
161 std::get<0>(it->second) != search.value().version ||
162 (track_read_versions &&
163 std::get<1>(it->second) != search.value().read_version))
164 {
165 LOG_DEBUG_FMT("Read depends on invalid version of entry");
166 return false;
167 }
168 }
169 }
170
171 return true;
172 }
173
174 void commit(Version v_, bool track_read_versions) override
175 {
176 if (change_set.writes.empty() && !track_read_versions)
177 {
178 commit_version = change_set.start_version;
179 return;
180 }
181
182 auto& roll = map.get_roll();
183 auto state = roll.commits->get_tail()->state;
184
185 DeletableVersion v = static_cast<DeletableVersion>(v_);
186
187 // To track conflicts the read version of all keys that are read or
188 // written within a transaction must be updated.
189 if (track_read_versions)
190 {
191 for (auto it = change_set.reads.begin(); it != change_set.reads.end();
192 ++it)
193 {
194 auto search = state.get(it->first);
195 if (!search.has_value())
196 {
197 continue;
198 }
199 state = state.put(
200 it->first, VersionV{search->version, v_, search->value});
201 }
202 if (change_set.writes.empty())
203 {
204 commit_version = change_set.start_version;
205 map.roll.commits->insert_back(map.roll.create_new_local_commit(
206 commit_version, std::move(state), change_set.writes));
207 return;
208 }
209 }
210
211 // Record our commit time.
212 commit_version = v;
213 committed_writes = true;
214
215 for (auto it = change_set.writes.begin(); it != change_set.writes.end();
216 ++it)
217 {
218 if (it->second.has_value())
219 {
220 // Write the new value with the global version.
221 changes = true;
222 state = state.put(it->first, VersionV{v, v_, it->second.value()});
223 }
224 else
225 {
226 // Write an empty value with the deleted global version only if
227 // the key exists.
228 auto search = state.get(it->first);
229 if (search.has_value())
230 {
231 changes = true;
232 state = state.put(it->first, VersionV{-v, v_, {}});
233 }
234 }
235 }
236
237 if (changes)
238 {
239 map.roll.commits->insert_back(map.roll.create_new_local_commit(
240 v, std::move(state), change_set.writes));
241 }
242 }
243
244 ConsensusHookPtr post_commit() override
245 {
246 // This is run separately from commit so that all commits in the Tx
247 // have been applied before map hooks are run. The maps in the Tx
248 // are still locked when post_commit is run.
249 return map.trigger_map_hook(commit_version, change_set.writes);
250 }

328 void serialise_changes(
329 const AbstractChangeSet* changes,
330 KvStoreSerialiser& s,
331 bool include_reads) override
332 {
333 const auto non_abstract =
334 dynamic_cast<const kv::untyped::ChangeSet*>(changes);
335 if (non_abstract == nullptr)
336 {
337 LOG_FAIL_FMT("Unable to serialise map due to type mismatch");
338 return;
339 }
340
341 const auto& change_set = *non_abstract;
342
343 s.start_map(name, security_domain);
344
345 if (include_reads)
346 {
347 s.serialise_entry_version(change_set.read_version);
348
349 s.serialise_count_header(change_set.reads.size());
350 for (auto it = change_set.reads.begin(); it != change_set.reads.end();
351 ++it)
352 {
353 s.serialise_read(it->first, std::get<0>(it->second));
354 }
355 }
356 else
357 {
358 s.serialise_entry_version(NoVersion);
359 s.serialise_count_header(0);
360 }
361
362 uint64_t write_ctr = 0;
363 uint64_t remove_ctr = 0;
364 for (auto it = change_set.writes.begin(); it != change_set.writes.end();
365 ++it)
366 {
367 if (it->second.has_value())
368 {
369 ++write_ctr;
370 }
371 else
372 {
373 ++remove_ctr;
374 }
375 }
376
377 s.serialise_count_header(write_ctr);
378 for (auto it = change_set.writes.begin(); it != change_set.writes.end();
379 ++it)
380 {
381 if (it->second.has_value())
382 {
383 s.serialise_write(it->first, it->second.value());
384 }
385 }
386
387 s.serialise_count_header(remove_ctr);
388 for (auto it = change_set.writes.begin(); it != change_set.writes.end();
389 ++it)
390 {
391 if (!it->second.has_value())
392 {
393 s.serialise_remove(it->first);
394 }
395 }

475 ChangeSetPtr deserialise_internal(KvStoreDeserialiser& d, Version version)
476 {
477 // Create a new change set, and deserialise d's contents into it.
478 auto change_set_ptr = create_change_set(version);
479 if (change_set_ptr == nullptr)
480 {
481 LOG_FAIL_FMT(
482 "Failed to create change set over '{}' at {} - too early",
483 name,
484 version);
485 throw std::logic_error("Can't create change set");
486 }
487
488 auto& change_set = *change_set_ptr;
489
490 uint64_t ctr;
491
492 auto rv = d.deserialise_entry_version();
493 if (rv != NoVersion)
494 {
495 change_set.read_version = rv;
496 }
497
498 ctr = d.deserialise_read_header();
499 for (size_t i = 0; i < ctr; ++i)
500 {
501 auto r = d.deserialise_read();
502 change_set.reads[std::get<0>(r)] =
503 std::make_tuple(std::get<1>(r), NoVersion);
504 }
505
506 ctr = d.deserialise_write_header();
507 for (size_t i = 0; i < ctr; ++i)
508 {
509 auto w = d.deserialise_write();
510 change_set.writes[std::get<0>(w)] = std::get<1>(w);
511 }
512
513 ctr = d.deserialise_remove_header();
514 for (size_t i = 0; i < ctr; ++i)
515 {
516 auto r = d.deserialise_remove();
517 change_set.writes[r] = std::nullopt;
518 }
519
520 return change_set_ptr;
521 }

595 bool operator==(const Map& that) const
596 {
597 if (name != that.name)
598 return false;
599
600 auto state1 = roll.commits->get_tail();
601 auto state2 = that.roll.commits->get_tail();
602
603 if (state1->version != state2->version)
604 return false;
605
606 size_t count = 0;
607 state2->state.foreach([&count](const K&, const VersionV&) {
608 count++;
609 return true;
610 });
611
612 size_t i = 0;
613 bool ok =
614 state1->state.foreach([&state2, &i](const K& k, const VersionV& v) {
615 auto search = state2->state.get(k);
616
617 if (search.has_value())
618 {
619 auto& found = search.value();
620 if (found.version != v.version)
621 {
622 return false;
623 }
624 else if (found.value != v.value)
625 {
626 return false;
627 }
628 }
629 else
630 {
631 return false;
632 }
633
634 i++;
635 return true;
636 });
637
638 if (i != count)
639 ok = false;
640
641 return ok;
642 }

651 std::unique_ptr<AbstractMap::Snapshot> snapshot(Version v) override
652 {
653 // This takes a snapshot of the state of the map at the last entry
654 // committed at or before this version. The Map expects to be locked while
655 // taking the snapshot.
656 auto r = roll.commits->get_head();
657
658 for (auto current = roll.commits->get_tail(); current != nullptr;
659 current = current->prev)
660 {
661 if (current->version <= v)
662 {
663 r = current;
664 break;
665 }
666 }
667
668 return std::make_unique<Snapshot>(
669 name, security_domain, r->version, r->state.make_snapshot());
670 }

672 void compact(Version v) override
673 {
674 // This discards available rollback state before version v, and
675 // populates the commit_deltas to be passed to the global commit hook,
676 // if there is one, up to version v. The Map expects to be locked during
677 // compaction.
678 while (roll.commits->get_head() != roll.commits->get_tail())
679 {
680 auto r = roll.commits->get_head();
681
682 // Globally committed but not discardable.
683 if (r->version == v)
684 {
685 // We know that write set is not empty.
686 if (global_hook)
687 {
688 commit_deltas.emplace_back(r->version, std::move(r->writes));
689 }
690 return;
691 }
692
693 // Discardable, so move to commit_deltas.
694 if (global_hook && !r->writes.empty())
695 {
696 commit_deltas.emplace_back(r->version, std::move(r->writes));
697 }
698
699 // Stop if the next state may be rolled back or is the only state.
700 // This ensures there is always a state present.
701 if (r->next->version > v)
702 return;
703
704 auto c = roll.commits->pop();
705 roll.empty_commits.insert(c);
706 }
707
708 // There is only one roll. We may need to call the commit hook.
709 auto r = roll.commits->get_head();
710
711 if (global_hook && !r->writes.empty())
712 {
713 commit_deltas.emplace_back(r->version, std::move(r->writes));
714 }
715 }
716
717 void post_compact() override
718 {
719 if (global_hook)
720 {
721 for (auto& [version, writes] : commit_deltas)
722 {
723 global_hook(version, writes);
724 }
725 }
726
727 commit_deltas.clear();
728 }

730 void rollback(Version v) override
731 {
732 // This rolls the current state back to version v.
733 // The Map expects to be locked during rollback.
734 bool advance = false;
735
736 while (roll.commits->get_head() != roll.commits->get_tail())
737 {
738 auto r = roll.commits->get_tail();
739
740 // The initial empty state has v = 0, so will not be discarded if it
741 // is present.
742 if (r->version <= v)
743 break;
744
745 advance = true;
746 auto c = roll.commits->pop_tail();
747 roll.empty_commits.insert(c);
748 }
749
750 if (advance)
751 roll.rollback_counter++;
752 }

782 ChangeSetPtr create_change_set(Version version)
783 {
784 lock();
785
786 ChangeSetPtr changes = nullptr;
787
788 // Find the last entry committed at or before this version.
789 for (auto current = roll.commits->get_tail(); current != nullptr;
790 current = current->prev)
791 {
792 if (current->version <= version)
793 {
794 changes = std::make_unique<untyped::ChangeSet>(
795 roll.rollback_counter,
796 current->state,
797 roll.commits->get_head()->state,
798 current->version);
799 break;
800 }
801 }
802
803 // Returning nullptr is allowed, and indicates that we have no suitable
804 // version - the version requested is _earlier_ than anything in the
805 // roll
806
807 unlock();
808 return changes;
809 }

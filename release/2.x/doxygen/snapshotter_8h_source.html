86 void record_snapshot(
87 consensus::Index idx,
88 consensus::Index evidence_idx,

91 size_t max_message_size = to_host->get_max_message_size();

96 serialised_snapshot.size(),
97 max_message_size);

101 consensus::snapshot, to_host, idx, evidence_idx, serialised_snapshot);

125 void snapshot_(
126 std::unique_ptr<kv::AbstractStore::AbstractSnapshot> snapshot)
127 {
128 auto snapshot_version = snapshot->get_version();
129
130 auto serialised_snapshot = store->serialise_snapshot(std::move(snapshot));
131
132 auto tx = store->create_tx();
133 auto evidence = tx.rw<SnapshotEvidence>(Tables::SNAPSHOT_EVIDENCE);
134 auto snapshot_hash = crypto::Sha256Hash(serialised_snapshot);
135 evidence->put({snapshot_hash, snapshot_version});
136
137 ccf::ClaimsDigest cd;
138 cd.set(std::move(snapshot_hash));
139
140 crypto::Sha256Hash ws_digest;
141 std::string commit_evidence;
142 auto capture_ws_digest_and_commit_evidence =
143 [&ws_digest, &commit_evidence](
144 const std::vector<uint8_t>& write_set,
145 const std::string& commit_evidence_) {
146 new (&ws_digest)
147 crypto::Sha256Hash({write_set.data(), write_set.size()});
148 commit_evidence = commit_evidence_;
149 };
150
151 auto rc =
152 tx.commit(cd, false, nullptr, capture_ws_digest_and_commit_evidence);
153 if (rc != kv::CommitResult::SUCCESS)
154 {
155 LOG_FAIL_FMT(
156 "Could not commit snapshot evidence for seqno {}: {}",
157 snapshot_version,
158 rc);
159 return;
160 }
161
162 auto evidence_version = tx.commit_version();
163
164 record_snapshot(snapshot_version, evidence_version, serialised_snapshot);
165 consensus::Index snapshot_idx =
166 static_cast<consensus::Index>(snapshot_version);
167 consensus::Index snapshot_evidence_idx =
168 static_cast<consensus::Index>(evidence_version);
169 pending_snapshots.emplace_back(
170 snapshot_idx,
171 snapshot_evidence_idx,
172 ws_digest,
173 commit_evidence,
174 cd.value());
175
176 LOG_DEBUG_FMT(
177 "Snapshot successfully generated for seqno {}, with evidence seqno "
178 "{}: "
179 "{}, ws digest: {}",
180 snapshot_idx,
181 snapshot_evidence_idx,
182 cd.value(),
183 ws_digest);
184 }

186 void update_indices(consensus::Index idx)
187 {
188 while ((next_snapshot_indices.size() > 1) &&
189 (std::next(next_snapshot_indices.begin())->idx <= idx))
190 {
191 next_snapshot_indices.pop_front();
192 }
193
194 for (auto it = pending_snapshots.begin(); it != pending_snapshots.end();)
195 {
196 if (idx > it->evidence_idx)
197 {
198 auto serialised_receipt = build_and_serialise_receipt(
199 it->sig.value(),
200 it->tree.value(),
201 it->node_id.value(),
202 it->node_cert.value(),
203 it->evidence_idx,
204 it->write_set_digest,
205 it->commit_evidence,
206 std::move(it->snapshot_digest));
207 commit_snapshot(it->idx, serialised_receipt);
208 auto it_ = it;
209 ++it;
210 pending_snapshots.erase(it_);
211 }
212 else
213 {
214 ++it;
215 }
216 }
217 }

265 bool record_committable(consensus::Index idx) override
266 {
267 // Returns true if the committable idx will require the generation of a
268 // snapshot, and thus a new ledger chunk
269 std::lock_guard<ccf::Mutex> guard(lock);
270
271 CCF_ASSERT_FMT(
272 idx >= next_snapshot_indices.back().idx,
273 "Committable seqno {} < next snapshot seqno {}",
274 idx,
275 next_snapshot_indices.back().idx);
276
277 bool forced = store->flag_enabled_unsafe(
278 kv::AbstractStore::Flag::SNAPSHOT_AT_NEXT_SIGNATURE);
279
280 consensus::Index last_unforced_idx = last_snapshot_idx;
281 for (auto it = next_snapshot_indices.rbegin();
282 it != next_snapshot_indices.rend();
283 it++)
284 {
285 if (!it->forced)
286 {
287 last_unforced_idx = it->idx;
288 break;
289 }
290 }
291
292 auto due = (idx - last_unforced_idx) >= snapshot_tx_interval;
293 if (due || forced)
294 {
295 next_snapshot_indices.push_back({idx, !due, false});
296 LOG_TRACE_FMT(
297 "{} {} as snapshot index", !due ? "Forced" : "Recorded", idx);
298 store->unset_flag_unsafe(
299 kv::AbstractStore::Flag::SNAPSHOT_AT_NEXT_SIGNATURE);
300 return due;
301 }
302
303 return false;
304 }

344 {
345 auto msg = std::make_unique<threading::Tmsg<SnapshotMsg>>(&snapshot_cb);
346 msg->data.self = shared_from_this();
347 msg->data.snapshot = store->snapshot(idx);
348 static uint32_t generation_count = 0;

350 threading::ThreadMessaging::get_execution_thread(generation_count++),

354 void commit(consensus::Index idx, bool generate_snapshot) override
355 {
356 // If generate_snapshot is true, takes a snapshot of the key value store
357 // at the last snapshottable index before idx, and schedule snapshot
358 // serialisation on another thread (round-robin). Otherwise, only record
359 // that a snapshot was generated.
360 std::lock_guard<ccf::Mutex> guard(lock);
361
362 update_indices(idx);
363
364 if (idx < last_snapshot_idx)
365 {
366 throw std::logic_error(fmt::format(
367 "Cannot snapshot at seqno {} which is earlier than last snapshot "
368 "seqno {}",
369 idx,
370 last_snapshot_idx));
371 }
372
373 CCF_ASSERT_FMT(
374 idx >= next_snapshot_indices.front().idx,
375 "Cannot commit snapshotter at {}, which is before last snapshottable "
376 "idx {}",
377 idx,
378 next_snapshot_indices.front().idx);
379
380 auto& next = next_snapshot_indices.front();
381 auto due = next.idx - last_snapshot_idx >= snapshot_tx_interval;
382 if (due || (next.forced && !next.done))
383 {
384 if (snapshot_generation_enabled && generate_snapshot && next.idx)
385 {
386 schedule_snapshot(next.idx);
387 next.done = true;
388 }
389
390 if (due && !next.forced)
391 {
392 // last_snapshot_idx records the last normally scheduled, i.e.
393 // unforced, snapshot index, so that backups (which don't know forced
394 // indices) continue the snapshot interval normally.
395 last_snapshot_idx = next.idx;
396 LOG_TRACE_FMT(
397 "Recorded {} as last snapshot index", last_snapshot_idx);
398 }
399 }
400 }

402 void rollback(consensus::Index idx) override
403 {
404 std::lock_guard<ccf::Mutex> guard(lock);
405
406 while (!next_snapshot_indices.empty() &&
407 (next_snapshot_indices.back().idx > idx))
408 {
409 next_snapshot_indices.pop_back();
410 }
411
412 if (next_snapshot_indices.empty())
413 {
414 next_snapshot_indices.push_back({last_snapshot_idx, false, true});
415 }
416
417 LOG_TRACE_FMT(
418 "Rolled back snapshotter: last snapshottable idx is now {}",
419 next_snapshot_indices.front().idx);
420
421 while (!pending_snapshots.empty() &&
422 (pending_snapshots.back().evidence_idx > idx))
423 {
424 pending_snapshots.pop_back();
425 }
426 }
427 };

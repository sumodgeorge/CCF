51 static Write deserialise_write(const kv::untyped::Write& w)
52 {
53 Write typed_writes;
54 for (const auto& [uk, opt_uv] : w)
55 {
56 if (!opt_uv.has_value())
57 {
58 // Deletions are indicated by nullopt. We cannot deserialise them,
59 // they are deletions here as well
60 typed_writes[KSerialiser::from_serialised(uk)] = std::nullopt;
61 }

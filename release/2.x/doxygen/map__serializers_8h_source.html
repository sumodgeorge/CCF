73 {
74 using KeyType = typename M::KeyType;
75 using ValueType = typename M::ValueType;

77 M map;
78 const uint8_t* data = serialized_state.data();

81 while (size != 0)
82 {
83 // Deserialize the key
84 size_t key_size = size;

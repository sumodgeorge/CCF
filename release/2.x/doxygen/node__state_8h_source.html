66#ifdef GET_QUOTE
67 static QuoteInfo generate_quote(
68 const std::vector<uint8_t>& node_public_key_der)
69 {
70 QuoteInfo node_quote_info;
71 node_quote_info.format = QuoteFormat::oe_sgx_v1;
72
73 crypto::Sha256Hash h{node_public_key_der};
74
75 Evidence evidence;
76 Endorsements endorsements;
77 SerialisedClaims serialised_custom_claims;
78
79 // Serialise hash of node's public key as a custom claim
80 const size_t custom_claim_length = 1;
81 oe_claim_t custom_claim;
82 custom_claim.name = const_cast<char*>(sgx_report_data_claim_name);
83 custom_claim.value = h.h.data();
84 custom_claim.value_size = h.SIZE;
85
86 auto rc = oe_serialize_custom_claims(
87 &custom_claim,
88 custom_claim_length,
89 &serialised_custom_claims.buffer,
90 &serialised_custom_claims.size);
91 if (rc != OE_OK)
92 {
93 throw std::logic_error(fmt::format(
94 "Could not serialise node's public key as quote custom claim: {}",
95 oe_result_str(rc)));
96 }
97
98 rc = oe_get_evidence(
99 &oe_quote_format,
100 0,
101 serialised_custom_claims.buffer,
102 serialised_custom_claims.size,
103 nullptr,
104 0,
105 &evidence.buffer,
106 &evidence.size,
107 &endorsements.buffer,
108 &endorsements.size);
109 if (rc != OE_OK)
110 {
111 throw std::logic_error(
112 fmt::format("Failed to get evidence: {}", oe_result_str(rc)));
113 }
114
115 node_quote_info.quote.assign(
116 evidence.buffer, evidence.buffer + evidence.size);
117 node_quote_info.endorsements.assign(
118 endorsements.buffer, endorsements.buffer + endorsements.size);
119
120 return node_quote_info;
121 }
122#endif

329 NodeCreateInfo create(StartType start_type, StartupConfig&& config_)
330 {
331 std::lock_guard<ccf::Mutex> guard(lock);
332 sm.expect(NodeStartupState::initialized);
333
334 config = std::move(config_);
335 subject_alt_names = get_subject_alternative_names();
336
337 js::register_class_ids();
338 self_signed_node_cert = create_self_signed_cert(
339 node_sign_kp,
340 config.node_certificate.subject_name,
341 subject_alt_names,
342 config.startup_host_time,
343 config.node_certificate.initial_validity_days);
344
345 accept_node_tls_connections();
346 open_frontend(ActorsType::nodes);
347
348#ifdef GET_QUOTE
349 quote_info = generate_quote(node_sign_kp->public_key_der());
350 auto code_id = EnclaveAttestationProvider::get_code_id(quote_info);
351 if (code_id.has_value())
352 {
353 node_code_id = code_id.value();
354 }
355 else
356 {
357 throw std::logic_error("Failed to extract code id from quote");
358 }
359#endif
360
361 // Signatures are only emitted on a timer once the public ledger has been
362 // recovered
363 setup_history(start_type != StartType::Recover);
364 setup_snapshotter();
365 setup_encryptor();
366
367 switch (start_type)
368 {
369 case StartType::Start:
370 {
371 network.identity = std::make_unique<ReplicatedNetworkIdentity>(
372 curve_id,
373 config.startup_host_time,
374 config.initial_service_certificate_validity_days);
375
376 network.ledger_secrets->init();
377
378 if (network.consensus_type == ConsensusType::BFT)
379 {
380 endorsed_node_cert = create_endorsed_node_cert(
381 config.node_certificate.initial_validity_days);
382 history->set_endorsed_certificate(endorsed_node_cert.value());
383 accept_network_tls_connections();
384 open_frontend(ActorsType::members);
385 }
386
387 setup_consensus(
388 ServiceStatus::OPENING,
389 config.start.service_configuration.reconfiguration_type.value_or(
390 ReconfigurationType::ONE_TRANSACTION),
391 false,
392 endorsed_node_cert);
393
394 // Become the primary and force replication
395 consensus->force_become_primary();
396
397 if (!create_and_send_boot_request(
398 aft::starting_view_change, true /* Create new consortium */))
399 {
400 throw std::runtime_error(
401 "Genesis transaction could not be committed");
402 }
403
404 auto_refresh_jwt_keys();
405
406 reset_data(quote_info.quote);
407 reset_data(quote_info.endorsements);
408 sm.advance(NodeStartupState::partOfNetwork);
409
410 LOG_INFO_FMT("Created new node {}", self);
411
412 return {self_signed_node_cert, network.identity->cert};
413 }
414 case StartType::Join:
415 {
416 if (config.startup_snapshot.empty() || initialise_startup_snapshot())
417 {
418 // Note: 2.x snapshots are self-verified so the ledger verification
419 // of its evidence can be skipped entirely
420 sm.advance(NodeStartupState::pending);
421 }
422 else
423 {
424 // Node joins from a 1.x snapshot
425 sm.advance(NodeStartupState::verifyingSnapshot);
426 }
427
428 LOG_INFO_FMT("Created join node {}", self);
429 return {self_signed_node_cert, {}};
430 }
431 case StartType::Recover:
432 {
433 if (!config.recover.previous_service_identity)
434 {
435 throw std::logic_error(
436 "Recovery requires the certificate of the previous service "
437 "identity");
438 }
439
440 network.identity = std::make_unique<ReplicatedNetworkIdentity>(
441 curve_id,
442 config.startup_host_time,
443 config.initial_service_certificate_validity_days);
444
445 bool from_snapshot = !config.startup_snapshot.empty();
446 setup_recovery_hook();
447
448 if (from_snapshot)
449 {
450 initialise_startup_snapshot(true);
451 snapshotter->set_last_snapshot_idx(last_recovered_idx);
452 }
453
454 sm.advance(NodeStartupState::readingPublicLedger);
455
456 LOG_INFO_FMT("Created recovery node {}", self);
457 return {self_signed_node_cert, network.identity->cert};
458 }
459 default:
460 {
461 throw std::logic_error(
462 fmt::format("Node was started in unknown mode {}", start_type));
463 }
464 }
465 }

467 //
468 // funcs in state "pending"
469 //
470
471 void initiate_join_unsafe()
472 {
473 sm.expect(NodeStartupState::pending);
474
475 auto network_ca = std::make_shared<tls::CA>(std::string(
476 config.join.service_cert.begin(), config.join.service_cert.end()));
477 auto join_client_cert = std::make_unique<tls::Cert>(
478 network_ca,
479 self_signed_node_cert,
480 node_sign_kp->private_key_pem(),
481 config.join.target_rpc_address);
482
483 // Create RPC client and connect to remote node
484 auto join_client =
485 rpcsessions->create_client(std::move(join_client_cert));
486
487 auto [target_host, target_port] =
488 split_net_address(config.join.target_rpc_address);
489
490 join_client->connect(
491 target_host,
492 target_port,
493 [this](
494 http_status status,
495 http::HeaderMap&& headers,
496 std::vector<uint8_t>&& data) {
497 std::lock_guard<ccf::Mutex> guard(lock);
498 if (!sm.check(NodeStartupState::pending))
499 {
500 return;
501 }
502
503 if (status != HTTP_STATUS_OK)
504 {
505 const auto& location = headers.find(http::headers::LOCATION);
506 if (
507 status == HTTP_STATUS_PERMANENT_REDIRECT &&
508 location != headers.end())
509 {
510 const auto& url = http::parse_url_full(location->second);
511 config.join.target_rpc_address =
512 make_net_address(url.host, url.port);
513 LOG_INFO_FMT("Target node redirected to {}", location->second);
514 }
515 else
516 {
517 LOG_FAIL_FMT(
518 "An error occurred while joining the network: {} {}{}",
519 status,
520 http_status_str(status),
521 data.empty() ?
522 "" :
523 fmt::format(" '{}'", std::string(data.begin(), data.end())));
524 }
525 return;
526 }
527
528 auto j = serdes::unpack(data, serdes::Pack::Text);
529
530 JoinNetworkNodeToNode::Out resp;
531 try
532 {
533 resp = j.get<JoinNetworkNodeToNode::Out>();
534 }
535 catch (const std::exception& e)
536 {
537 LOG_FAIL_FMT(
538 "An error occurred while parsing the join network response");
539 LOG_DEBUG_FMT(
540 "An error occurred while parsing the join network response: {}",
541 j.dump());
542 return;
543 }
544
545 // Set network secrets, node id and become part of network.
546 if (
547 resp.node_status == NodeStatus::TRUSTED ||
548 resp.node_status == NodeStatus::LEARNER)
549 {
550 if (resp.network_info->consensus_type != network.consensus_type)
551 {
552 throw std::logic_error(fmt::format(
553 "Enclave initiated with consensus type {} but target node "
554 "responded with consensus {}",
555 network.consensus_type,
556 resp.network_info->consensus_type));
557 }
558
559 network.identity = std::make_unique<ReplicatedNetworkIdentity>(
560 resp.network_info->identity);
561 network.ledger_secrets->init_from_map(
562 std::move(resp.network_info->ledger_secrets));
563
564 crypto::Pem n2n_channels_cert;
565 if (!resp.network_info->endorsed_certificate.has_value())
566 {
567 // Endorsed node certificate is included in join response
568 // from 2.x (CFT only). When joining an existing 1.x service,
569 // self-sign own certificate and use it to endorse TLS
570 // connections.
571 endorsed_node_cert = create_endorsed_node_cert(
572 default_node_cert_validity_period_days);
573 history->set_endorsed_certificate(endorsed_node_cert.value());
574 n2n_channels_cert = endorsed_node_cert.value();
575 open_frontend(ActorsType::members);
576 open_user_frontend();
577 accept_network_tls_connections();
578 }
579 else
580 {
581 n2n_channels_cert =
582 resp.network_info->endorsed_certificate.value();
583 }
584
585 setup_consensus(
586 resp.network_info->service_status.value_or(
587 ServiceStatus::OPENING),
588 resp.network_info->reconfiguration_type.value_or(
589 ReconfigurationType::ONE_TRANSACTION),
590 resp.network_info->public_only,
591 n2n_channels_cert);
592 auto_refresh_jwt_keys();
593
594 if (resp.network_info->public_only)
595 {
596 last_recovered_signed_idx =
597 resp.network_info->last_recovered_signed_idx;
598 setup_recovery_hook();
599 snapshotter->set_snapshot_generation(false);
600 }
601
602 View view = VIEW_UNKNOWN;
603 std::vector<kv::Version> view_history = {};
604 if (startup_snapshot_info)
605 {
606 // It is only possible to deserialise the entire snapshot then,
607 // once the ledger secrets have been passed in by the network
608 kv::ConsensusHookPtrs hooks;
609 deserialise_snapshot(
610 network.tables,
611 startup_snapshot_info->raw,
612 hooks,
613 &view_history,
614 resp.network_info->public_only,
615 startup_snapshot_info->evidence_seqno,
616 config.recover.previous_service_identity);
617
618 for (auto& hook : hooks)
619 {
620 hook->call(consensus.get());
621 }
622
623 auto tx = network.tables->create_read_only_tx();
624 auto signatures = tx.ro(network.signatures);
625 auto sig = signatures->get();
626 if (!sig.has_value())
627 {
628 throw std::logic_error(
629 fmt::format("No signatures found after applying snapshot"));
630 }
631 view = sig->view;
632
633 if (!resp.network_info->public_only)
634 {
635 // Only clear snapshot if not recovering. When joining the
636 // public network the snapshot is used later to initialise the
637 // recovery store
638 startup_snapshot_info.reset();
639 }
640
641 LOG_INFO_FMT(
642 "Joiner successfully resumed from snapshot at seqno {} and "
643 "view {}",
644 network.tables->current_version(),
645 view);
646 }
647
648 consensus->init_as_backup(
649 network.tables->current_version(),
650 view,
651 view_history,
652 last_recovered_signed_idx);
653
654 snapshotter->set_last_snapshot_idx(
655 network.tables->current_version());
656
657 if (resp.network_info->public_only)
658 {
659 sm.advance(NodeStartupState::partOfPublicNetwork);
660 }
661 else
662 {
663 reset_data(quote_info.quote);
664 reset_data(quote_info.endorsements);
665 sm.advance(NodeStartupState::partOfNetwork);
666 }
667
668 LOG_INFO_FMT(
669 "Node has now joined the network as node {}: {}",
670 self,
671 (resp.network_info->public_only ? "public only" : "all domains"));
672 }
673 else if (resp.node_status == NodeStatus::PENDING)
674 {
675 LOG_INFO_FMT(
676 "Node {} is waiting for votes of members to be trusted", self);
677 }
678 },
679 [this](const std::string& error_msg) {
680 std::lock_guard<ccf::Mutex> guard(lock);
681 auto long_error_msg = fmt::format(
682 "Early error when joining existing network at {}: {}. Shutting "
683 "down node gracefully...",
684 config.join.target_rpc_address,
685 error_msg);
686 LOG_FAIL_FMT("{}", long_error_msg);
687 RINGBUFFER_WRITE_MESSAGE(
688 AdminMessage::fatal_error_msg, to_host, long_error_msg);
689 });
690
691 // Send RPC request to remote node to join the network.
692 JoinNetworkNodeToNode::In join_params;
693
694 join_params.node_info_network = config.network;
695 join_params.public_encryption_key = node_encrypt_kp->public_key_pem();
696 join_params.quote_info = quote_info;
697 join_params.consensus_type = network.consensus_type;
698 join_params.startup_seqno = startup_seqno;
699 join_params.certificate_signing_request = node_sign_kp->create_csr(
700 config.node_certificate.subject_name, subject_alt_names);
701 join_params.node_data = config.node_data;
702
703 LOG_DEBUG_FMT(
704 "Sending join request to {}", config.join.target_rpc_address);
705
706 const auto body = serdes::pack(join_params, serdes::Pack::Text);
707
708 http::Request r(
709 fmt::format("/{}/{}", get_actor_prefix(ActorsType::nodes), "join"));
710 r.set_header(
711 http::headers::CONTENT_TYPE, http::headervalues::contenttype::JSON);
712 r.set_body(&body);
713
714 join_client->send_request(r.build_request());
715 }

808 void recover_public_ledger_entries(const std::vector<uint8_t>& entries)
809 {
810 std::lock_guard<ccf::Mutex> guard(lock);
811
812 std::shared_ptr<kv::Store> store;
813 if (sm.check(NodeStartupState::readingPublicLedger))
814 {
815 // In recovery, use the main store to deserialise public entries
816 store = network.tables;
817 }
818 else if (sm.check(NodeStartupState::verifyingSnapshot))
819 {
820 store = startup_snapshot_info->store;
821 }
822 else
823 {
824 LOG_FAIL_FMT(
825 "Node should be in state {} or {} to recover public ledger entries",
826 NodeStartupState::readingPublicLedger,
827 NodeStartupState::verifyingSnapshot);
828 return;
829 }
830
831 auto data = entries.data();
832 auto size = entries.size();
833
834 if (size == 0)
835 {
836 if (is_verifying_snapshot())
837 {
838 verify_snapshot_end_unsafe();
839 }
840 else
841 {
842 recover_public_ledger_end_unsafe();
843 }
844 return;
845 }
846
847 while (size > 0)
848 {
849 auto entry = consensus::LedgerEnclave::get_entry(data, size);
850
851 LOG_INFO_FMT("Deserialising public ledger entry [{}]", entry.size());
852
853 // When reading the private ledger, deserialise in the recovery store
854
855 kv::ApplyResult result = kv::ApplyResult::FAIL;
856 try
857 {
858 auto r = store->deserialize(entry, ConsensusType::CFT, true);
859 result = r->apply();
860 if (result == kv::ApplyResult::FAIL)
861 {
862 LOG_FAIL_FMT(
863 "Failed to deserialise public ledger entry: {}", result);
864 recover_public_ledger_end_unsafe();
865 return;
866 }
867 ++last_recovered_idx;
868
869 // Not synchronised because consensus isn't effectively running then
870 for (auto& hook : r->get_hooks())
871 {
872 hook->call(consensus.get());
873 }
874 }
875 catch (const std::exception& e)
876 {
877 LOG_FAIL_FMT(
878 "Failed to deserialise public ledger entry: {}", e.what());
879 recover_public_ledger_end_unsafe();
880 return;
881 }
882
883 // If the ledger entry is a signature, it is safe to compact the store
884 if (result == kv::ApplyResult::PASS_SIGNATURE)
885 {
886 // If the ledger entry is a signature, it is safe to compact the store
887 store->compact(last_recovered_idx);
888 auto tx = store->create_read_only_tx();
889 auto last_sig = tx.ro(network.signatures)->get();
890
891 if (!last_sig.has_value())
892 {
893 throw std::logic_error("Signature missing");
894 }
895
896 LOG_DEBUG_FMT(
897 "Read signature at {} for view {}",
898 last_recovered_idx,
899 last_sig->view);
900 // Initial transactions, before the first signature, must have
901 // happened in the first signature's view (eg - if the first
902 // signature is at seqno 20 in view 4, then transactions 1->19 must
903 // also have been in view 4). The brief justification is that while
904 // the first node may start in an arbitrarily high view (it does not
905 // necessarily start in view 1), it cannot _change_ view before a
906 // valid signature.
907 const auto view_start_idx =
908 view_history.empty() ? 1 : last_recovered_signed_idx + 1;
909 CCF_ASSERT_FMT(
910 last_sig->view >= 0,
911 "last_sig->view is invalid, {}",
912 last_sig->view);
913 for (auto i = view_history.size();
914 i < static_cast<size_t>(last_sig->view);
915 ++i)
916 {
917 view_history.push_back(view_start_idx);
918 }
919 last_recovered_signed_idx = last_recovered_idx;
920
921 if (
922 startup_snapshot_info && startup_snapshot_info->has_evidence &&
923 startup_snapshot_info->evidence_seqno.has_value() &&
924 static_cast<consensus::Index>(last_sig->commit_seqno) >=
925 startup_snapshot_info->evidence_seqno.value())
926 {
927 startup_snapshot_info->is_evidence_committed = true;
928 }
929 }
930 else if (
931 result == kv::ApplyResult::PASS_SNAPSHOT_EVIDENCE &&
932 startup_snapshot_info)
933 {
934 auto tx = store->create_read_only_tx();
935 auto snapshot_evidence = tx.ro(network.snapshot_evidence);
936
937 if (
938 startup_snapshot_info->evidence_seqno.has_value() &&
939 last_recovered_idx == startup_snapshot_info->evidence_seqno.value())
940 {
941 auto evidence = snapshot_evidence->get();
942 if (!evidence.has_value())
943 {
944 throw std::logic_error("Invalid snapshot evidence");
945 }
946
947 if (
948 evidence->hash == crypto::Sha256Hash(startup_snapshot_info->raw))
949 {
950 LOG_DEBUG_FMT(
951 "Snapshot evidence for snapshot found at {}",
952 startup_snapshot_info->evidence_seqno.value());
953 startup_snapshot_info->has_evidence = true;
954 }
955 }
956 }
957 }
958
959 read_ledger_entries(
960 last_recovered_idx + 1, last_recovered_idx + recovery_batch_size);
961 }

1010 void recover_public_ledger_end_unsafe()
1011 {
1012 sm.expect(NodeStartupState::readingPublicLedger);
1013
1014 if (
1015 startup_snapshot_info &&
1016 startup_snapshot_info->requires_ledger_verification())
1017 {
1018 if (!startup_snapshot_info->is_snapshot_verified())
1019 {
1020 throw std::logic_error(
1021 "Snapshot evidence was not committed in ledger");
1022 }
1023
1024 if (
1025 last_recovered_signed_idx <
1026 startup_snapshot_info->evidence_seqno.value())
1027 {
1028 throw std::logic_error("Snapshot evidence would be rolled back");
1029 }
1030 }
1031
1032 // When reaching the end of the public ledger, truncate to last signed
1033 // index
1034 const auto last_recovered_term = view_history.size();
1035 auto new_term = last_recovered_term + aft::starting_view_change;
1036 LOG_INFO_FMT("Setting term on public recovery store to {}", new_term);
1037
1038 // Note: KV term must be set before the first Tx is committed
1039 network.tables->rollback(
1040 {last_recovered_term, last_recovered_signed_idx}, new_term);
1041 ledger_truncate(last_recovered_signed_idx, true);
1042 snapshotter->rollback(last_recovered_signed_idx);
1043
1044 LOG_INFO_FMT(
1045 "End of public ledger recovery - Truncating ledger to last signed "
1046 "TxID: {}.{}",
1047 last_recovered_term,
1048 last_recovered_signed_idx);
1049
1050 auto tx = network.tables->create_read_only_tx();
1051 if (network.consensus_type == ConsensusType::BFT)
1052 {
1053 endorsed_node_cert = create_endorsed_node_cert(
1054 config.node_certificate.initial_validity_days);
1055 history->set_endorsed_certificate(endorsed_node_cert.value());
1056 accept_network_tls_connections();
1057 open_frontend(ActorsType::members);
1058 }
1059
1060 network.ledger_secrets->init(last_recovered_signed_idx + 1);
1061
1062 // Initialise snapshotter after public recovery
1063 snapshotter->init_after_public_recovery();
1064 snapshotter->set_snapshot_generation(false);
1065
1066 kv::Version index = 0;
1067 kv::Term view = 0;
1068 kv::Version global_commit = 0;
1069
1070 auto ls = tx.ro(network.signatures)->get();
1071 if (ls.has_value())
1072 {
1073 auto s = ls.value();
1074 index = s.seqno;
1075 view = s.view;
1076 global_commit = s.commit_seqno;
1077 }
1078
1079 auto h = dynamic_cast<MerkleTxHistory*>(history.get());
1080 if (h)
1081 {
1082 h->set_node_id(self);
1083 }
1084
1085 auto service_config = tx.ro(network.config)->get();
1086 auto reconfiguration_type = service_config->reconfiguration_type.value_or(
1087 ReconfigurationType::ONE_TRANSACTION);
1088
1089 setup_consensus(ServiceStatus::OPENING, reconfiguration_type, true);
1090 auto_refresh_jwt_keys();
1091
1092 LOG_DEBUG_FMT(
1093 "Restarting consensus at view: {} seqno: {} commit_seqno {}",
1094 view,
1095 index,
1096 global_commit);
1097
1098 consensus->force_become_primary(index, view, view_history, index);
1099
1100 // First recovery transaction is asynchronous to avoid deadlocks
1101 // (https://github.com/microsoft/CCF/issues/3788)
1102 auto msg = std::make_unique<threading::Tmsg<NodeStateMsg>>(
1103 [](std::unique_ptr<threading::Tmsg<NodeStateMsg>> msg) {
1104 if (!msg->data.self.create_and_send_boot_request(
1105 msg->data.create_view,
1106 false /* Restore consortium from ledger */))
1107 {
1108 throw std::runtime_error(
1109 "End of public recovery transaction could not be committed");
1110 }
1111 msg->data.self.advance_part_of_public_network();
1112 },
1113 *this,
1114 new_term);
1115 threading::ThreadMessaging::thread_messaging.add_task(
1116 threading::get_current_thread_id(), std::move(msg));
1117 }

1188 std::min(last_recovered_idx + recovery_batch_size, recovery_v));

1192 void recover_private_ledger_end_unsafe()
1193 {
1194 // When reaching the end of the private ledger, make sure the same
1195 // ledger has been read and swap in private state
1196
1197 sm.expect(NodeStartupState::readingPrivateLedger);
1198
1199 if (recovery_v != recovery_store->current_version())
1200 {
1201 throw std::logic_error(fmt::format(
1202 "Private recovery did not reach public ledger seqno: {}/{}",
1203 recovery_store->current_version(),
1204 recovery_v));
1205 }
1206
1207 auto h =
1208 dynamic_cast<MerkleTxHistory*>(recovery_store->get_history().get());
1209 if (h->get_replicated_state_root() != recovery_root)
1210 {
1211 throw std::logic_error(fmt::format(
1212 "Root of public store does not match root of private store at {}",
1213 recovery_v));
1214 }
1215
1216 network.tables->swap_private_maps(*recovery_store.get());
1217 recovery_store.reset();
1218
1219 // Raft should deserialise all security domains when network is opened
1220 consensus->enable_all_domains();
1221
1222 // Snapshots are only generated after recovery is complete
1223 snapshotter->set_snapshot_generation(true);
1224
1225 // Open the service
1226 if (consensus->can_replicate())
1227 {
1228 setup_one_off_secret_hook();
1229 auto tx = network.tables->create_tx();
1230
1231 // Clear recovery shares that were submitted to initiate the recovery
1232 // procedure
1233 share_manager.clear_submitted_recovery_shares(tx);
1234
1235 // Shares for the new ledger secret can only be issued now, once the
1236 // previous ledger secrets have been recovered
1237 share_manager.issue_recovery_shares(tx);
1238
1239 GenesisGenerator g(network, tx);
1240 if (!g.open_service())
1241 {
1242 throw std::logic_error("Service could not be opened");
1243 }
1244
1245 // Trigger a snapshot (at next signature) to ensure we have a working
1246 // snapshot signed by the current (now new) service identity, in case
1247 // we need to recover soon again.
1248 trigger_snapshot(tx);
1249
1250 if (tx.commit() != kv::CommitResult::SUCCESS)
1251 {
1252 throw std::logic_error(
1253 "Could not commit transaction when finishing network recovery");
1254 }
1255 }
1256 recovered_encrypted_ledger_secrets.clear();
1257 reset_data(quote_info.quote);
1258 reset_data(quote_info.endorsements);
1259 sm.advance(NodeStartupState::partOfNetwork);
1260 }

1397 void transition_service_to_open(
1398 kv::Tx& tx,
1399 AbstractGovernanceEffects::ServiceIdentities identities) override
1400 {
1401 std::lock_guard<ccf::Mutex> guard(lock);
1402
1403 auto service = tx.rw<Service>(Tables::SERVICE);
1404 auto service_info = service->get();
1405 if (!service_info.has_value())
1406 {
1407 throw std::logic_error(
1408 "Service information cannot be found to transition service to "
1409 "open");
1410 }
1411
1412 // Idempotence: if the service is already open or waiting for recovery
1413 // shares, this function should succeed with no effect
1414 if (
1415 service_info->status == ServiceStatus::WAITING_FOR_RECOVERY_SHARES ||
1416 service_info->status == ServiceStatus::OPEN)
1417 {
1418 LOG_DEBUG_FMT(
1419 "Service in state {} is already open", service_info->status);
1420 return;
1421 }
1422
1423 if (service_info->status == ServiceStatus::RECOVERING)
1424 {
1425 const auto prev_ident = tx.ro<ccf::PreviousServiceIdentity>(
1426 ccf::Tables::PREVIOUS_SERVICE_IDENTITY)
1427 ->get();
1428 if (!prev_ident.has_value() || !identities.previous.has_value())
1429 {
1430 throw std::logic_error(
1431 "Recovery with service certificates requires both, a previous "
1432 "service identity written to the KV during recovery genesis and a "
1433 "transition_service_to_open proposal that contains previous and "
1434 "next service certificates");
1435 }
1436
1437 const crypto::Pem from_proposal(
1438 identities.previous->data(), identities.previous->size());
1439 if (prev_ident.value() != from_proposal)
1440 {
1441 throw std::logic_error(fmt::format(
1442 "Previous service identity does not match.\nActual:\n{}\nIn "
1443 "proposal:\n{}",
1444 prev_ident->str(),
1445 from_proposal.str()));
1446 }
1447 }
1448
1449 if (identities.next != service_info->cert)
1450 {
1451 throw std::logic_error(fmt::format(
1452 "Service identity mismatch: the next service identity in the "
1453 "transition_service_to_open proposal does not match the current "
1454 "service identity:\nNext:\n{}\nCurrent:\n{}",
1455 identities.next.str(),
1456 service_info->cert.str()));
1457 }
1458
1459 service_info->previous_service_identity_version =
1460 service->get_version_of_previous_write();
1461
1462 if (is_part_of_public_network())
1463 {
1464 // If the node is in public mode, start accepting member recovery
1465 // shares
1466 share_manager.clear_submitted_recovery_shares(tx);
1467 service_info->status = ServiceStatus::WAITING_FOR_RECOVERY_SHARES;
1468 service->put(service_info.value());
1469 return;
1470 }
1471 else if (is_part_of_network())
1472 {
1473 // Otherwise, if the node is part of the network. Open the network
1474 // straight away. Recovery shares are allocated to each recovery
1475 // member.
1476 try
1477 {
1478 share_manager.issue_recovery_shares(tx);
1479 }
1480 catch (const std::logic_error& e)
1481 {
1482 throw std::logic_error(
1483 fmt::format("Failed to issue recovery shares: {}", e.what()));
1484 }
1485

1715 bool is_ip(const std::string_view& hostname)
1716 {
1717 // IP address components are purely numeric. DNS names may be largely
1718 // numeric, but at least the final component (TLD) must not be
1719 // all-numeric. So this distinguishes "1.2.3.4" (and IP address) from
1720 // "1.2.3.c4m" (a DNS name). "1.2.3." is invalid for either, and will
1721 // throw. Attempts to handle IPv6 by also splitting on ':', but this is
1722 // untested.
1723 const auto final_component =
1724 nonstd::split(nonstd::split(hostname, ".").back(), ":").back();
1725 if (final_component.empty())
1726 {
1727 throw std::runtime_error(fmt::format(
1728 "{} has a trailing period, is not a valid hostname",
1729 final_component));
1730 }
1731 for (const auto c : final_component)
1732 {
1733 if (c < '0' || c > '9')
1734 {
1735 return false;
1736 }
1737 }
1738
1739 return true;
1740 }

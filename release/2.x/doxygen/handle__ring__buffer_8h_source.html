40 // Register message handler for log message from enclave
41 DISPATCHER_SET_MESSAGE_HANDLER(
42 bp, AdminMessage::log_msg, [](const uint8_t* data, size_t size) {
43 auto
44 [log_time_us_count,
45 file_name,
46 line_number,
47 log_level,
48 thread_id,
49 msg] = ringbuffer::read_message<AdminMessage::log_msg>(data, size);
50
51 logger::LogLine ll(
52 log_level, file_name.c_str(), line_number, thread_id);
53 ll.msg = msg;
54
55 // Represent offset as a real (counting seconds) to handle both small
56 // negative _and_ positive numbers. Since the system clock used is not
57 // monotonic, the offset we calculate could go in either direction,
58 // and tm can't represent small negative values.
59 std::optional<double> offset_time = std::nullopt;
60
61 // If enclave doesn't know the
62 // current time yet, don't try to produce an offset, just give them
63 // the host's time (producing offset of 0)
64 if (log_time_us_count != 0)
65 {
66 // Enclave time is recomputed every time. If multiple threads
67 // log inside the enclave, offsets may not always increase
68 const double enclave_time_s = log_time_us_count / 1'000'000.0;
69
70 ::timespec ts;
71 ::timespec_get(&ts, TIME_UTC);
72 const double host_time_s =
73 ts.tv_sec + (ts.tv_nsec / 1'000'000'000.0);
74
75 offset_time = enclave_time_s - host_time_s;
76 }
77
78 auto& loggers = logger::config::loggers();
79 for (auto const& logger : loggers)
80 {
81 logger->write(ll, offset_time);
82 }
83 });
84
85 DISPATCHER_SET_MESSAGE_HANDLER(
86 bp,
87 AdminMessage::fatal_error_msg,
88 [](const uint8_t* data, size_t size) {
89 auto [msg] =
90 ringbuffer::read_message<AdminMessage::fatal_error_msg>(data, size);
91
92 std::cerr << msg << std::endl << std::flush;
93 throw std::logic_error(msg);
94 });
95
96 DISPATCHER_SET_MESSAGE_HANDLER(
97 bp, AdminMessage::stopped, [](const uint8_t*, size_t) {
98 uv_stop(uv_default_loop());
99 LOG_INFO_FMT("Host stopped successfully");
100 });
101 }

107
108 // ...flush any pending inbound messages...

110 }
111 };

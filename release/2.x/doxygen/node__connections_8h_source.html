22 private:
23 public:
24 NodeConnections& parent;
25 std::optional<ccf::NodeId> node;
26 std::optional<size_t> msg_size = std::nullopt;
27 std::vector<uint8_t> pending;
28
29 NodeConnectionBehaviour(
30 NodeConnections& parent,
31 std::optional<ccf::NodeId> node = std::nullopt) :
32 parent(parent),
33 node(node)
34 {}
35
36 void on_read(size_t len, uint8_t*& incoming)
37 {
38 LOG_DEBUG_FMT(
39 "from node {} received {} bytes",
40 node.value_or(UnassociatedNode),
41 len);
42
43 pending.insert(pending.end(), incoming, incoming + len);
44
45 const uint8_t* data = pending.data();
46 size_t size = pending.size();
47 const auto size_before = size;
48
49 while (true)
50 {
51 if (!msg_size.has_value())
52 {
53 if (size < sizeof(uint32_t))
54 {
55 break;
56 }
57
58 msg_size = serialized::read<uint32_t>(data, size);
59 }
60
61 if (size < msg_size.value())
62 {
63 LOG_DEBUG_FMT(
64 "from node {} have {}/{} bytes",
65 node.value_or(UnassociatedNode),
66 size,
67 msg_size.value());
68 break;
69 }
70
71 const auto size_pre_headers = size;
72 auto msg_type = serialized::read<ccf::NodeMsgType>(data, size);
73 ccf::NodeId from = serialized::read<ccf::NodeId::Value>(data, size);
74 const auto size_post_headers = size;
75 const size_t payload_size =
76 msg_size.value() - (size_pre_headers - size_post_headers);
77
78 if (!node.has_value())
79 {
80 associate_incoming(from);
81 node = from;
82 }
83
84 LOG_DEBUG_FMT(
85 "node in: from node {}, size {}, type {}",
86 node.value(),
87 msg_size.value(),
88 msg_type);
89
90 RINGBUFFER_WRITE_MESSAGE(
91 ccf::node_inbound,
92 parent.to_enclave,
93 msg_type,
94 from.value(),
95 serializer::ByteRange{data, payload_size});
96
97 data += payload_size;
98 size -= payload_size;
99 msg_size.reset();
100 }
101
102 const auto size_after = size;
103 const auto used = size_before - size_after;
104 if (used > 0)
105 {
106 pending.erase(pending.begin(), pending.begin() + used);
107 }
108 }
109
110 virtual void associate_incoming(const ccf::NodeId&) {}
111 };

113 class NodeIncomingBehaviour : public NodeConnectionBehaviour
114 {
115 public:
116 size_t id;
117 std::optional<ccf::NodeId> node_id;
118
119 NodeIncomingBehaviour(NodeConnections& parent, size_t id_) :
120 NodeConnectionBehaviour(parent),
121 id(id_)
122 {}
123
124 void on_disconnect() override
125 {
126 LOG_DEBUG_FMT("Disconnecting incoming connection {}", id);
127 parent.unassociated_incoming.erase(id);
128
129 if (node_id.has_value())
130 {
131 parent.remove_connection(node_id.value());
132 }
133 }
134
135 void associate_incoming(const ccf::NodeId& n) override
136 {
137 node_id = n;
138
139 const auto unassociated = parent.unassociated_incoming.find(id);
140 CCF_ASSERT_FMT(
141 unassociated != parent.unassociated_incoming.end(),
142 "Associating node {} with incoming ID {}, but have already forgotten "
143 "the incoming connection",
144 n,
145 id);
146
147 // Always prefer this (probably) newer connection. Pathological case is
148 // where both nodes open outgoings to each other at the same time, both
149 // see the corresponding incoming connections and _drop_ their outgoing
150 // connections. Both have a useless incoming connection they think they
151 // can use. Assumption is that they progress at different rates, and one
152 // of them eventually spots the dead connection and opens a new one
153 // which succeeds.
154 parent.connections[n] = unassociated->second;
155 parent.unassociated_incoming.erase(unassociated);
156
157 LOG_DEBUG_FMT(
158 "Node incoming connection ({}) associated with {}", id, n);
159 }
160 };

288 DISPATCHER_SET_MESSAGE_HANDLER(
289 disp, ccf::node_outbound, [this](const uint8_t* data, size_t size) {
290 // Read piece-by-piece rather than all at once
291 ccf::NodeId to = serialized::read<ccf::NodeId::Value>(data, size);
292
293 TCP outbound_connection = nullptr;
294 {
295 const auto connection_it = connections.find(to);
296 if (connection_it == connections.end())
297 {
298 const auto address_it = node_addresses.find(to);
299 if (address_it == node_addresses.end())
300 {
301 LOG_FAIL_FMT("Ignoring node_outbound to unknown node {}", to);
302 return;
303 }
304
305 const auto& [host, service] = address_it->second;
306 outbound_connection = create_connection(to, host, service);
307 if (outbound_connection.is_null())
308 {
309 LOG_FAIL_FMT(
310 "Unable to connect to {}, dropping outbound message message",
311 to);
312 return;
313 }
314 }
315 else
316 {
317 outbound_connection = connection_it->second;
318 }
319 }
320
321 // Rather than reading and reserialising, use the msg_type and from_id
322 // that are already serialised on the ringbuffer
323 auto data_to_send = data;
324 auto size_to_send = size;
325
326 // If the message is a consensus append entries message, affix the
327 // corresponding ledger entries
328 auto msg_type = serialized::read<ccf::NodeMsgType>(data, size);
329 serialized::read<ccf::NodeId::Value>(data, size); // Ignore from_id
330 if (
331 msg_type == ccf::NodeMsgType::consensus_msg &&
332 (serialized::read<aft::RaftMsgType>(data, size) ==
333 aft::raft_append_entries))
334 {
335 // Parse the indices to be sent to the recipient.
336 const auto& ae =
337 serialized::overlay<consensus::AppendEntriesIndex>(data, size);
338
339 // Find the total frame size, and write it along with the header.
340 uint32_t frame = (uint32_t)size_to_send;
341 std::optional<std::vector<uint8_t>> framed_entries = std::nullopt;
342
343 framed_entries = ledger.read_entries(ae.prev_idx + 1, ae.idx);
344 if (framed_entries.has_value())
345 {
346 frame += (uint32_t)framed_entries->size();
347 outbound_connection->write(sizeof(uint32_t), (uint8_t*)&frame);
348 outbound_connection->write(size_to_send, data_to_send);
349
350 frame = (uint32_t)framed_entries->size();
351 outbound_connection->write(frame, framed_entries->data());
352 }
353 else
354 {
355 // Header-only AE
356 outbound_connection->write(sizeof(uint32_t), (uint8_t*)&frame);
357 outbound_connection->write(size_to_send, data_to_send);
358 }
359
360 LOG_DEBUG_FMT(
361 "send AE to node {} [{}]: {}, {}",
362 to,
363 frame,
364 ae.idx,
365 ae.prev_idx);
366 }
367 else
368 {
369 // Write as framed data to the recipient.
370 uint32_t frame = (uint32_t)size_to_send;
371
372 LOG_DEBUG_FMT("node send to {} [{}]", to, frame);
373
374 outbound_connection->write(sizeof(uint32_t), (uint8_t*)&frame);
375 outbound_connection->write(size_to_send, data_to_send);
376 }
377 });
378 }

381 TCP create_connection(
382 const ccf::NodeId& node_id,
383 const std::string& host,
384 const std::string& service)
385 {
386 auto s = TCP(true, client_connection_timeout);
387 s->set_behaviour(std::make_unique<NodeOutgoingBehaviour>(*this, node_id));
388
389 if (!s->connect(host, service, client_interface))
390 {
391 LOG_FAIL_FMT(
392 "Failed to connect to {} on {}:{}", node_id, host, service);
393 return nullptr;
394 }
395
396 connections.emplace(node_id, s);
397 LOG_DEBUG_FMT(
398 "Added node connection with {} ({}:{})", node_id, host, service);
399
400 return s;
401 }
402
403 bool remove_connection(const ccf::NodeId& node)
404 {
405 if (connections.erase(node) < 1)
406 {
407 LOG_DEBUG_FMT("Cannot remove node connection {}: does not exist", node);
408 return false;
409 }
410
411 LOG_DEBUG_FMT("Removed node connection with {}", node);
412 return true;
413 }
414
415 size_t get_next_id()
416 {
417 auto id = next_id++;
418
419 while (unassociated_incoming.find(id) != unassociated_incoming.end())
420 {
421 id = next_id++;

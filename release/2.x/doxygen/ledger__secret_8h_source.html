25 const crypto::HashBytes& get_commit_secret()

82 inline std::vector<uint8_t> decrypt_previous_ledger_secret_raw(
83 const LedgerSecretPtr& ledger_secret,
84 const std::vector<uint8_t>& encrypted_previous_secret_raw)
85 {
86 crypto::GcmCipher encrypted_ls;
87 encrypted_ls.deserialise(encrypted_previous_secret_raw);
88 std::vector<uint8_t> decrypted_ls_raw;
89
90 if (!ledger_secret->key->decrypt(
91 encrypted_ls.hdr.get_iv(),
92 encrypted_ls.hdr.tag,
93 encrypted_ls.cipher,
94 {},
95 decrypted_ls_raw))
96 {
97 throw std::logic_error("Decryption of previous ledger secret failed");
98 }
99
100 return decrypted_ls_raw;
101 }

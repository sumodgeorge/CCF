58 struct TxID
59 {
60 Term term = 0;
61 Version version = 0;
62
63 TxID() = default;
64 TxID(Term t, Version v) : term(t), version(v) {}
65
66 // Would like to remove these duplicate types, but for now we just do free
67 // conversion
68 TxID(const ccf::TxID& other) : term(other.view), version(other.seqno) {}
69
70 operator ccf::TxID() const
71 {
72 return {term, version};
73 }
74
75 bool operator==(const TxID& other) const
76 {
77 return term == other.term && version == other.version;
78 }
79

81 {
82 return fmt::format("{}.{}", term, version);
83 }

312 constexpr auto public_domain_prefix = "public:";
313
314 static inline SecurityDomain get_security_domain(const std::string& name)
315 {
316 if (nonstd::starts_with(name, public_domain_prefix))
317 {
318 return SecurityDomain::PUBLIC;
319 }
320
321 return SecurityDomain::PRIVATE;
322 }
323
324 static inline std::pair<SecurityDomain, AccessCategory> parse_map_name(
325 const std::string& name)
326 {
327 constexpr auto internal_category_prefix = "ccf.internal.";
328 constexpr auto governance_category_prefix = "public:ccf.gov.";
329 constexpr auto reserved_category_prefix = "ccf.";
330
331 auto security_domain = SecurityDomain::PRIVATE;
332 const auto core_name = nonstd::remove_prefix(name, public_domain_prefix);
333 if (core_name != name)
334 {
335 security_domain = SecurityDomain::PUBLIC;
336 }
337
338 auto access_category = AccessCategory::APPLICATION;
339 if (nonstd::starts_with(core_name, internal_category_prefix))
340 {
341 access_category = AccessCategory::INTERNAL;
342 }
343 else if (nonstd::starts_with(name, governance_category_prefix))
344 {
345 access_category = AccessCategory::GOVERNANCE;
346 }
347 else if (nonstd::starts_with(core_name, reserved_category_prefix))
348 {
349 throw std::logic_error(fmt::format(
350 "Map name '{}' includes disallowed reserved prefix '{}'",
351 name,
352 reserved_category_prefix));
353 }
354
355 return {security_domain, access_category};
356 }

691 // Setting a short rollback is a work around that should be fixed
692 // shortly. In BFT mode when we deserialize and realize we need to
693 // create a new map we remember this. If we need to create the same
694 // map multiple times (for tx in the same group of append entries) the
695 // first create successes but the second fails because the map is
696 // already there. This works around the problem by stopping just
697 // before the 2nd create (which failed at this point) and when the
698 // primary resends the append entries we will succeed as the map is
699 // already there. This will only occur on BFT startup so not a perf
700 // problem but still need to be resolved.

702 // Thus, a large rollback is one which did not result from the map creating
703 // issue. https://github.com/microsoft/CCF/issues/2799

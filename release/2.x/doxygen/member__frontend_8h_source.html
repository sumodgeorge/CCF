83 bool set_js_app(kv::Tx& tx, const JsBundle& bundle)
84 {
85 std::string module_prefix = "/";
86 remove_modules(tx, module_prefix);
87 set_modules(tx, module_prefix, bundle.modules);
88
89 remove_endpoints(tx);
90
91 auto endpoints =
92 tx.rw<ccf::endpoints::EndpointsMap>(ccf::endpoints::Tables::ENDPOINTS);
93
94 for (auto& [url, endpoint] : bundle.metadata.endpoints)
95 {
96 for (auto& [method, info] : endpoint)
97 {
98 const std::string& js_module = info.js_module;
99 if (std::none_of(
100 bundle.modules.cbegin(),
101 bundle.modules.cend(),
102 [&js_module](const SetModule& item) {
103 return item.name == js_module;
104 }))
105 {
106 LOG_FAIL_FMT(
107 "{} {}: module '{}' not found in bundle",
108 method,
109 url,
110 info.js_module);
111 return false;
112 }
113 auto info_ = info;
114 info_.js_module = module_prefix + info_.js_module;
115 auto verb = nlohmann::json(method).get<RESTVerb>();
116 endpoints->put(
117 ccf::endpoints::EndpointKey{
118 nonstd::starts_with(url, "/") ? url : fmt::format("/{}", url),
119 verb},
120 info_);
121 }
122 }
123
124 return true;
125 }

130 remove_endpoints(tx);
131 return true;

211 ccf::jsgov::ProposalInfoSummary resolve_proposal(
212 kv::Tx& tx,
213 const ProposalId& proposal_id,
214 const std::vector<uint8_t>& proposal,
215 const std::string& constitution)
216 {
217 auto pi =
218 tx.rw<ccf::jsgov::ProposalInfoMap>(jsgov::Tables::PROPOSALS_INFO);
219 auto pi_ = pi->get(proposal_id);
220
221 std::vector<std::pair<MemberId, bool>> votes;
222 std::optional<ccf::jsgov::Votes> final_votes = std::nullopt;
223 std::optional<ccf::jsgov::VoteFailures> vote_failures = std::nullopt;
224 for (const auto& [mid, mb] : pi_->ballots)
225 {
226 js::Runtime rt;
227 js::Context context(rt);
228 rt.add_ccf_classdefs();
229 js::TxContext txctx{&tx, js::TxAccess::GOV_RO};
230 js::populate_global(
231 &txctx,
232 nullptr,
233 nullptr,
234 std::nullopt,
235 nullptr,
236 nullptr,
237 nullptr,
238 nullptr,
239 nullptr,
240 nullptr,
241 context);
242 auto ballot_func = context.function(
243 mb,
244 "vote",
245 fmt::format(
246 "public:ccf.gov.proposal_info[{}].ballots[{}]", proposal_id, mid));
247
248 std::vector<js::JSWrappedValue> argv = {
249 context.new_string_len((const char*)proposal.data(), proposal.size()),
250 context.new_string_len(
251 pi_->proposer_id.data(), pi_->proposer_id.size())};
252
253 auto val = context.call(ballot_func, argv);
254 if (!JS_IsException(val))
255 {
256 votes.emplace_back(mid, JS_ToBool(context, val));
257 }
258 else
259 {
260 if (!vote_failures.has_value())
261 {
262 vote_failures = ccf::jsgov::VoteFailures();
263 }
264 auto [reason, trace] = js::js_error_message(context);
265 vote_failures.value()[mid] = ccf::jsgov::Failure{reason, trace};
266 }
267 }
268
269 {
270 js::Runtime rt;
271 js::Context js_context(rt);
272 rt.add_ccf_classdefs();
273 js::TxContext txctx{&tx, js::TxAccess::GOV_RO};
274 js::populate_global(
275 &txctx,
276 nullptr,
277 nullptr,
278 std::nullopt,
279 nullptr,
280 nullptr,
281 nullptr,
282 nullptr,
283 nullptr,
284 nullptr,
285 js_context);
286 auto resolve_func = js_context.function(
287 constitution, "resolve", "public:ccf.gov.constitution[0]");
288
289 std::vector<js::JSWrappedValue> argv;
290 argv.push_back(js_context.new_string_len(
291 (const char*)proposal.data(), proposal.size()));
292
293 argv.push_back(js_context.new_string_len(
294 pi_->proposer_id.data(), pi_->proposer_id.size()));
295
296 auto vs = js_context.new_array();
297 size_t index = 0;
298 for (auto& [mid, vote] : votes)
299 {
300 auto v = JS_NewObject(js_context);
301 auto member_id = JS_NewStringLen(js_context, mid.data(), mid.size());
302 JS_DefinePropertyValueStr(
303 js_context, v, "member_id", member_id, JS_PROP_C_W_E);
304 auto vote_status = JS_NewBool(js_context, vote);
305 JS_DefinePropertyValueStr(
306 js_context, v, "vote", vote_status, JS_PROP_C_W_E);
307 JS_DefinePropertyValueUint32(
308 js_context, vs, index++, v, JS_PROP_C_W_E);
309 }
310 argv.push_back(vs);
311
312 auto val = js_context.call(resolve_func, argv);
313
314 std::optional<jsgov::Failure> failure = std::nullopt;
315 if (JS_IsException(val))
316 {
317 pi_.value().state = ProposalState::FAILED;
318 auto [reason, trace] = js::js_error_message(js_context);
319 failure = ccf::jsgov::Failure{
320 fmt::format("Failed to resolve(): {}", reason), trace};
321 }
322 else if (JS_IsString(val))
323 {
324 auto status = js_context.to_str(val).value_or("");
325 if (status == "Open")
326 {
327 pi_.value().state = ProposalState::OPEN;
328 }
329 else if (status == "Accepted")
330 {
331 pi_.value().state = ProposalState::ACCEPTED;
332 }
333 else if (status == "Withdrawn")
334 {
335 pi_.value().state = ProposalState::FAILED;
336 }
337 else if (status == "Rejected")
338 {
339 pi_.value().state = ProposalState::REJECTED;
340 }
341 else if (status == "Failed")
342 {
343 pi_.value().state = ProposalState::FAILED;
344 }
345 else if (status == "Dropped")
346 {
347 pi_.value().state = ProposalState::DROPPED;
348 }
349 else
350 {
351 pi_.value().state = ProposalState::FAILED;
352 failure = ccf::jsgov::Failure{
353 fmt::format(
354 "resolve() returned invalid status value: \"{}\"", status),
355 std::nullopt};
356 }
357 }
358 else
359 {
360 pi_.value().state = ProposalState::FAILED;
361 failure = ccf::jsgov::Failure{
362 "resolve() returned invalid status value", std::nullopt};
363 }
364
365 if (pi_.value().state != ProposalState::OPEN)
366 {
367 remove_all_other_non_open_proposals(tx, proposal_id);
368 final_votes = std::unordered_map<ccf::MemberId, bool>();
369 for (auto& [mid, vote] : votes)
370 {
371 final_votes.value()[mid] = vote;
372 }
373 if (pi_.value().state == ProposalState::ACCEPTED)
374 {
375 js::Runtime rt;
376 js::Context js_context(rt);
377 rt.add_ccf_classdefs();
378 js::TxContext txctx{&tx, js::TxAccess::GOV_RW};
379
380 auto gov_effects =
381 context.get_subsystem<AbstractGovernanceEffects>();
382 if (gov_effects == nullptr)
383 {
384 throw std::logic_error(
385 "Unexpected: Could not access GovEffects subsytem");
386 }
387
388 js::populate_global(
389 &txctx,
390 nullptr,
391 nullptr,
392 std::nullopt,
393 nullptr,
394 gov_effects.get(),
395 nullptr,
396 &network,
397 nullptr,
398 this,
399 js_context);
400 auto apply_func = js_context.function(
401 constitution, "apply", "public:ccf.gov.constitution[0]");
402
403 std::vector<js::JSWrappedValue> argv = {
404 js_context.new_string_len(
405 (const char*)proposal.data(), proposal.size()),
406 js_context.new_string_len(
407 proposal_id.c_str(), proposal_id.size())};
408
409 auto val = js_context.call(apply_func, argv);
410
411 if (JS_IsException(val))
412 {
413 pi_.value().state = ProposalState::FAILED;
414 auto [reason, trace] = js::js_error_message(js_context);
415 failure = ccf::jsgov::Failure{
416 fmt::format("Failed to apply(): {}", reason), trace};
417 }
418 }
419 }
420
421 return jsgov::ProposalInfoSummary{
422 proposal_id,
423 pi_->proposer_id,
424 pi_.value().state,
425 pi_.value().ballots.size(),
426 final_votes,
427 vote_failures,
428 failure};
429 }
430 }

529 auto ack = [this](auto& ctx, nlohmann::json&& params) {
530 const auto& caller_identity =
531 ctx.template get_caller<ccf::MemberSignatureAuthnIdentity>();
532 const auto& signed_request = caller_identity.signed_request;
533
534 auto mas = ctx.tx.rw(this->network.member_acks);
535 const auto ma = mas->get(caller_identity.member_id);
536 if (!ma)
537 {
538 return make_error(
539 HTTP_STATUS_FORBIDDEN,
540 ccf::errors::AuthorizationFailed,
541 fmt::format(
542 "No ACK record exists for caller {}.",
543 caller_identity.member_id));
544 }
545
546 const auto digest = params.get<StateDigest>();
547 if (ma->state_digest != digest.state_digest)
548 {
549 return make_error(
550 HTTP_STATUS_BAD_REQUEST,
551 ccf::errors::StateDigestMismatch,
552 "Submitted state digest is not valid.");
553 }
554
555 auto sig = ctx.tx.rw(this->network.signatures);
556 const auto s = sig->get();
557 if (!s)
558 {
559 mas->put(caller_identity.member_id, MemberAck({}, signed_request));
560 }
561 else
562 {
563 mas->put(
564 caller_identity.member_id, MemberAck(s->root, signed_request));
565 }
566
567 // update member status to ACTIVE
568 GenesisGenerator g(this->network, ctx.tx);
569 try
570 {
571 g.activate_member(caller_identity.member_id);
572 }
573 catch (const std::logic_error& e)
574 {
575 return make_error(
576 HTTP_STATUS_FORBIDDEN,
577 ccf::errors::AuthorizationFailed,
578 fmt::format("Error activating new member: {}", e.what()));
579 }
580
581 auto service_status = g.get_service_status();
582 if (!service_status.has_value())
583 {
584 return make_error(
585 HTTP_STATUS_INTERNAL_SERVER_ERROR,
586 ccf::errors::InternalError,
587 "No service currently available.");
588 }
589
590 auto members = ctx.tx.rw(this->network.member_info);
591 auto member_info = members->get(caller_identity.member_id);
592 if (
593 service_status.value() == ServiceStatus::OPEN &&
594 g.is_recovery_member(caller_identity.member_id))
595 {
596 // When the service is OPEN and the new active member is a recovery
597 // member, all recovery members are allocated new recovery shares
598 try
599 {
600 share_manager.shuffle_recovery_shares(ctx.tx);
601 }
602 catch (const std::logic_error& e)
603 {
604 return make_error(
605 HTTP_STATUS_INTERNAL_SERVER_ERROR,
606 ccf::errors::InternalError,
607 fmt::format("Error issuing new recovery shares: {}", e.what()));
608 }
609 }
610 return make_success();
611 };
612 make_endpoint("/ack", HTTP_POST, json_adapter(ack), member_sig_only)
613 .set_auto_schema<StateDigest, void>()
614 .install();

698 auto submit_recovery_share = [this](auto& ctx, nlohmann::json&& params) {
699 // Only active members can submit their shares for recovery
700 const auto member_id = get_caller_member_id(ctx);
701 if (!member_id.has_value())
702 {
703 return make_error(
704 HTTP_STATUS_FORBIDDEN,
705 ccf::errors::AuthorizationFailed,
706 "Member is unknown.");
707 }
708 if (!check_member_active(ctx.tx, member_id.value()))
709 {
710 return make_error(
711 HTTP_STATUS_FORBIDDEN,
712 errors::AuthorizationFailed,
713 "Member is not active.");
714 }
715
716 GenesisGenerator g(this->network, ctx.tx);
717 if (
718 g.get_service_status() != ServiceStatus::WAITING_FOR_RECOVERY_SHARES)
719 {
720 return make_error(
721 HTTP_STATUS_FORBIDDEN,
722 errors::ServiceNotWaitingForRecoveryShares,
723 "Service is not waiting for recovery shares.");
724 }
725
726 auto node_operation = context.get_subsystem<AbstractNodeOperation>();
727 if (node_operation == nullptr)
728 {
729 throw std::logic_error(
730 "Unexpected: Could not access NodeOperation subsystem");
731 }
732
733 if (node_operation->is_reading_private_ledger())
734 {
735 return make_error(
736 HTTP_STATUS_FORBIDDEN,
737 errors::NodeAlreadyRecovering,
738 "Node is already recovering private ledger.");
739 }
740
741 const auto in = params.get<SubmitRecoveryShare::In>();
742 auto raw_recovery_share = crypto::raw_from_b64(in.share);
743 OPENSSL_cleanse(const_cast<char*>(in.share.data()), in.share.size());
744
745 size_t submitted_shares_count = 0;
746 try
747 {
748 submitted_shares_count = share_manager.submit_recovery_share(
749 ctx.tx, member_id.value(), raw_recovery_share);
750 }
751 catch (const std::exception& e)
752 {
753 constexpr auto error_msg = "Error submitting recovery shares.";
754 LOG_FAIL_FMT(error_msg);
755 LOG_DEBUG_FMT("Error: {}", e.what());
756 return make_error(
757 HTTP_STATUS_INTERNAL_SERVER_ERROR,
758 errors::InternalError,
759 error_msg);
760 }
761 OPENSSL_cleanse(raw_recovery_share.data(), raw_recovery_share.size());
762
763 if (submitted_shares_count < g.get_recovery_threshold())
764 {
765 // The number of shares required to re-assemble the secret has not yet
766 // been reached
767 return make_success(SubmitRecoveryShare::Out{fmt::format(
768 "{}/{} recovery shares successfully submitted.",
769 submitted_shares_count,
770 g.get_recovery_threshold())});
771 }
772
773 LOG_DEBUG_FMT(
774 "Reached recovery threshold {}", g.get_recovery_threshold());
775
776 try
777 {
778 node_operation->initiate_private_recovery(ctx.tx);
779 }

56 void start()
57 {
58 auto refresh_msg = std::make_unique<threading::Tmsg<RefreshTimeMsg>>(
59 [](std::unique_ptr<threading::Tmsg<RefreshTimeMsg>> msg) {
60 if (!msg->data.self.consensus->can_replicate())
61 {
62 LOG_DEBUG_FMT(
63 "JWT key auto-refresh: Node is not primary, skipping");
64 }
65 else
66 {
67 msg->data.self.refresh_jwt_keys();
68 }
69 LOG_DEBUG_FMT(
70 "JWT key auto-refresh: Scheduling in {}s",
71 msg->data.self.refresh_interval_s);
72 auto delay = std::chrono::seconds(msg->data.self.refresh_interval_s);
73 threading::ThreadMessaging::thread_messaging.add_task_after(
74 std::move(msg), delay);
75 },
76 *this);
77
78 LOG_DEBUG_FMT(
79 "JWT key auto-refresh: Scheduling in {}s", refresh_interval_s);
80 auto delay = std::chrono::seconds(refresh_interval_s);
81 threading::ThreadMessaging::thread_messaging.add_task_after(
82 std::move(refresh_msg), delay);
83 }

101 LOG_DEBUG_FMT("JWT key one-off refresh: Scheduling without delay");
102 auto delay = std::chrono::seconds(0);
103 threading::ThreadMessaging::thread_messaging.add_task_after(
104 std::move(refresh_msg), delay);

107 template <typename T>
108 void send_refresh_jwt_keys(T msg)
109 {
110 auto body = serdes::pack(msg, serdes::Pack::Text);
111
112 http::Request request(fmt::format(
113 "/{}/{}",
114 ccf::get_actor_prefix(ccf::ActorsType::nodes),
115 "jwt_keys/refresh"));
116 request.set_header(
117 http::headers::CONTENT_TYPE, http::headervalues::contenttype::JSON);
118 request.set_body(&body);
119
120 auto packed = request.build_request();
121
122 auto node_session = std::make_shared<ccf::SessionContext>(
123 ccf::InvalidSessionId, node_cert.raw());
124 auto ctx = ccf::make_rpc_context(node_session, packed);
125
126 const auto actor_opt = http::extract_actor(*ctx);
127 if (!actor_opt.has_value())
128 {
129 throw std::logic_error("Unable to get actor");
130 }
131
132 const auto actor = rpc_map->resolve(actor_opt.value());
133 auto frontend_opt = this->rpc_map->find(actor);
134 if (!frontend_opt.has_value())
135 {
136 throw std::logic_error(
137 "RpcMap::find returned invalid (empty) frontend");
138 }
139 auto frontend = frontend_opt.value();
140 frontend->process(ctx);
141 }

143 void send_refresh_jwt_keys_error()

145 // A message that the endpoint fails to parse, leading to 500.
146 // This is done purely for exposing errors as endpoint metrics.

151 void handle_jwt_jwks_response(
152 const std::string& issuer,
153 http_status status,
154 std::vector<uint8_t>&& data)
155 {
156 if (status != HTTP_STATUS_OK)
157 {
158 LOG_FAIL_FMT(
159 "JWT key auto-refresh: Error while requesting JWKS: {} {}{}",
160 status,
161 http_status_str(status),
162 data.empty() ?
163 "" :
164 fmt::format(" '{}'", std::string(data.begin(), data.end())));
165 send_refresh_jwt_keys_error();
166 return;
167 }
168
169 LOG_DEBUG_FMT(
170 "JWT key auto-refresh: Received JWKS for issuer '{}'", issuer);
171
172 JsonWebKeySet jwks;
173 try
174 {
175 jwks = nlohmann::json::parse(data).get<JsonWebKeySet>();
176 }
177 catch (const std::exception& e)
178 {
179 LOG_FAIL_FMT(
180 "JWT key auto-refresh: Cannot parse JWKS for issuer '{}': {}",
181 issuer,
182 e.what());
183 send_refresh_jwt_keys_error();
184 return;
185 }
186
187 // call internal endpoint to update keys
188 auto msg = SetJwtPublicSigningKeys{issuer, jwks};
189 send_refresh_jwt_keys(msg);
190 }

192 void handle_jwt_metadata_response(
193 const std::string& issuer,
194 std::shared_ptr<tls::CA> ca,
195 http_status status,
196 std::vector<uint8_t>&& data)
197 {
198 if (status != HTTP_STATUS_OK)
199 {
200 LOG_FAIL_FMT(
201 "JWT key auto-refresh: Error while requesting OpenID metadata: {} "
202 "{}{}",
203 status,
204 http_status_str(status),
205 data.empty() ?
206 "" :
207 fmt::format(" '{}'", std::string(data.begin(), data.end())));
208 send_refresh_jwt_keys_error();
209 return;
210 }
211
212 LOG_DEBUG_FMT(
213 "JWT key auto-refresh: Received OpenID metadata for issuer '{}'",
214 issuer);
215
216 std::string jwks_url_str;
217 try
218 {
219 auto metadata = nlohmann::json::parse(data);
220 jwks_url_str = metadata.at("jwks_uri").get<std::string>();
221 }
222 catch (const std::exception& e)
223 {
224 LOG_FAIL_FMT(
225 "JWT key auto-refresh: Cannot parse OpenID metadata for issuer '{}': "
226 "{}",
227 issuer,
228 e.what());
229 send_refresh_jwt_keys_error();
230 return;
231 }
232 http::URL jwks_url;
233 try
234 {
235 jwks_url = http::parse_url_full(jwks_url_str);
236 }
237 catch (const std::invalid_argument& e)
238 {
239 LOG_FAIL_FMT(
240 "JWT key auto-refresh: Cannot parse jwks_uri for issuer '{}': {}",
241 issuer,
242 jwks_url_str);
243 send_refresh_jwt_keys_error();
244 return;
245 }
246 auto jwks_url_port = !jwks_url.port.empty() ? jwks_url.port : "443";
247
248 auto ca_cert = std::make_shared<tls::Cert>(
249 ca, std::nullopt, std::nullopt, jwks_url.host);
250
251 LOG_DEBUG_FMT(
252 "JWT key auto-refresh: Requesting JWKS at https://{}:{}{}",
253 jwks_url.host,
254 jwks_url_port,
255 jwks_url.path);
256 auto http_client = rpcsessions->create_client(ca_cert);
257 // Note: Connection errors are not signalled and hence not tracked in
258 // endpoint metrics currently.
259 http_client->connect(
260 std::string(jwks_url.host),
261 std::string(jwks_url_port),
262 [this, issuer](
263 http_status status, http::HeaderMap&&, std::vector<uint8_t>&& data) {
264 handle_jwt_jwks_response(issuer, status, std::move(data));
265 return true;
266 });
267 http::Request r(jwks_url.path, HTTP_GET);
268 r.set_header(http::headers::HOST, std::string(jwks_url.host));
269 http_client->send_request(r.build_request());
270 }

272 void refresh_jwt_keys()
273 {
274 auto tx = network.tables->create_read_only_tx();
275 auto jwt_issuers = tx.ro(network.jwt_issuers);
276 auto ca_cert_bundles = tx.ro(network.ca_cert_bundles);
277 jwt_issuers->foreach([this, &ca_cert_bundles](
278 const JwtIssuer& issuer,
279 const JwtIssuerMetadata& metadata) {
280 if (!metadata.auto_refresh)
281 {
282 LOG_DEBUG_FMT(
283 "JWT key auto-refresh: Skipping issuer '{}', auto-refresh is "
284 "disabled",
285 issuer);
286 return true;
287 }
288
289 // Increment attempts, only when auto-refresh is enabled.
290 attempts++;
291
292 LOG_DEBUG_FMT(
293 "JWT key auto-refresh: Refreshing keys for issuer '{}'", issuer);
294 auto& ca_cert_bundle_name = metadata.ca_cert_bundle_name.value();
295 auto ca_cert_bundle_pem = ca_cert_bundles->get(ca_cert_bundle_name);
296 if (!ca_cert_bundle_pem.has_value())
297 {
298 LOG_FAIL_FMT(
299 "JWT key auto-refresh: CA cert bundle with name '{}' for issuer "
300 "'{}' not "
301 "found",
302 ca_cert_bundle_name,
303 issuer);
304 send_refresh_jwt_keys_error();
305 return true;
306 }
307
308 auto metadata_url_str = issuer + "/.well-known/openid-configuration";
309 auto metadata_url = http::parse_url_full(metadata_url_str);
310 auto metadata_url_port =
311 !metadata_url.port.empty() ? metadata_url.port : "443";
312
313 auto ca = std::make_shared<tls::CA>(ca_cert_bundle_pem.value());
314 auto ca_cert = std::make_shared<tls::Cert>(
315 ca, std::nullopt, std::nullopt, metadata_url.host);
316
317 LOG_DEBUG_FMT(
318 "JWT key auto-refresh: Requesting OpenID metadata at https://{}:{}{}",
319 metadata_url.host,
320 metadata_url_port,
321 metadata_url.path);
322 auto http_client = rpcsessions->create_client(ca_cert);
323 // Note: Connection errors are not signalled and hence not tracked in
324 // endpoint metrics currently.
325 http_client->connect(
326 std::string(metadata_url.host),
327 std::string(metadata_url_port),
328 [this, issuer, ca](
329 http_status status,
330 http::HeaderMap&&,
331 std::vector<uint8_t>&& data) {
332 handle_jwt_metadata_response(issuer, ca, status, std::move(data));
333 return true;
334 });
335 http::Request r(metadata_url.path, HTTP_GET);
336 r.set_header(http::headers::HOST, std::string(metadata_url.host));
337 http_client->send_request(r.build_request());
338 return true;
339 });
340 }

342 // Returns a copy of the current attempts
343 size_t get_attempts() const
344 {
345 return attempts.load();
346 }

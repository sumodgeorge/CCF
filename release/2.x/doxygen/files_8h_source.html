42 std::vector<uint8_t> slurp(const std::string& file, bool optional = false)
43 {
44 std::ifstream f(file, std::ios::binary | std::ios::ate);
45
46 if (!f)
47 {
48 if (optional)
49 {
50 return {};
51 }
52 else
53 {

55 exit(-1);

59 auto size = f.tellg();

65 if (!optional && !f)

69 }
